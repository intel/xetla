104 __XETLA_API KERNEL_FUNC void operator()(arguments_t *args, uint32_t wg_idx,
105 uint32_t wg_idy, uint32_t sg_idx, uint32_t sg_idy,
106 uint32_t linear_idx) {
107 xetla_vector<uint64_t, 1> rand_offset_ptr_v
108 = xetla_load_global<uint64_t, 1, data_size::default_size,
109 cache_hint::cached, cache_hint::cached>(
110 args->rand_offset_ptr, 0);
111 uint32_t threshold = uint32_t(args->dropout_prob * float(4294967296));
112 mask_out_tile_t mask_out;
113 int start_m = wg_idy * wg_tile_m + sg_idy * sg_tile_m;
114 int start_n = wg_idx * wg_tile_n + sg_idx * sg_tile_n;
115 mask_out_payload_t mask_out_payload(args->mask_ptr, args->matrix_n,
116 args->matrix_m, args->mask_ld, start_n, start_m);
117 static constexpr uint32_t random_len = 4 * random_simd;
118 xetla_rand_t<random_simd> rand_gen;
119 rand_gen.init(args->rand_seed, linear_idx, rand_offset_ptr_v[0]);
120
121 xetla_vector<dtype_mask, tile_size> mask;
122#pragma unroll
123 for (int i = 0; i < tile_size / random_len; i++) {
124 auto mask_sub = mask.xetla_select<random_len, 1>(i * random_len);
125 xetla_vector<uint32_t, random_len> rand_val = rand_gen.rand();
126 xetla_mask<random_len> mask_flag = rand_val < threshold;
127 mask_sub.xetla_merge(1, 0, mask_flag);
128 }
129 if constexpr (tile_size % random_len != 0) {
130 constexpr uint32_t remain_len = tile_size % random_len;
131 constexpr uint32_t remain_start
132 = tile_size / random_len * random_len;
133 auto mask_sub = mask.xetla_select<remain_len, 1>(remain_start);
134 // drop, still generate random_len
135 xetla_vector<uint32_t, random_len> rand_val = rand_gen.rand();
136 xetla_mask<random_len> mask_flag = rand_val < threshold;
137 mask_sub.xetla_merge(
138 1, 0, mask_flag.xetla_select<remain_len, 1>(0));
139 }
140 mask_out.reg = mask;
141 subgroup::tile_store<cache_hint::uncached>(mask_out, mask_out_payload);
142 }
143};
