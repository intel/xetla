270 __XETLA_API KERNEL_FUNC void operator()(xetla_exec_item<3> &ei,
271 const arguments_t &args, uint32_t slm_base = 0,
272 uint32_t nbarrier_base = 0) {
273 // set up workgroup level coordinates and boundaries
274 int start_n = ei.get_group(2) * wg_tile_n;
275 int start_m = ei.get_group(1) * wg_tile_m;
276 int start_k = 0;
277 uint32_t wg_tile_k = args.matrix_k;
278 uint32_t boundary_n = (start_n + wg_tile_n) > args.matrix_n
279 ? args.matrix_n
280 : (start_n + wg_tile_n);
281 uint32_t boundary_m = (start_m + wg_tile_m) > args.matrix_m
282 ? args.matrix_m
283 : (start_m + wg_tile_m);
284 uint32_t boundary_k = wg_tile_k;
285
286 uint32_t brgemm_slm_base = slm_base;
287 uint32_t brgemm_nbarr_base = nbarrier_base;
288 uint32_t epilogue_slm_base = brgemm_slm_base + brgemm_t::slm_size;
289 uint32_t epilogue_nbarr_base
290 = brgemm_nbarr_base + brgemm_t::barrier_count;
291
292 // set up arguments
293 work_group_t g;
294 g.init(ei.get_local_linear_id());
295 mem_desc_a_t mem_desc_a;
296 mem_desc_b_t mem_desc_b;
297 mem_desc_c_t mem_desc_c;
298 //setup for matA
299 if constexpr (mem_desc_a_t::is_local) {
300 mem_desc_a.init(
301 args.matA_base, {wg_tile_k, wg_tile_m, wg_tile_k}, {0, 0});
302 } else {
303 mem_desc_a.init(args.matA_base,
304 {boundary_k, boundary_m, args.matA_ld}, {start_k, start_m});
305 }
306 //setup for matB
307 if constexpr (mem_desc_b_t::is_local) {
308 mem_desc_b.init(
309 args.matB_base, {wg_tile_n, wg_tile_k, wg_tile_n}, {0, 0});
310 } else {
311 mem_desc_b.init(args.matB_base,
312 {boundary_n, boundary_k, args.matB_ld}, {start_n, start_k});
313 }
314 //setup for matC
315 if constexpr (mem_desc_c_t::is_local) {
316 mem_desc_c.init(
317 args.matC_base, {wg_tile_n, wg_tile_m, wg_tile_n}, {0, 0});
318 } else {
319 mem_desc_c.init(args.matC_base,
320 {boundary_n, boundary_m, args.matC_ld}, {start_n, start_m});
321 }
322 uint32_t inner_loop_count = (wg_tile_k + k_stride - 1) / k_stride;
323 brgemm_args_t brgemm_args(mem_desc_a, mem_desc_b, inner_loop_count);
324 brgemm_t brgemm;
325 epilogue_t epilogue;
326
327 matAcc_t matAcc(0);
328 brgemm(g, matAcc, brgemm_args, brgemm_slm_base, brgemm_nbarr_base);
329 epilogue(g, matAcc, mem_desc_c, args.epilogue_args, epilogue_slm_base,
330 epilogue_nbarr_base);
331 }
332};
