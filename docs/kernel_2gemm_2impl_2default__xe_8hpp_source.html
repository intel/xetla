176 uint32_t local_range_m = (wg_tile_m + sg_tile_m - 1) / sg_tile_m;

214 __XETLA_API KERNEL_FUNC void operator()(xetla_exec_item<3> &ei,
215 arguments_t &args, uint32_t slm_base = 0,
216 uint32_t nbarrier_base = 0) {
217 // set up workgroup level coordinates and boundaries
218 int start_n = ei.get_group(2) * wg_tile_n;
219 int start_m = ei.get_group(1) * wg_tile_m;
220 int start_k = 0;
221 uint32_t wg_tile_k = args.matrix_k;
222 uint32_t boundary_n = (start_n + wg_tile_n) > args.matrix_n
223 ? args.matrix_n
224 : (start_n + wg_tile_n);
225 uint32_t boundary_m = (start_m + wg_tile_m) > args.matrix_m
226 ? args.matrix_m
227 : (start_m + wg_tile_m);
228 uint32_t boundary_k = wg_tile_k;
229
230 uint32_t brgemm_slm_base = slm_base;
231 uint32_t brgemm_nbarr_base = nbarrier_base;
232 uint32_t epilogue_slm_base = brgemm_slm_base + brgemm_t::slm_size;
233 uint32_t epilogue_nbarr_base
234 = brgemm_nbarr_base + brgemm_t::barrier_count;
235
236 // set up arguments
237 work_group_t g;
238 g.init(ei.get_local_linear_id());
239 mem_desc_a_t mem_desc_a;
240 mem_desc_b_t mem_desc_b;
241 mem_desc_c_t mem_desc_c;
242 //setup for matA
243 if constexpr (mem_desc_a_t::is_local) {
244 mem_desc_a.init(
245 args.matA_base, {wg_tile_k, wg_tile_m, wg_tile_k}, {0, 0});
246 } else {
247 mem_desc_a.init(args.matA_base,
248 {boundary_k, boundary_m, args.matA_ld}, {start_k, start_m});
249 }
250 //setup for matB
251 if constexpr (mem_desc_b_t::is_local) {
252 mem_desc_b.init(
253 args.matB_base, {wg_tile_n, wg_tile_k, wg_tile_n}, {0, 0});
254 } else {
255 mem_desc_b.init(args.matB_base,
256 {boundary_n, boundary_k, args.matB_ld}, {start_n, start_k});
257 }
258 //setup for matC
259 if constexpr (mem_desc_c_t::is_local) {
260 mem_desc_c.init(
261 args.matC_base, {wg_tile_n, wg_tile_m, wg_tile_n}, {0, 0});
262 } else {
263 mem_desc_c.init(args.matC_base,
264 {boundary_n, boundary_m, args.matC_ld}, {start_n, start_m});
265 }
266 uint32_t inner_loop_count = (wg_tile_k + accum_step - 1) / accum_step;
267 brgemm_args_t brgemm_args(mem_desc_a, mem_desc_b, inner_loop_count);
268 brgemm_t brgemm;
269 epilogue_t epilogue;
270
271 matAcc_t matAcc(0);
272 brgemm(g, matAcc, brgemm_args, brgemm_slm_base, brgemm_nbarr_base);
273 epilogue(g, matAcc, mem_desc_c, args.epilogue_args, epilogue_slm_base,
274 epilogue_nbarr_base);
275 }
