60 inline KERNEL_FUNC xetla_vector<T, N> operator()(
61 xetla_vector<T, N * SZ> buffer) {
62 local_st_t local_st;
63 local_st_payload_t local_st_payload;
64 xetla_vector<T, N> ret = sg_reduce(buffer);
65 local_st.reg = ret;
66 local_st_payload.init(slm_base, N_SG * N, 1, N_SG * N, sg_id * N, 0);
67 subgroup::tile_store(local_st, local_st_payload);
68 xetla_fence<memory_kind::shared_local>();
69 nbarrier.arrive();
70 nbarrier.wait();
71 if constexpr (is_all_reduce) {
72 local_ld_t local_ld;
73 local_ld_payload_t local_ld_payload(
74 slm_base, N_SG * N, 1, N_SG * N, 0, 0);
75 subgroup::tile_load(local_ld, local_ld_payload);
76 ret = recur_row_reduce<Op, T, N, N_SG>(local_ld.reg);
77 } else {
78 if (sg_id == 0) {
79 local_ld_t local_ld;
80 local_ld_payload_t local_ld_payload;
81 local_ld_payload.init(slm_base, N_SG * N, 1, N_SG * N, 0, 0);

90template <typename T, uint32_t SZ, uint32_t N, reduce_op Op, bool is_all_reduce>
91struct group_reduce_t<T, SZ, N, Op, 1, is_all_reduce, gpu_arch::Xe> {

98 inline KERNEL_FUNC xetla_vector<T, N> operator()(
99 xetla_vector<T, N * SZ> buffer) {
100 auto buffer_2d = buffer.xetla_format<T, N, SZ>();
101 xetla_vector<T, N> ret;
102#pragma unroll
103 for (int i = 0; i < N; i++) {
104 ret[i] = xetla_reduce<T, T, SZ, Op>(buffer_2d.row(i));
105 }
106 return ret;
107 }
