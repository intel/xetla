46template <cache_hint L1H, cache_hint L3H>
47constexpr uint32_t get_load_cache_hint_code() {
48 check_lsc_cache_hint<lsc_action::load, L1H, L3H>();
49 if (L1H == cache_hint::none && L3H == cache_hint::none) {
50 return 0;
51 } else if (L3H == cache_hint::uncached) {
52 if (L1H == cache_hint::uncached) { return 1; }
53 if (L1H == cache_hint::cached) { return 3; }
54 if (L1H == cache_hint::streaming) { return 5; }
55 } else if (L3H == cache_hint::cached) {
56 if (L1H == cache_hint::uncached) { return 2; }
57 if (L1H == cache_hint::cached) { return 4; }
58 if (L1H == cache_hint::streaming) { return 6; }
59 if (L1H == cache_hint::read_invalidate) { return 7; }
60 }
61}
