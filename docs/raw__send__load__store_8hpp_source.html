302template <typename Ty, uint32_t N, cache_hint L1H = cache_hint::none,
303 cache_hint L3H = cache_hint::none, bool transpose = false,
304 bool transform = false>
305__XETLA_API xetla_vector<Ty, N> xetla_tload_global(xetla_tdescriptor tdesc) {
306
307 constexpr uint32_t numDst = 31 < ((N * sizeof(Ty) + 63) / 64)
308 ? 31
309 : ((N * sizeof(Ty) + 63) / 64);
310 uint32_t msg_desc = 3;
311 msg_desc |= (transform ? 1 : 0) << 7;
312 msg_desc |= detail::get_element_size_code<sizeof(Ty)>() << 9;
313 msg_desc |= (transpose ? 1 : 0) << 15;
314 msg_desc |= detail::get_load_cache_hint_code<L1H, L3H>() << 17;
315 msg_desc |= 1 << 25;
316 msg_desc |= numDst << 20;
317
318 constexpr uint32_t numSrc0 = 1;
319 constexpr uint32_t execSize = 0;
320 constexpr uint32_t sfid = 0xF;
321 constexpr uint32_t exDesc = 0;
322
323 constexpr uint32_t ret_N = (N * sizeof(Ty)) >= 32 ? N : 32 / sizeof(Ty);
324 xetla_vector<Ty, ret_N> ret;
325
326 xetla_raw_send<Ty, ret_N, uint32_t, 16, execSize, sfid, numSrc0, numDst>(
327 ret.xetla_format<native_type_t<Ty>>(), tdesc, exDesc, msg_desc);
328
329 return ret.xetla_select<N, 1>(0);
330}

343template <typename Ty, uint32_t N, cache_hint L1H = cache_hint::none,
344 cache_hint L3H = cache_hint::none>
345__XETLA_API void xetla_tstore_global(
346 xetla_tdescriptor tdesc, xetla_vector<Ty, N> data) {
347
348 uint32_t msg_desc = 7; // store operation
349 msg_desc |= detail::get_element_size_code<sizeof(Ty)>() << 9;
350 msg_desc |= detail::get_store_cache_hint_code<L1H, L3H>() << 17;
351 msg_desc |= 1 << 25;
352
353 constexpr uint32_t numSrc1 = (N * sizeof(Ty) + 63) / 64;
354 constexpr uint32_t numSrc0 = 1;
355 constexpr uint32_t execSize = 0;
356 constexpr uint32_t sfid = 0xF;
357 constexpr uint32_t exDesc = 0;
358
359 xetla_raw_send<uint32_t, 16, Ty, N, execSize, sfid, numSrc0, numSrc1>(
360 tdesc, data, exDesc, msg_desc);
361}

403template <typename Ty, uint32_t N, cache_hint L1H = cache_hint::none,
404 cache_hint L3H = cache_hint::none, atomic_op Op>
405__XETLA_API void xetla_tatomic_store_global(xetla_vector<uint64_t, N> address,
406 xetla_vector<Ty, N> data, xetla_mask<N> pred = 1) {
407
408 constexpr uint32_t numSrc0 = (N * sizeof(uint64_t) + 63) / 64;
409 constexpr uint32_t numSrc1 = (N * sizeof(Ty) + 63) / 64;
410 constexpr uint32_t num_dest = (N * sizeof(Ty) + 63) / 64;
411
412 uint32_t msg_desc = detail::get_atomic_opcode<Op>();
414 msg_desc |= 3 << 7;
415 msg_desc |= detail::get_element_size_code<sizeof(Ty)>() << 9;
416 msg_desc |= detail::get_atomic_cache_hint_code<L1H, L3H>() << 17;
417 msg_desc |= numSrc0 << 25;
418
419 constexpr uint32_t execSize = detail::get_execSize_code<N>();
420 constexpr uint32_t sfid = 0xF;
421 constexpr uint32_t exDesc = 0;
422
423 xetla_raw_send<uint64_t, N, Ty, N, execSize, sfid, numSrc0, numSrc1>(
424 address, data, exDesc, msg_desc, pred);
425}
