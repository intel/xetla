310 XETLA_MARKER(
311 "This release function will wait until all the r/w and nbarrier "
312 "id used in this brgemm have been committed. By default, it will "
313 "use barrier_id 0 to do the entire workgroup sync if wg_size > 1. "
314 "If you call this function, please set a free barrier id or make "
315 "sure barrier_id 0 is not being occupied and you need to allocate "
316 "one more barrier count in addition to the brgemm barrier counts.")
317 __XETLA_API static void release(uint8_t nbarrier_id = 0) {
318 static constexpr bool need_local_fence
319 = (mem_space_a == mem_space::local)
320 || (mem_space_b == mem_space::local);
321 if constexpr (need_local_fence) {
322 xetla_fence<memory_kind::shared_local>();
323 }
324 xetla_fence<memory_kind::untyped_global>();
325 static constexpr uint32_t wg_size = wg_size_x * wg_size_y;
326 if constexpr (wg_size > 1) {
327 xetla_nbarrier_t<wg_size, wg_size> nbarrier;
328 nbarrier.init_nbarrier(
329 nbarrier_id, nbarrier_role::producer_consumer);
330 nbarrier.arrive_wait();
331 }
332 }

341 __XETLA_API KERNEL_FUNC void operator()(work_group_t &g, matAcc_t &matAcc,
342 arguments_t args, uint32_t slm_base = 0,
343 uint32_t nbarrier_base = 0) {
344 int32_t sg_idx = g.get_id() % wg_size_x;
345 int32_t sg_idy = g.get_id() / wg_size_x;
346 update_sg_tile_tdesc(args, sg_idx, sg_idy);
347 pre_processing_t pre_processing;
348 matA_t matA;
349 matB_t matB;
350 // >>>>>>>>>>>>>>>>>> pre_processing init
351 pre_processing.init(g, args.pre_processing_args);
352 matA_payload_t matA_payload(args.matA_base_desc);
353 matB_payload_t matB_payload(args.matB_base_desc);
354 matA_prefetch_payload_t matA_prefetch_payload(
355 args.matA_base_desc, sg_idx);
356 matB_prefetch_payload_t matB_prefetch_payload(
357 args.matB_base_desc, sg_idy);
358 xetla_nbarrier_t<wg_size_x, wg_size_x> nbarrier_a;
359 nbarrier_a.init_nbarrier(
360 sg_idy + nbarrier_base, nbarrier_role::producer_consumer);
361 xetla_nbarrier_t<wg_size_y, wg_size_y> nbarrier_b;
362 nbarrier_b.init_nbarrier(sg_idx + barrier_count_y + nbarrier_base,
363 nbarrier_role::producer_consumer);
364
365#pragma unroll
366 for (int i = 0; i < stages; i++) {
367 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
368 matA_prefetch_payload);
369 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
370 matB_prefetch_payload);
371 matA_prefetch_payload.template update_tdesc<update_dir_a>(
372 matA_t::tile_size_x);
373 matB_prefetch_payload.template update_tdesc<update_dir_b>(
374 matB_t::tile_size_y);
375 }
376
377 for (int i = 0; i < args.inner_loop_count; i++) {
378 if constexpr (enable_periodic_sync) {
379 if ((i % sync_freq) == 0) {
380 if constexpr (wg_size_x > 1) { nbarrier_a.arrive(); }
381 if constexpr (wg_size_y > 1) { nbarrier_b.arrive(); }
382 }
383 }
384 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
385 matA, matA_payload);
386 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
387 matB, matB_payload);
388 SW_BARRIER();
389 if constexpr (stages != 0) {
390 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
391 matA_prefetch_payload);
392 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
393 matB_prefetch_payload);
394 }
395 SW_BARRIER();
396 matA_payload.template update_tdesc<update_dir_a>(
397 matA_t::tile_size_x);
398 matB_payload.template update_tdesc<update_dir_b>(
399 matB_t::tile_size_y);
400 if constexpr (stages != 0) {
401 matA_prefetch_payload.template update_tdesc<update_dir_a>(
402 matA_t::tile_size_x);
403 matB_prefetch_payload.template update_tdesc<update_dir_b>(
404 matB_t::tile_size_y);
405 }
406 SW_BARRIER();
407 matA_acc_t matA_acc;
408 matB_acc_t matB_acc;
409 if constexpr (is_vnni_tiled_a) { subgroup::vnni_reverse(matA); }
410 pre_processing(matA_acc, matB_acc, matA, matB);
411 SW_BARRIER();
412 tile_mma::mma(matA_acc, matB_acc, matAcc, matAcc);
413 SW_BARRIER();
414 if constexpr (enable_periodic_sync) {
415 if ((i % sync_freq) == 0) {
416 if constexpr (wg_size_x > 1) { nbarrier_a.wait(); }
417 if constexpr (wg_size_y > 1) { nbarrier_b.wait(); }
418 }
419 }
420 }
421 SW_BARRIER();
422 }
