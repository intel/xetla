241 XETLA_MARKER(
242 "This release function will wait until all the r/w and nbarrier "
243 "id used in this brgemm have been committed. By default, it will "
244 "use barrier_id 0 to do the entire workgroup sync if wg_size > 1. "
245 "If you call this function, please set a free barrier id or make "
246 "sure barrier_id 0 is not being occupied and you need to allocate "
247 "one more barrier count in addition to the brgemm barrier counts.")
248 __XETLA_API static void release(uint8_t nbarrier_id = 0) {
249 static constexpr bool need_local_fence
250 = (mem_space_a == mem_space::local)
251 || (mem_space_b == mem_space::local);
252 if constexpr (need_local_fence) {
253 xetla_fence<memory_kind::shared_local>();
254 }
255 xetla_fence<memory_kind::untyped_global>();
256 static constexpr uint32_t wg_size = wg_size_x * wg_size_y;
257 if constexpr (wg_size > 1) {
258 xetla_nbarrier_t<wg_size, wg_size> nbarrier;
259 nbarrier.init_nbarrier(
260 nbarrier_id, nbarrier_role::producer_consumer);
261 nbarrier.arrive_wait();
262 }
263 }

272 __XETLA_API KERNEL_FUNC void operator()(work_group_t &g, matAcc_t &matAcc,
273 arguments_t args, uint32_t slm_base = 0,
274 uint32_t nbarrier_base = 0) {
275 int32_t sg_idx = g.get_id() % wg_size_x;
276 int32_t sg_idy = g.get_id() / wg_size_x;
277
278 DEVICE_ASSERT(g.get_id() < (wg_size_x * wg_size_y),
279 "Thread id(%d) should less than wg_size(%d)", g.get_id(),
280 wg_size_x * wg_size_y);
281
282 update_sg_tile_tdesc(args, sg_idx, sg_idy);
283 pre_processing_t pre_processing;
284 matA_t matA;
285 matB_t matB;
286 // >>>>>>>>>>>>>>>>>> pre_processing init
287 pre_processing.init(g, args.pre_processing_args);
288 matA_payload_t matA_payload(args.matA_base_desc);
289 matB_payload_t matB_payload(args.matB_base_desc);
290 matA_prefetch_payload_t matA_prefetch_payload(
291 args.matA_base_desc, sg_idx);
292 matB_prefetch_payload_t matB_prefetch_payload(
293 args.matB_base_desc, sg_idy);
294 xetla_nbarrier_t<wg_size_x, wg_size_x> nbarrier_a;
295 nbarrier_a.init_nbarrier(
296 sg_idy + nbarrier_base, nbarrier_role::producer_consumer);
297 xetla_nbarrier_t<wg_size_y, wg_size_y> nbarrier_b;
298 nbarrier_b.init_nbarrier(sg_idx + barrier_count_y + nbarrier_base,
299 nbarrier_role::producer_consumer);
300
301#pragma unroll
302 for (int i = 0; i < stages; i++) {
303 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
304 matA_prefetch_payload);
305 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
306 matB_prefetch_payload);
307 matA_prefetch_payload.template update_tdesc<update_dir_a>(
308 matA_t::tile_size_x);
309 matB_prefetch_payload.template update_tdesc<update_dir_b>(
310 matB_t::tile_size_y);
311 }
312
313 for (int i = 0; i < args.inner_loop_count; i++) {
314 if constexpr (enable_periodic_sync) {
315 if ((i % sync_freq) == 0) {
316 if constexpr (wg_size_x > 1) { nbarrier_a.arrive(); }
317 if constexpr (wg_size_y > 1) { nbarrier_b.arrive(); }
318 }
319 }
320 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
321 matB, matB_payload);
322 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
323 matA, matA_payload);
324 if constexpr (stages != 0) {
325 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
326 matA_prefetch_payload);
327 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
328 matB_prefetch_payload);
329 }
330 SW_BARRIER();
331 matA_payload.template update_tdesc<update_dir_a>(
332 matA_t::tile_size_x);
333 matB_payload.template update_tdesc<update_dir_b>(
334 matB_t::tile_size_y);
335 if constexpr (stages != 0) {
336 matA_prefetch_payload.template update_tdesc<update_dir_a>(
337 matA_t::tile_size_x);
338 matB_prefetch_payload.template update_tdesc<update_dir_b>(
339 matB_t::tile_size_y);
340 }
341 SW_BARRIER();
342 matA_acc_t matA_acc;
343 matB_acc_t matB_acc;
344 subgroup::elemwise_cvt(matA_acc, matA);
345 subgroup::vnni_transform(matB_acc, matB);
346 pre_processing(matA_acc, matB_acc, matA, matB);
347 SW_BARRIER();
348 tile_mma::mma(matAcc, matAcc, matB_acc, matA_acc);
349 SW_BARRIER();
350 if constexpr (enable_periodic_sync) {
351 if ((i % sync_freq) == 0) {
352 if constexpr (wg_size_x > 1) { nbarrier_a.wait(); }
353 if constexpr (wg_size_y > 1) { nbarrier_b.wait(); }
354 }
355 }
356 }
357 SW_BARRIER();
358 }
