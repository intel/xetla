33 using matB_t = matB_t_;
34 using matSrc_t = matAcc_src_t_;
35 using matDst_t = matAcc_dst_t_;
36 using dtype_a = typename matA_t::dtype;
37 using dtype_b = typename matB_t::dtype;
38 using dtype_src = typename matSrc_t::dtype;
39 using dtype_dst = typename matDst_t::dtype;

41 using mma_attr = typename arch_attr_t<arch_tag_>::mma_attr;

43 static constexpr uint32_t a_tile_size_y = matA_t::tile_size_y;
44 static constexpr uint32_t a_tile_size_x = matA_t::tile_size_x;
45 static constexpr uint32_t a_tile_elems = matA_t::tile_elems;
46 static constexpr uint32_t a_block_size_y = matA_t::block_size_y;
47 static constexpr uint32_t a_block_size_x = matA_t::block_size_x;
48 static constexpr uint32_t a_block_elems = matA_t::block_elems;
49
50 static constexpr uint32_t b_tile_size_x = matB_t::tile_size_x;
51 static constexpr uint32_t b_tile_size_y = matB_t::tile_size_y;
52 static constexpr uint32_t b_tile_elems = matB_t::tile_elems;
53 static constexpr uint32_t b_block_size_x = matB_t::block_size_x;

68 static_assert(a_tile_size_x == b_tile_size_y,

85 static constexpr int32_t num_block_n = matDst_t::num_block_x;

87 static constexpr int32_t num_block_k = tile_size_k / block_size_k;
88
89 static constexpr int32_t mma_m = mma_attr::mma_m_in_elem;
90 static constexpr int32_t mma_k

92 static_assert(tile_size_m % mma_m == 0,

95 __XETLA_API static void mma(
96 matDst_t &dst, matSrc_t &src, matB_t &b, matA_t &a) {
97 constexpr int32_t a_mma_elems = mma_m * a_block_size_x;
98 constexpr int32_t c_mma_elems = mma_m * block_size_n;
99#pragma unroll
100 for (int j = 0; j < num_block_n; j++) {
101#pragma unroll
102 for (int i = 0; i < tile_size_m / block_size_m; i++) {
103 auto src_block = src.reg.xetla_select<block_elems, 1>(
104 (i * num_block_n + j) * block_elems);
105 auto dst_block = dst.reg.xetla_select<block_elems, 1>(
106 (i * num_block_n + j) * block_elems);
107#pragma unroll
108 for (int mma_i = 0; mma_i < block_size_m / mma_m; mma_i++) {
109 auto src_sub_blk = src_block.xetla_select<c_mma_elems, 1>(
110 mma_i * c_mma_elems);
111 auto dst_sub_blk = dst_block.xetla_select<c_mma_elems, 1>(
112 mma_i * c_mma_elems);
113 { //k=0
114 auto a_block = a.reg.xetla_select<a_block_elems, 1>(
115 (i * num_block_k) * a_block_elems);
116 auto a_sub_blk = a_block.xetla_select<a_mma_elems, 1>(
117 mma_i * a_mma_elems);
118 auto b_sub_blk = b.reg.xetla_select<b_block_elems, 1>(
119 j * b_block_elems);
120 dst_sub_blk = xetla_mma<
121 gpu::xetla::detail::mma_argument_type<
122 dtype_b>(),
123 gpu::xetla::detail::mma_argument_type<
124 dtype_a>(),
125 mma_k, mma_m, dtype_src, uint32_t, uint32_t,
126 c_mma_elems,
127 b_block_elems
128 / (sizeof(uint32_t) / sizeof(dtype_b)),
129 a_mma_elems
130 / (sizeof(uint32_t) / sizeof(dtype_a))>(
131 src_sub_blk, b_sub_blk.xetla_format<uint32_t>(),
132 a_sub_blk.xetla_format<uint32_t>());
133 }
134
135#pragma unroll
136 for (int k = 1; k < num_block_k; k++) {
137 auto a_block = a.reg.xetla_select<a_block_elems, 1>(
138 (i * num_block_k + k) * a_block_elems);
139 auto a_sub_blk = a_block.xetla_select<a_mma_elems, 1>(
140 mma_i * a_mma_elems);
141 auto b_sub_blk = b.reg.xetla_select<b_block_elems, 1>(
142 (j + k * num_block_n) * b_block_elems);
143 dst_sub_blk = xetla_mma<
144 gpu::xetla::detail::mma_argument_type<
145 dtype_b>(),
146 gpu::xetla::detail::mma_argument_type<
147 dtype_a>(),
148 mma_k, mma_m, dtype_src, uint32_t, uint32_t,
149 c_mma_elems,
150 b_block_elems
151 / (sizeof(uint32_t) / sizeof(dtype_b)),
152 a_mma_elems
153 / (sizeof(uint32_t) / sizeof(dtype_a))>(
154 dst_sub_blk, b_sub_blk.xetla_format<uint32_t>(),
155 a_sub_blk.xetla_format<uint32_t>());
156 }
157 }
158 }
159 if constexpr ((tile_size_m % block_size_m) != 0) {
160 constexpr uint32_t tail_block_size_m
161 = tile_size_m % block_size_m;
162 constexpr uint32_t tail_block_elems
163 = block_size_n * tail_block_size_m;
164 constexpr uint32_t a_tail_block_elems
165 = tail_block_size_m * a_block_size_x;
166 constexpr uint32_t tail_m_start
167 = tile_size_m / block_size_m * block_size_m;
168 constexpr uint32_t tail_elems_start
169 = tail_m_start * tile_size_n;
170 constexpr uint32_t a_tail_elems_start
171 = tail_m_start * a_tile_size_x;
172 auto src_block = src.reg.xetla_select<tail_block_elems, 1>(
173 tail_elems_start + j * tail_block_elems);
174 auto dst_block = dst.reg.xetla_select<tail_block_elems, 1>(
175 tail_elems_start + j * tail_block_elems);
176#pragma unroll
177 for (int mma_i = 0; mma_i < tail_block_size_m / mma_m;
178 mma_i++) {
179 auto src_sub_blk = src_block.xetla_select<c_mma_elems, 1>(
180 mma_i * c_mma_elems);
181 auto dst_sub_blk = dst_block.xetla_select<c_mma_elems, 1>(
182 mma_i * c_mma_elems);
183 { //k=0
184 auto a_block
185 = a.reg.xetla_select<a_tail_block_elems, 1>(
186 a_tail_elems_start);
187 auto a_sub_blk = a_block.xetla_select<a_mma_elems, 1>(
188 mma_i * a_mma_elems);
189 auto b_sub_blk = b.reg.xetla_select<b_block_elems, 1>(
190 j * b_block_elems);
191 dst_sub_blk = xetla_mma<
192 gpu::xetla::detail::mma_argument_type<
193 dtype_b>(),
194 gpu::xetla::detail::mma_argument_type<
195 dtype_a>(),
196 mma_k, mma_m, dtype_src, uint32_t, uint32_t,
197 c_mma_elems,
198 b_block_elems
199 / (sizeof(uint32_t) / sizeof(dtype_b)),
200 a_mma_elems
201 / (sizeof(uint32_t) / sizeof(dtype_a))>(
202 src_sub_blk, b_sub_blk.xetla_format<uint32_t>(),
203 a_sub_blk.xetla_format<uint32_t>());
204 }
205#pragma unroll
206 for (int k = 1; k < num_block_k; k++) {
207 auto a_block
208 = a.reg.xetla_select<a_tail_block_elems, 1>(
209 a_tail_elems_start
210 + k * a_tail_block_elems);
211 auto a_sub_blk = a_block.xetla_select<a_mma_elems, 1>(
212 mma_i * a_mma_elems);
213 auto b_sub_blk = b.reg.xetla_select<b_block_elems, 1>(
214 (j + k * num_block_n) * b_block_elems);
215 dst_sub_blk = xetla_mma<
216 gpu::xetla::detail::mma_argument_type<
217 dtype_b>(),
218 gpu::xetla::detail::mma_argument_type<
219 dtype_a>(),
220 mma_k, mma_m, dtype_src, uint32_t, uint32_t,
221 c_mma_elems,
222 b_block_elems
223 / (sizeof(uint32_t) / sizeof(dtype_b)),
224 a_mma_elems
225 / (sizeof(uint32_t) / sizeof(dtype_a))>(
226 dst_sub_blk, b_sub_blk.xetla_format<uint32_t>(),
227 a_sub_blk.xetla_format<uint32_t>());
228 }
229 }
230 }
231 }
232 if constexpr (num_block_k > 1) {
233 constexpr uint32_t last_uint16_idx
234 = tile_elems * sizeof(dtype_dst) / sizeof(uint16_t) - 1;
235 xetla_wait(dst.reg.xetla_format<uint16_t>()[last_uint16_idx]);
236 }
237 }
238};
