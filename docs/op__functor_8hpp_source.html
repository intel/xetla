57struct gelu_fwd_op_t {
58 struct arguments_t {};
59 template <typename matAcc_t, typename coord_t>
60 __XETLA_API KERNEL_FUNC void operator()(matAcc_t &matAcc,
61 const coord_t &coord, const arguments_t &args,
62 uint32_t slm_base = 0, uint32_t nbarrier_base = 0) {
63 using dtype = typename matAcc_t::dtype;
64 constexpr dtype C0 = 0.044715f;
65 constexpr dtype sqrt_two_over_pi = 0.79788458347320556640625f;
66 // total flag register
67 constexpr int elems = 8 * 16;
68 constexpr int rounds = matAcc_t::tile_elems / elems;
69#pragma unroll
70 for (int i = 0; i < rounds; ++i) {
71 auto sub_vec = matAcc.reg.xetla_select<elems, 1>(elems * i);
72 xetla_vector<dtype, elems> sub_vec_x = (sqrt_two_over_pi * sub_vec
73 * (1.f + C0 * sub_vec * sub_vec));
74 xetla_vector<dtype, elems> tanh_value
75 = xetla_tanh<dtype, elems>(sub_vec_x);
76 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);
77 }
78 constexpr int remained_elems = matAcc_t::tile_elems % elems;
79 if constexpr (remained_elems != 0) {
80 auto sub_vec = matAcc.reg.xetla_select<remained_elems, 1>(
81 elems * (matAcc_t::tile_elems / elems));
82 xetla_vector<dtype, remained_elems> sub_vec_x = (sqrt_two_over_pi
83 * sub_vec * (1.f + C0 * sub_vec * sub_vec));
84 xetla_vector<dtype, remained_elems> tanh_value
85 = xetla_tanh<dtype, remained_elems>(sub_vec_x);
86 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);
87 }
88 }

126 mem_desc_w_t mem_desc_w(args.base, args.shape, coord);
127 using bwd_w_tile_desc_t = tile_desc_t<block_size_x, block_size_y,
128 block_size_x, block_size_y, reg_layout::tiled>;
129 using bwd_w_tile_t = tile_t<dtype_out, bwd_w_tile_desc_t>;
130 using bwd_w_payload_t = mem_payload_t<dtype_out, bwd_w_tile_desc_t,

134 bwd_w_payload_t bwd_w_payload(mem_desc_w);
135 // start compute
136 constexpr dtype_acc c0 = 0.044715f;
137 constexpr dtype_acc d0 = 0.134145f;
138 constexpr dtype_acc sqrt_two_over_pi = 0.79788458347320556640625f;
139 constexpr uint32_t block_elems = matAcc_t::block_elems;
140 constexpr uint32_t num_block_x = matAcc_t::num_block_x;
141#pragma unroll
142 for (int i = 0; i < tile_size_y / block_size_y; ++i) {
143#pragma unroll
144 for (int j = 0; j < num_block_x; ++j) {

146 block_elems * (i * num_block_x + j));
147 xetla_vector<dtype_acc, block_elems> z
148 = xetla_tanh<dtype_acc, block_elems>(
149 sqrt_two_over_pi * (x + c0 * x * x * x));
150 xetla_vector<dtype_acc, block_elems> w = (0.5f * (1.f + z)
151 + 0.5f * x * (1.f - z * z)
152 * (sqrt_two_over_pi * (1.f + d0 * x * x)));
153 x = 0.5f * x * (1.f + z);
154 bwd_w.reg = xetla_cvt<dtype_out, dtype_acc, block_elems>(w);
155 tile_store<cache_hint::uncached>(bwd_w, bwd_w_payload);
156 bwd_w_payload.template update_tdesc<tdesc_update_dir::x_dir>(
157 block_size_x);
158 }

160 -1 * tile_size_x);
161 bwd_w_payload.template update_tdesc<tdesc_update_dir::y_dir>(
162 block_size_y);
163 }
164 if constexpr (tile_size_y % block_size_y != 0) {
165 constexpr uint32_t remain_size_y = tile_size_y % block_size_y;
166 constexpr uint32_t remain_y_start
167 = tile_size_y / block_size_y * block_size_y;
168 constexpr uint32_t remain_elems_start
169 = remain_y_start * tile_size_x;
170 constexpr uint32_t remain_block_elems
171 = remain_size_y * block_size_x;
172 // using remain_bwd_w_tile_t = xetla_tile_store_t<dtype_out,
173 // block_size_x, remain_size_y, block_size_x, remain_size_y,
174 // mem_desc_w_t::layout, mem_desc_w_t::space, store_op::normal,
175 // gpu_arch::Xe, reg_layout::tiled>;
176 using remain_bwd_w_tile_desc_t

179 using remain_bwd_w_tile_t

181 using remain_bwd_w_payload_t = mem_payload_t<dtype_out,
182 remain_bwd_w_tile_desc_t, msg_type::block_2d,
183 mem_desc_w_t::layout, mem_desc_w_t::space, gpu_arch::Xe>;
184
185 mem_desc_w.update_coord_y(remain_y_start);
186 remain_bwd_w_payload_t remain_bwd_w_payload(mem_desc_w);
187 remain_bwd_w_tile_t remain_bwd_w;
188#pragma unroll
189 for (int j = 0; j < num_block_x; ++j) {
190 auto x = matAcc.reg.xetla_select<remain_block_elems, 1>(
191 remain_elems_start + remain_block_elems * j);
192 xetla_vector<dtype_acc, remain_block_elems> z
193 = xetla_tanh<dtype_acc, remain_block_elems>(
194 sqrt_two_over_pi * (x + c0 * x * x * x));
195 xetla_vector<dtype_acc, remain_block_elems> w = (0.5f
196 * (1.f + z)
197 + 0.5f * x * (1.f - z * z)
198 * (sqrt_two_over_pi * (1.f + d0 * x * x)));
199 x = 0.5f * x * (1.f + z);
200 remain_bwd_w.reg
201 = xetla_cvt<dtype_out, dtype_acc, remain_block_elems>(
202 w);
203 tile_store<cache_hint::uncached>(
204 remain_bwd_w, remain_bwd_w_payload);
205 remain_bwd_w_payload
206 .template update_tdesc<tdesc_update_dir::x_dir>(
207 block_size_x);
208 }
209 }
