83struct sigmoid_op_t {
84 struct arguments_t {};
85 template <typename matAcc_t, typename coord_t>
86 __XETLA_API KERNEL_FUNC void operator()(matAcc_t &matAcc,
87 const coord_t &coord, uint32_t slm_base = 0,
88 uint32_t nbarrier_base = 0) {
89 constexpr int elems = matAcc_t::tile_desc::block_elems;
90 constexpr int rounds = matAcc_t::tile_desc::tile_elems / elems;
91 constexpr float one = 1.0f;
92#pragma unroll
93 for (int i = 0; i < rounds; ++i) {
94 auto sub_vec = matAcc.reg.xetla_select<elems, 1>(elems * i);
95 xetla_mask<elems> mask = sub_vec >= 10;
96 xetla_vector<typename matAcc_t::dtype, elems> temp_vec
97 = xetla_exp<typename matAcc_t::dtype, elems>(sub_vec);
98 xetla_vector<typename matAcc_t::dtype, elems> sigmoid_value
99 = temp_vec / (temp_vec + one);
100 sigmoid_value.xetla_merge(1, mask);
101 sub_vec = sigmoid_value;
102 }
103 constexpr int remained_elems = matAcc_t::tile_desc::tile_elems % elems;
104 if constexpr (remained_elems != 0) {
105 auto sub_vec = matAcc.reg.xetla_select<remained_elems, 1>(
106 elems * (matAcc_t::tile_elems / elems));
107 xetla_mask<remained_elems> mask = sub_vec >= 250;
108 xetla_vector<typename matAcc_t::dtype, remained_elems> temp_vec
109 = xetla_exp<typename matAcc_t::dtype, remained_elems>(
110 sub_vec);
111 xetla_vector<typename matAcc_t::dtype, remained_elems> sigmoid_value
112 = temp_vec / (temp_vec + one);
113 sigmoid_value.xetla_merge(1, mask);
114 sub_vec = sigmoid_value;
115 }

122struct gelu_fwd_op_t {
123 struct arguments_t {};
124 template <typename matAcc_t, typename coord_t>
125 __XETLA_API KERNEL_FUNC void operator()(matAcc_t &matAcc,
126 const coord_t &coord, const arguments_t &args,
127 uint32_t slm_base = 0, uint32_t nbarrier_base = 0) {
128 using dtype = typename matAcc_t::dtype;
129 constexpr dtype C0 = 0.044715f;
130 constexpr dtype sqrt_two_over_pi = 0.79788458347320556640625f;
131 // total flag register
132 constexpr int elems = 8 * 16;
133 constexpr int rounds = matAcc_t::tile_elems / elems;
134#pragma unroll

136 auto sub_vec = matAcc.reg.xetla_select<elems, 1>(elems * i);
137 xetla_vector<dtype, elems> sub_vec_x = (sqrt_two_over_pi * sub_vec
138 * (1.f + C0 * sub_vec * sub_vec));
139 xetla_vector<dtype, elems> tanh_value
140 = xetla_tanh<dtype, elems>(sub_vec_x);
141 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);

143 constexpr int remained_elems = matAcc_t::tile_elems % elems;
144 if constexpr (remained_elems != 0) {
145 auto sub_vec = matAcc.reg.xetla_select<remained_elems, 1>(
146 elems * (matAcc_t::tile_elems / elems));
147 xetla_vector<dtype, remained_elems> sub_vec_x = (sqrt_two_over_pi
148 * sub_vec * (1.f + C0 * sub_vec * sub_vec));
149 xetla_vector<dtype, remained_elems> tanh_value
150 = xetla_tanh<dtype, remained_elems>(sub_vec_x);
151 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);

224 }

252 for (int j = 0; j < num_block_x; ++j) {
253 auto x = matAcc.reg.xetla_select<remain_block_elems, 1>(
254 remain_elems_start + remain_block_elems * j);
