27// dim=1 : reduce along x dir;
28template <reduce_op reduce_kind, typename mat_t, typename dtype_acc, int dim>
29__XETLA_API typename std::enable_if_t<(dim == 1),
30 xetla_vector<typename mat_t::dtype, mat_t::tile_desc::tile_size_y>>
31tile_reduce(mat_t &src) {
32 static constexpr uint32_t tile_size_y = mat_t::tile_desc::tile_size_y;
33 static constexpr uint32_t tile_size_x = mat_t::tile_desc::tile_size_x;
34 static constexpr uint32_t tile_elems = mat_t::tile_desc::tile_elems;
35 static constexpr uint32_t block_size_y = mat_t::tile_desc::block_size_y;
36 static constexpr uint32_t block_size_x = mat_t::tile_desc::block_size_x;
37 static constexpr uint32_t block_elems = mat_t::tile_desc::block_elems;
38 static constexpr int32_t num_block_y = mat_t::tile_desc::num_block_y;
39 static constexpr int32_t num_block_x = mat_t::tile_desc::num_block_x;
40 using dtype = typename mat_t::dtype;
45 xetla_vector<dtype_acc, tile_size_y * block_size_x> acc;
46 auto acc_2d = acc.xetla_format<dtype_acc, tile_size_y, block_size_x>();
47#pragma unroll
48 for (int i = 0; i < tile_size_y / block_size_y; i++) {
49 //j=0, initial the buffer
50 {
51 auto src_reg = (src.reg).xetla_select<block_elems, 1>(
52 (i * num_block_x) * block_elems);
53 auto src_reg_acc
54 = xetla_cvt<dtype_acc, dtype, block_elems>(src_reg);
55 auto dst_reg_acc
56 = acc.xetla_select<block_elems, 1>(i * block_elems);
57 dst_reg_acc = src_reg_acc;
58 }
59#pragma unroll
60 for (int j = 1; j < num_block_x; j++) {
61 auto src_reg = (src.reg).xetla_select<block_elems, 1>(
62 (i * num_block_x + j) * block_elems);
63 auto src_reg_acc
64 = xetla_cvt<dtype_acc, dtype, block_elems>(src_reg);
65 auto dst_reg_acc
66 = acc.xetla_select<block_elems, 1>(i * block_elems);
67 dst_reg_acc = reduce_helper<reduce_kind, dtype_acc, block_elems>(
68 dst_reg_acc, src_reg_acc);
69 }
70 }
71
72 // process the tail
73 if constexpr ((tile_size_y % block_size_y) != 0) {
74 constexpr uint32_t tail_start_y
75 = tile_size_y / block_size_y * block_size_y;
76 constexpr uint32_t tail_size_y = tile_size_y % block_size_y;
77 constexpr uint32_t tail_block_elems = tail_size_y * block_size_x;
78 //j=0, initial the buffer
79 {
80 auto src_reg = (src.reg).xetla_select<tail_block_elems, 1>(
81 tail_start_y * tile_size_x);
82 auto src_reg_acc
83 = xetla_cvt<dtype_acc, dtype, tail_block_elems>(src_reg);
84 auto dst_reg_acc = acc.xetla_select<tail_block_elems, 1>(
85 tail_start_y * block_size_x);
86 dst_reg_acc = src_reg_acc;
87 }
88#pragma unroll
89 for (int j = 1; j < num_block_x; j++) {
90 auto src_reg = (src.reg).xetla_select<tail_block_elems, 1>(
91 tail_start_y * tile_size_x + j * tail_block_elems);
92 auto src_reg_acc
93 = xetla_cvt<dtype_acc, dtype, tail_block_elems>(src_reg);
94 auto dst_reg_acc = acc.xetla_select<tail_block_elems, 1>(
95 tail_start_y * block_size_x);
96 dst_reg_acc
97 = reduce_helper<reduce_kind, dtype_acc, tail_block_elems>(
98 dst_reg_acc, src_reg_acc);
99 }
100 }
101
102 xetla_vector<dtype_acc, tile_size_y> out = recur_col_reduce<reduce_kind,
103 dtype_acc, block_size_x, tile_size_y>(acc);
104
105 return xetla_cvt<dtype, dtype_acc, tile_size_y>(out);
106}

118template <typename T_dst, typename T_src, bool accumulate = true,

133 static constexpr int32_t num_block_x = T_src::tile_desc::num_block_x;
134 using dtype_dst = typename T_dst::dtype;
135 using dtype_src = typename T_src::dtype;
137 static constexpr uint32_t SIMD = 64 / sizeof(dtype_acc);
138 static constexpr uint32_t accum_len
139 = ((block_size_x % SIMD) && (sizeof(dtype_src) < 4)) == 0
140 ? SIMD
141 : block_size_x;
142 static constexpr uint32_t accum_elems = accum_len * block_size_y;
149 xetla_vector<dtype_acc, tile_size_x * num_acc> acc(0);
150 auto acc_2d = acc.xetla_format<dtype_acc, num_acc, tile_size_x>();
151 if constexpr (accumulate) {
152 acc_2d.row(0) = xetla_cvt<dtype_acc, dtype_dst, tile_size_x>(dst.reg);
153 }
154#pragma unroll
155 for (int i = 0; i < tile_size_y / block_size_y; i++) {
156#pragma unroll
157 for (int j = 0; j < num_block_x; j++) {
158 auto src_reg = (src.reg).xetla_select<block_elems, 1>(
159 (i * num_block_x + j) * block_elems);
160 auto src_reg_dtype_acc
161 = xetla_cvt<dtype_acc, dtype_src, block_elems>(src_reg);
162 auto src_reg_2d = src_reg_dtype_acc.xetla_format<dtype_acc,
163 block_size_y, block_size_x>();
164#pragma unroll
165 for (int row_i = 0; row_i < block_size_y; row_i += num_acc) {
166#pragma unroll
167 for (int acc_i = 0;
168 (acc_i < num_acc) && (row_i + acc_i < block_size_y);
169 acc_i++) {
170 auto acc_sub = acc.xetla_select<block_size_x, 1>(
171 acc_i * tile_size_x + j * block_size_x);
172#pragma unroll
173 for (int k = 0; k < block_size_x / accum_len; k++) {
174 acc_sub.xetla_select<accum_len, 1>(k * accum_len)
175 = acc_sub.xetla_select<accum_len, 1>(
176 k * accum_len)
177 + src_reg_2d.row(row_i + acc_i)
178 .xetla_select<accum_len, 1>(
179 k * accum_len);
180 }
181 }
182 }
183 }
184 }
185
186 // process the tail
187 if constexpr ((tile_size_y % block_size_y) != 0) {
188 constexpr uint32_t tail_start_y
189 = tile_size_y / block_size_y * block_size_y;
190 constexpr uint32_t tail_size_y = tile_size_y % block_size_y;
191 constexpr uint32_t tail_block_elems = tail_size_y * block_size_x;
192#pragma unroll
193 for (int j = 0; j < num_block_x; j++) {
194 auto src_reg = (src.reg).xetla_select<tail_block_elems, 1>(
195 tail_start_y * tile_size_x + j * tail_block_elems);
196 auto src_reg_dtype_acc
197 = xetla_cvt<dtype_acc, dtype_src, tail_block_elems>(
198 src_reg);
199 auto src_reg_2d = src_reg_dtype_acc.xetla_format<dtype_acc,
200 tail_size_y, block_size_x>();
201#pragma unroll
202 for (int row_i = 0; row_i < tail_size_y; row_i += num_acc) {
203#pragma unroll
204 for (int acc_i = 0;
205 (acc_i < num_acc) && (row_i + acc_i < tail_size_y);
206 acc_i++) {
207 auto acc_sub = acc.xetla_select<block_size_x, 1>(
208 acc_i * tile_size_x + j * block_size_x);
209#pragma unroll
210 for (int k = 0; k < block_size_x / accum_len; k++) {
211 acc_sub.xetla_select<accum_len, 1>(k * accum_len)
212 = acc_sub.xetla_select<accum_len, 1>(
213 k * accum_len)
214 + src_reg_2d.row(row_i + acc_i)
215 .xetla_select<accum_len, 1>(
216 k * accum_len);
217 }
218 }
219 }
220 }
221 }
222
223#pragma unroll
224 for (int i = 1; i < num_acc; i++) {
225 acc_2d.row(0) += acc_2d.row(i);
226 }
227
228 dst.reg = xetla_cvt<dtype_dst, dtype_acc, tile_size_x>(acc_2d.row(0));
