400template <typename T, int SZ>
401__XETLA_API xetla_vector<T, SZ> xetla_tanh(xetla_vector<T, SZ> src) {
402 static_assert(std::is_same<remove_const_t<T>, float>::value,
403 "Only support fp32! ");
404 constexpr uint32_t flag_elems = 8 * 16;
405 xetla_vector<T, SZ> ret;
406#pragma unroll
407 for (int i = 0; i < SZ / flag_elems; i++) {
408 auto src_sub = src.xetla_select<flag_elems, 1>(i * flag_elems);
409 auto ret_sub = ret.xetla_select<flag_elems, 1>(i * flag_elems);
410 xetla_mask<flag_elems> mask = src_sub >= 10;
411 xetla_vector<T, flag_elems> exp2x
412 = xetla_exp<T, flag_elems>(src_sub * 2.f);
413 ret_sub = (exp2x - 1.f) / (exp2x + 1.f);
414 xetla_vector<T, flag_elems> ones(1);
415 ret_sub.xetla_merge(ones, mask);
416 }
417 if constexpr (SZ % flag_elems != 0) {
418 constexpr uint32_t start_pos = SZ / flag_elems * flag_elems;
419 constexpr uint32_t remain_elems = SZ % flag_elems;
420
421 auto src_sub = src.xetla_select<remain_elems, 1>(start_pos);
422 auto ret_sub = ret.xetla_select<remain_elems, 1>(start_pos);
423 xetla_mask<remain_elems> mask = src_sub >= 10;
424 xetla_vector<T, remain_elems> exp2x
425 = xetla_exp<T, remain_elems>(src_sub * 2.f);
426 ret_sub = (exp2x - 1.f) / (exp2x + 1.f);
427 xetla_vector<T, remain_elems> ones(1);
428 ret_sub.xetla_merge(ones, mask);
429 }
430
431 return ret;
432}

439template <typename T>
440__XETLA_API T xetla_tanh(T src) {
441 static_assert(std::is_same<remove_const_t<T>, float>::value,
442 "Only support fp32! ");
443 T exp2x = xetla_exp<T>(src * 2.f);
444 T ret = (exp2x - 1.f) / (exp2x + 1.f);
445 return (src >= 10) ? 1 : ret;
446}

515template <typename T0, typename T1, int SZ, reduce_op BinaryOperation>
516__XETLA_API T0 xetla_reduce(xetla_vector<T1, SZ> v) {
517 if constexpr (BinaryOperation == reduce_op::sum) {
518 return __ESIMD_NS::detail::sum<T0, T1, SZ>(v);
519 } else if constexpr (BinaryOperation == reduce_op::prod) {
520 return __ESIMD_NS::detail::prod<T0, T1, SZ>(v);
521 } else if constexpr (BinaryOperation == reduce_op::min) {
522 return __ESIMD_NS::hmin<T0, T1, SZ>(v);
523 } else if constexpr (BinaryOperation == reduce_op::max) {
524 return __ESIMD_NS::hmax<T0, T1, SZ>(v);
525 }
526}

550template <typename T1, typename T0, int SZ,
551 typename Sat = xetla_saturation_off_tag>
552__XETLA_API xetla_vector<T1, SZ> xetla_add(
553 xetla_vector<T0, SZ> src0, xetla_vector<T0, SZ> src1, Sat sat = {}) {
554 static_assert(
555 !((is_internal_type<T0>::value) || (is_internal_type<T1>::value)),
556 "The internal types are not yet supported!");
558 xetla_vector<T0, SZ> temp = src0 + src1;
559 if constexpr (std::is_same_v<Sat, xetla_saturation_off_tag>)
560 return xetla_vector<T1, SZ>(temp);
561 else
562 return __ESIMD_NS::saturate<T1, T0, SZ>(temp);
563}
