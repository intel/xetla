99template <uint32_t remained_len, uint32_t base_len, process_flag flag,
100 cache_hint L1, cache_hint L3, typename payload_t, typename tile_t>
101__XETLA_API typename std::enable_if_t<base_len != 0
102 && payload_t::memory_space == mem_space::global>
103process_1d_tail(tile_t &tile, payload_t &payload, uint32_t offset) {
104 using dtype = typename payload_t::dtype;
105 using mem_dtype = typename payload_t::mem_dtype;
106 if constexpr (remained_len >= base_len) {
107 uint32_t address_offset = offset * sizeof(dtype);
108 auto reg_sub
109 = tile.reg.xetla_select<base_len * payload_t::scale_factor, 1>(
110 offset);
111 if constexpr (flag == process_flag::load) {
112 reg_sub.xetla_format<mem_dtype>() = xetla_load_global<mem_dtype,
113 base_len, data_size::default_size, L1, L3>(
114 payload.base_ptr, payload.base_offset + address_offset);
115 } else {
116 xetla_store_global<mem_dtype, base_len, data_size::default_size, L1,
117 L3>(payload.base_ptr, payload.base_offset + address_offset,
118 reg_sub.xetla_format<mem_dtype>());
119 }
120 process_1d_tail<remained_len - base_len, (base_len >> 1), flag, L1, L3>(
121 tile, payload, offset + base_len * payload_t::scale_factor);
122 } else {
123 process_1d_tail<remained_len, (base_len >> 1), flag, L1, L3>(
124 tile, payload, offset);
125 }
126}

128template <uint32_t remained_len, uint32_t base_len, process_flag flag,
129 cache_hint L1, cache_hint L3, typename payload_t, typename tile_t>
130__XETLA_API typename std::enable_if_t<base_len != 0
131 && payload_t::memory_space == mem_space::local>
132process_1d_tail(tile_t &tile, payload_t &payload, uint32_t offset) {
133 using dtype = typename payload_t::dtype;
134 using mem_dtype = typename payload_t::mem_dtype;
135 if constexpr (remained_len >= base_len) {
136 auto reg_sub
137 = tile.reg.xetla_select<base_len * payload_t::scale_factor, 1>(
138 offset);
139 uint32_t address_offset = offset * sizeof(typename tile_t::dtype);
140 if constexpr (flag == process_flag::load) {
141 uint32_t base_addr = payload.base_addr;
142 reg_sub.xetla_format<mem_dtype>() = xetla_load_local<mem_dtype,
143 base_len, data_size::default_size>(
144 base_addr + payload.base_offset + address_offset);
145 } else {
146 xetla_store_local<mem_dtype, base_len>(
147 payload.address + address_offset,
148 reg_sub.xetla_format<mem_dtype>());
149 }
150 process_1d_tail<remained_len - base_len, (base_len >> 1), flag, L1, L3,
151 payload_t, tile_t>(
152 tile, payload, offset + base_len * payload_t::scale_factor);
153 } else {
154 process_1d_tail<remained_len, (base_len >> 1), flag, L1, L3, payload_t,
155 tile_t>(tile, payload, offset);
156 }
157}

166template <uint32_t remained_len, uint32_t base_len, cache_hint L1,
167 cache_hint L3, typename payload_t>
168__XETLA_API typename std::enable_if_t<base_len >= 8> process_1d_tail(
169 payload_t &payload, uint32_t offset) {
170 using dtype = typename payload_t::dtype;
171 using prefetch_dtype = typename payload_t::prefetch_dtype;
172 if constexpr (remained_len >= base_len) {
173 uint32_t address_offset = offset * sizeof(dtype);
174 xetla_prefetch_global<prefetch_dtype, base_len, data_size::default_size,
175 L1, L3>(payload.base_ptr, payload.base_offset + address_offset);
176 process_1d_tail<remained_len - base_len, (base_len >> 1), L1, L3,
177 payload_t>(
178 payload, offset + base_len * payload_t::scale_factor);
179 } else {
180 process_1d_tail<remained_len, (base_len >> 1), L1, L3, payload_t>(
181 payload, offset);
182 }
183}

185template <uint32_t num_tdesc, uint32_t size_x, uint32_t size_y,
186 uint32_t scale_factor, uint8_t arr_len, bool trans>
187__XETLA_API static void reset_tile_desc_core(
188 xetla_matrix_ref<uint32_t, num_tdesc, 16> __REF__ payload_row) {
189#pragma unroll
190 for (int j = 0; j < num_tdesc; j++) {
191 constexpr uint8_t block_width
192 = trans ? (size_y / scale_factor) : size_x;
193 constexpr uint8_t block_height = trans ? size_x : size_y;
194 constexpr uint32_t block_widthx_widthy_arrlen = (block_width - 1)
195 | ((block_height - 1) << 8) | ((arr_len - 1) << 16);
196 gpu::xetla::detail::xetla_set_block_widthx_widthy_arrlen(
197 payload_row.row(j), block_widthx_widthy_arrlen);
198 }
199}
