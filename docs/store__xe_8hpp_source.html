29template <typename tile_t, typename payload_t>
30struct check_store_type {
31 static constexpr bool is_global_2d_xe
32 = (payload_t::memory_space == mem_space::global
33 && (payload_t::message_type == msg_type::block_2d)
34 && (payload_t::arch_tag == gpu_arch::Xe));
35
36 static constexpr bool is_global_block_1d_xe
37 = ((payload_t::memory_space == mem_space::global)
38 && (tile_t::tile_size_y == 1) && (tile_t::block_size_y == 1)
39 && (payload_t::message_type == msg_type::block_1d)
40 && (payload_t::arch_tag == gpu_arch::Xe));
41
42 static constexpr bool is_global_atomic_xe
43 = ((payload_t::memory_space == mem_space::global)
44 && (payload_t::message_type == msg_type::atomic_add)
45 && (payload_t::arch_tag == gpu_arch::Xe));
46
47 static constexpr bool is_local_scatter_xe = ((payload_t::memory_space
48 == mem_space::local)
49 && (payload_t::message_type == msg_type::scatter)
50 && (payload_t::arch_tag == gpu_arch::Xe)
51 && (payload_t::tile_desc::register_layout == reg_layout::tiled));
52
53 static constexpr bool is_local_scatter_vnni_col_xe
54 = ((payload_t::memory_space == mem_space::local)
55 && (payload_t::message_type == msg_type::scatter)
56 && (payload_t::arch_tag == gpu_arch::Xe)
57 && (payload_t::tile_desc::register_layout
58 == reg_layout::vnni_tiled_col_major));
59
60 static constexpr bool is_local_block_1d_xe = ((payload_t::memory_space
61 == mem_space::local)
62 && (payload_t::message_type == msg_type::block_1d)
63 && (payload_t::arch_tag == gpu_arch::Xe)
64 && (payload_t::tile_desc::register_layout == reg_layout::tiled));
65};
66
67template <typename tile_t, typename payload_t>
68inline constexpr void check_store_condition() {
69 using dtype = typename payload_t::dtype;
70 if constexpr (check_store_type<tile_t, payload_t>::is_global_2d_xe) {
71 using load_store_attr =
72 typename arch_attr_t<payload_t::arch_tag>::load_store_attr;
73
74 constexpr int32_t max_block_width
75 = load_store_attr::max_load_width_in_bytes / sizeof(dtype);
76 static_assert(
77 (max_block_width % payload_t::tile_desc::block_size_x) == 0,
78 "max_block_width should be a multiply of block size x.");
79
80 } else if constexpr (check_store_type<tile_t,
81 payload_t>::is_global_block_1d_xe) {
82
83 static_assert(sizeof(typename payload_t::mem_dtype) >= 4,
84 "store size should at least DW aligned");
85
86 } else if constexpr (check_store_type<tile_t,
87 payload_t>::is_global_atomic_xe) {
88 static_assert(std::is_same<remove_const_t<dtype>, uint32_t>::value
89 || std::is_same<remove_const_t<dtype>, uint64_t>::value
90 || std::is_same<remove_const_t<dtype>, int>::value
91 || std::is_same<remove_const_t<dtype>, float>::value
92 || std::is_same<remove_const_t<dtype>, double>::value,
93 "for global atomic add, we only support fp32 and fp64, "
94 "uin32_t, uint64_t, int");
95 static_assert(
96 ((payload_t::tile_bytes % payload_t::min_store_bytes) == 0
97 && (payload_t::block_bytes % payload_t::min_store_bytes)
98 == 0),
99 "currently, we are not able to handle the corner case");
100 static_assert((payload_t::num_channel_x > 0)
101 && (payload_t::num_channel_x <= payload_t::num_channel),
102 "The number of simd channel x should be greater than 0 and "
103 "less "
104 "than num_channel");
105 static_assert(sizeof(dtype) >= 4, "Only support DW and QW atomic add");
106
107 } else if constexpr (check_store_type<tile_t,
108 payload_t>::is_local_scatter_xe) {
109 static_assert(((payload_t::tile_bytes % payload_t::min_bytes) == 0
110 && (payload_t::block_bytes % payload_t::min_bytes)
111 == 0),
112 "currently, we are not able to handle the corner case");
113 static_assert((payload_t::num_channel_x > 0)
114 && (payload_t::num_channel_x <= payload_t::num_channel),
115 "The number of simd channel x should be greater than 0 and "
116 "less "
117 "than num_channel");
118
119 } else if constexpr (check_store_type<tile_t,
120 payload_t>::is_local_scatter_vnni_col_xe) {
121 static_assert(
122 ((payload_t::tile_bytes % payload_t::min_store_bytes) == 0
123 && (payload_t::block_bytes % payload_t::min_store_bytes)
124 == 0),
125 "currently, we are not able to handle the corner case");
126
127 static_assert((payload_t::num_channel_x > 0)
128 && (payload_t::num_channel_x <= payload_t::num_channel),
129 "The number of simd channel x should be greater than 0 and "
130 "less "
131 "than num_channel");
132 } else if constexpr (check_store_type<tile_t,
133 payload_t>::is_local_block_1d_xe) {
134 static_assert(sizeof(typename payload_t::mem_dtype) >= 4,
135 "store size should at least DW aligned");
136 }
137}

151template <cache_hint L1 = cache_hint::write_back,
152 cache_hint L3 = cache_hint::write_back, typename tile_t,
153 typename payload_t>
154__XETLA_API typename std::enable_if_t<
155 detail::check_store_type<tile_t, payload_t>::is_global_2d_xe>
156tile_store(tile_t &tile, payload_t &payload) {
157 detail::check_store_condition<tile_t, payload_t>();
158 using dtype = typename tile_t::dtype;
159 using tile_desc = typename tile_t::tile_desc;
160 static constexpr uint32_t tile_size_x = tile_desc::tile_size_x;
161 static constexpr uint32_t tile_size_y = tile_desc::tile_size_y;
162 static constexpr uint32_t block_size_x = tile_desc::block_size_x;
163 static constexpr uint32_t block_size_y = tile_desc::block_size_y;
164 static constexpr uint32_t remained_size_y = tile_desc::remained_size_y;
165
166 static constexpr uint32_t block_elems = tile_desc::block_elems;
167
168 static constexpr uint32_t num_block_x = tile_desc::num_block_x;
169 static constexpr uint32_t num_block_y = tile_desc::num_block_y;
170 static constexpr uint32_t num_block = tile_desc::num_block;
171
172 using load_store_attr =
173 typename arch_attr_t<payload_t::arch_tag>::load_store_attr;
174
175 static constexpr int32_t max_block_width
176 = load_store_attr::max_load_width_in_bytes / sizeof(dtype);
177 static constexpr int32_t max_store_block_height
178 = load_store_attr::max_store_height_in_elem;
179 static_assert((max_block_width % block_size_x) == 0,
180 "max_block_width should be a multiply of block size x.");
181 static constexpr uint32_t elems_per_CL
182 = load_store_attr::cache_line_size_in_bytes / sizeof(dtype);
183 static constexpr uint32_t st_blk_size_y
184 = block_size_y > max_store_block_height ? max_store_block_height
185 : block_size_y;
186 // to make sure full CL store
187 static constexpr uint32_t st_block_x
188 = (((tile_size_x % elems_per_CL) == 0) ? elems_per_CL
189 : block_size_x);
190
191 static constexpr uint8_t arr_len = st_block_x / block_size_x;
192
193 auto payload_2d = payload.payloads.xetla_format<uint32_t, num_block, 16>();
194 uint32_t base_offset_y = 0;
195#pragma unroll
196 for (int i = 0; i < num_block_y; ++i) {
197 constexpr uint32_t store_block_elems = block_elems * arr_len;
198 auto payload_row = payload_2d.xetla_select<num_block_x, 1, 16, 1>(
199 i * num_block_x, 0);
200 detail::reset_tile_desc_core<num_block_x, block_size_x * arr_len,
201 st_blk_size_y, 1, 1, false>(payload_row);
202 base_offset_y += block_size_y;
203#pragma unroll
204 for (int j = 0; j < num_block_x; j += arr_len) {
205 xetla_tdescriptor tdesc = payload_row.row(j);
206 auto reg_blk = tile.reg.xetla_select<store_block_elems, 1>(
207 (i * num_block_x + j) * block_elems);
208 xetla_vector<dtype, store_block_elems> combine_blk;
209 auto combine_blk_2d = combine_blk.xetla_format<native_type_t<dtype>,
210 block_size_y, block_size_x * arr_len>();
211#pragma unroll
212 for (int combine_i = 0; combine_i < arr_len; ++combine_i) {
213 combine_blk_2d.xetla_select<block_size_y, 1, block_size_x, 1>(
214 0, combine_i * block_size_x)
215 = reg_blk.xetla_select<block_elems, 1>(
216 combine_i * block_elems);
217 }
218#pragma unroll
219 for (int ii = 0; ii < block_size_y / st_blk_size_y; ++ii) {
220 constexpr uint32_t store_elems
221 = st_blk_size_y * block_size_x * arr_len;
222 auto st_blk = combine_blk.xetla_select<store_elems, 1>(
223 ii * store_elems);
224 xetla_tstore_global<dtype, store_elems, L1, L3>(tdesc, st_blk);
225 xetla_update_tdesc_offsety(
226 tdesc.xetla_format<uint32_t>(), st_blk_size_y);
227 }
228 // exceed hardware limitation
229 if constexpr ((block_size_y % st_blk_size_y) != 0) {
230 constexpr uint32_t blk_remained_start = block_size_y
231 / st_blk_size_y * st_blk_size_y * block_size_x
232 * arr_len;
233 constexpr uint8_t blk_remained_y = block_size_y % st_blk_size_y;
234 constexpr uint8_t blk_remained_elems
235 = blk_remained_y * block_size_x * arr_len;
236 auto st_blk = combine_blk.xetla_select<blk_remained_elems, 1>(
237 blk_remained_start);
238 constexpr uint32_t block_widthx_widthy_arrlen
239 = (block_size_x * arr_len - 1)
240 | ((blk_remained_y - 1) << 8);
241 gpu::xetla::detail::xetla_set_block_widthx_widthy_arrlen(
242 tdesc.xetla_format<uint32_t>(),
243 block_widthx_widthy_arrlen);
244 xetla_tstore_global<dtype, blk_remained_elems, L1, L3>(
245 tdesc, st_blk);
246 }
247 }
248 }
249 // process tail
250 if constexpr (remained_size_y > 0) {
251 constexpr uint32_t remained_block_elems
252 = block_size_x * remained_size_y;
253 constexpr uint32_t processed_elems
254 = num_block_y * num_block_x * block_elems;
255 constexpr uint32_t remained_st_blk_size_y
256 = st_blk_size_y > remained_size_y ? remained_size_y
257 : st_blk_size_y;
258 auto payload_row = payload_2d.xetla_select<num_block_x, 1, 16, 1>(
259 num_block_y * num_block_x, 0);
260 detail::reset_tile_desc_core<num_block_x, block_size_x * arr_len,
261 remained_st_blk_size_y, 1, 1, false>(payload_row);
262#pragma unroll
263 for (int j = 0; j < num_block_x; j += arr_len) {
264 xetla_tdescriptor tdesc = payload_row.row(j);
265 auto reg_blk
266 = tile.reg.xetla_select<remained_block_elems * arr_len, 1>(
267 processed_elems + j * remained_block_elems);
268 // Do combination
269 xetla_vector<dtype, remained_block_elems * arr_len> combine_blk;
270 auto combine_blk_2d = combine_blk.xetla_format<native_type_t<dtype>,
271 remained_size_y, block_size_x * arr_len>();
272#pragma unroll
273 for (int combine_i = 0; combine_i < arr_len; ++combine_i) {
274 combine_blk_2d
275 .xetla_select<remained_size_y, 1, block_size_x, 1>(
276 0, combine_i * block_size_x)
277 = reg_blk.xetla_select<remained_block_elems, 1>(
278 combine_i * remained_block_elems);
279 }
280#pragma unroll
281 for (int ii = 0; ii < remained_size_y / remained_st_blk_size_y;
282 ++ii) {
283 constexpr uint32_t store_elems
284 = remained_st_blk_size_y * block_size_x * arr_len;
285 auto st_blk = combine_blk.xetla_select<store_elems, 1>(
286 ii * store_elems);
287 xetla_tstore_global<dtype, store_elems, L1, L3>(tdesc, st_blk);
288 xetla_update_tdesc_offsety(
289 tdesc.xetla_format<uint32_t>(), remained_st_blk_size_y);
290 }
291 constexpr uint32_t final_st_blk_size_y
292 = remained_size_y % remained_st_blk_size_y;
293 if constexpr (final_st_blk_size_y != 0) {
294 constexpr uint32_t final_start = remained_size_y
295 / remained_st_blk_size_y * remained_st_blk_size_y
296 * block_size_x * arr_len;
297 constexpr uint32_t final_store_elems
298 = final_st_blk_size_y * block_size_x * arr_len;
299 auto st_blk = combine_blk.xetla_select<final_store_elems, 1>(
300 final_start);
301 constexpr uint32_t block_widthx_widthy_arrlen
302 = (block_size_x * arr_len - 1)
303 | ((final_st_blk_size_y - 1) << 8);
304 gpu::xetla::detail::xetla_set_block_widthx_widthy_arrlen(
305 tdesc.xetla_format<uint32_t>(),
306 block_widthx_widthy_arrlen);
307 xetla_tstore_global<dtype, final_store_elems, L1, L3>(
308 tdesc, st_blk);
309 }
310 }
311 }

324template <cache_hint L1 = cache_hint::write_back,
325 cache_hint L3 = cache_hint::write_back, typename tile_t,
326 typename payload_t>
327__XETLA_API typename std::enable_if_t<
328 detail::check_store_type<tile_t, payload_t>::is_global_block_1d_xe>
329tile_store(tile_t &tile, payload_t &payload) {
330 detail::check_store_condition<tile_t, payload_t>();
331 using dtype = typename tile_t::dtype;
332 using store_dtype = typename payload_t::mem_dtype;
333 static constexpr uint32_t tile_size_x = tile_t::tile_size_x;
334 static constexpr uint32_t scale_factor = payload_t::scale_factor;
335
336 constexpr uint32_t store_len = tile_size_x / scale_factor;
337 if constexpr (store_len >= 64) {
338#pragma unroll
339 for (int i = 0; i < store_len / 64; i++) {
340 uint32_t offset_x = i * 64 * scale_factor;
341 auto reg_sub
342 = tile.reg.xetla_select<64 * scale_factor, 1>(offset_x);
343 uint32_t address_offset = offset_x * sizeof(dtype);
344
345 xetla_store_global<store_dtype, 64, data_size::default_size, L1,
346 L3>(payload.base_ptr, payload.base_offset + address_offset,
347 reg_sub.xetla_format<store_dtype>());
348 }
349 }
350 constexpr uint32_t tail_len = store_len % 64;
351 uint32_t tail_offset = store_len / 64 * 64 * scale_factor;
352 detail::process_1d_tail<tail_len, 32, detail::process_flag::store, L1, L3>(
353 tile, payload, tail_offset);
354}

367template <cache_hint L1 = cache_hint::uncached,
368 cache_hint L3 = cache_hint::write_back, typename tile_t,
369 typename payload_t>
370__XETLA_API typename std::enable_if_t<
371 detail::check_store_type<tile_t, payload_t>::is_global_atomic_xe>
372tile_store(tile_t &tile, payload_t &payload) {
373 detail::check_store_condition<tile_t, payload_t>();
374 using dtype = typename tile_t::dtype;
375 using tile_desc = typename tile_t::tile_desc;
376 static constexpr uint32_t tile_size_x = tile_desc::tile_size_x;
377 static constexpr uint32_t tile_size_y = tile_desc::tile_size_y;
378 static constexpr uint32_t block_size_x = tile_desc::block_size_x;
379 static constexpr uint32_t block_size_y = tile_desc::block_size_y;
380 static constexpr uint32_t block_elems = tile_desc::block_elems;
381 static constexpr uint32_t num_block_x = tile_desc::num_block_x;
382
383 static constexpr atomic_op op_kind
384 = (std::is_same<remove_const_t<dtype>, float>::value
385 || std::is_same<remove_const_t<dtype>, double>::value)
386 ? atomic_op::fadd
387 : atomic_op::iadd;
388
389#pragma unroll
390 for (int i = 0; i < tile_size_y / block_size_y; i++) {
391 uint32_t offset_y = i * block_size_y;
392#pragma unroll
393 for (int j = 0; j < num_block_x; j++) {
394 uint32_t offset_x = j * block_size_x;
395 auto reg_sub = tile.reg.xetla_select<block_elems, 1>(
396 (i * num_block_x + j) * block_elems);
397 xetla_mask<payload_t::num_channel> pred_x
398 = (payload.step_x + offset_x + payload.base_x)
399 < payload.width_in_elems;
400#pragma unroll
401 for (int sub_block_y = 0; sub_block_y < block_size_y;
402 sub_block_y += payload_t::num_channel_y) {
403 xetla_mask<payload_t::num_channel> pred_y
404 = (payload.step_y + offset_y + payload.base_y
405 + sub_block_y)
406 < payload.height_in_elems;
407 uint32_t address_offset = offset_x * sizeof(dtype)
408 + (sub_block_y + offset_y) * payload.pitch_in_bytes;
409
410 xetla_tatomic_store_global<dtype, payload_t::num_channel, L1,
411 L3, op_kind>((uint64_t)payload.base_pointer
412 + payload.address + address_offset,
413 reg_sub.xetla_select<payload_t::store_elems, 1>(
414 sub_block_y * block_size_x),
415 pred_x & pred_y);
416 }
417 }
418 }
419 // process the tail
420 if constexpr ((tile_size_y % block_size_y) != 0) {
421 constexpr int i = tile_size_y / block_size_y;
422 constexpr uint32_t offset_y = i * block_size_y;
423 constexpr uint32_t processed_elems = offset_y * tile_size_x;
424 constexpr uint32_t remain_block_size_y = tile_size_y % block_size_y;
425 constexpr uint32_t remain_block_elems
426 = remain_block_size_y * block_size_x;
427#pragma unroll
428 for (int j = 0; j < num_block_x; j++) {
429 uint32_t offset_x = j * block_size_x;
430 auto reg_sub = tile.reg.xetla_select<remain_block_elems, 1>(
431 processed_elems + j * remain_block_elems);
432 xetla_mask<payload_t::num_channel> pred_x
433 = (payload.step_x + offset_x + payload.base_x)
434 < payload.width_in_elems;
435#pragma unroll
436 for (int sub_block_y = 0; sub_block_y < remain_block_size_y;
437 sub_block_y += payload_t::num_channel_y) {
438 xetla_mask<payload_t::num_channel> pred_y
439 = (payload.step_y + offset_y + payload.base_y
440 + sub_block_y)
441 < payload.height_in_elems;
442 uint32_t address_offset = offset_x * sizeof(dtype)
443 + (sub_block_y + offset_y) * payload.pitch_in_bytes;
444
445 xetla_tatomic_store_global<dtype, payload_t::num_channel, L1,
446 L3, op_kind>((uint64_t)payload.base_pointer
447 + payload.address + address_offset,
448 reg_sub.xetla_select<payload_t::store_elems, 1>(
449 sub_block_y * block_size_x),
450 pred_x & pred_y);
451 }
452 }
453 }

467template <cache_hint L1 = cache_hint::write_back,
468 cache_hint L3 = cache_hint::write_back, typename tile_t,

505 = tile_desc::tile_size_y % tile_desc::block_size_y;

513#pragma unroll

515 sub_block_y += num_channel_y) {
516 uint32_t address_offset = offset_x * sizeof(dtype)
517 + (sub_block_y + offset_y) * payload.pitch_in_bytes;
518 xetla_store_local<store_dtype>(payload.address + address_offset,
519 reg_sub.xetla_select<store_elems, 1>(
520 sub_block_y * tile_desc::block_size_x)
521 .xetla_format<store_dtype>());
522 }

538template <cache_hint L1 = cache_hint::write_back,
539 cache_hint L3 = cache_hint::write_back, typename tile_t,
540 typename payload_t>
541__XETLA_API typename std::enable_if_t<detail::check_store_type<tile_t,
542 payload_t>::is_local_scatter_vnni_col_xe>
543tile_store(tile_t &tile, payload_t &payload) {
544 detail::check_store_condition<tile_t, payload_t>();
545 using dtype = typename tile_t::dtype;
546 using tile_desc = typename tile_t::tile_desc;
