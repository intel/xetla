33 template <typename matAcc_t, typename coord_t>
34 __XETLA_API KERNEL_FUNC void operator()(matAcc_t &matAcc, coord_t coord,
35 arguments_t args, uint32_t slm_base = 0,
36 uint32_t nbarrier_base = 0) {
37 using dtype_acc = typename matAcc_t::dtype;
38 // total flag register
39 constexpr uint32_t elems = 8 * 16;
40 constexpr uint32_t rounds = matAcc_t::tile_elems / elems;
41#pragma unroll
42 for (uint32_t r = 0; r < rounds; ++r) {
43 xetla_vector<dtype_acc, elems> res(0.0f);
44 auto x = matAcc.reg.xetla_select<elems, 1>(elems * r);
45#pragma unroll
46 for (uint32_t i = 0; i < N; ++i) {
47 res = x * res;
48 res += static_cast<dtype_acc>(args.coeff[i]);
49 }
50 x = res;
51 }
52 constexpr uint32_t remained_elems = matAcc_t::tile_elems % elems;
53 if constexpr (remained_elems != 0) {
54 xetla_vector<dtype_acc, remained_elems> res(0.0f);
55 auto x = matAcc.reg.xetla_select<remained_elems, 1>(
56 elems * (matAcc_t::tile_elems / elems));
57#pragma unroll
58 for (uint32_t i = 0; i < N; ++i) {
59 res = x * res;
60 res += static_cast<dtype_acc>(args.coeff[i]);
61 }
62 x = res;
63 }
64 }
65};
