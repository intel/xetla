294 static bool can_implement(arguments_t &args) {
295 bool implementable = true;
296 if (brgemm_t::is_2d_block_a) {
297 bool implementable_a = detail::check_2d_block_restriction<dtype_a>(
298 args.matA_base.base, args.matA_ld);
299 if (!implementable_a) {
300 std::cout << "matA is not well aligned!" << std::endl;
301 }
302 implementable &= implementable_a;
303 } else {
304 bool implementable_a = detail::check_dw_align<dtype_a>(
305 args.matA_base.base, args.matA_ld);
306 if (!implementable_a) {
307 std::cout << "matA is not well aligned!" << std::endl;
308 }
309 implementable &= implementable_a;
310 }
311 if (brgemm_t::is_2d_block_b) {
312 // for 4bit matB
313 bool implementable_b = detail::check_2d_block_restriction<dtype_b>(
314 args.matB_base.base, args.matB_ld / pack_ratio);
315 if (!implementable_b) {
316 std::cout << "matB is not well aligned!" << std::endl;
317 }
318 implementable &= implementable_b;
319 } else {
320 // for 4bit matB
321 bool implementable_b = detail::check_dw_align<dtype_b>(
322 args.matB_base.base, args.matB_ld / pack_ratio);
323 if (!implementable_b) {
324 std::cout << "matB is not well aligned!" << std::endl;
325 }
326 implementable &= implementable_b;
327 }
328 if (epilogue_t::is_2d_block_c) {
329 bool implementable_c = detail::check_2d_block_restriction<dtype_c>(
330 args.matC_base.base, args.matC_ld);
331 if (!implementable_c) {
332 std::cout << "matC is not well aligned!" << std::endl;
333 }
334 implementable &= implementable_c;
335 } else {
336 bool implementable_c = detail::check_dw_align<dtype_c>(
337 args.matC_base.base, args.matC_ld);
338 if (!implementable_c) {
339 std::cout << "matC is not well aligned!" << std::endl;
340 }
341 implementable &= implementable_c;
342 }
343 // check for int4x2
344 implementable &= ((args.matB_ld % pack_ratio == 0)
345 && (args.zero_pt_ld % pack_ratio == 0)
346 && (args.matrix_n % pack_ratio == 0));
347
348 return implementable;
349 }

358 __XETLA_API KERNEL_FUNC void operator()(xetla_exec_item<3> &ei,
359 const arguments_t &args, uint32_t slm_base = 0,
360 uint32_t nbarrier_base = 0) {
361 // set up workgroup level coordinates and boundaries
362 work_group_t g(ei.get_local_linear_id() % work_group_size);
363 uint32_t wg_id = ei.get_local_linear_id() / work_group_size;
364 int start_n = ei.get_group(2) * wg_tile_n;
365 int start_m = ei.get_group(1) * wg_tile_m;
366 int start_k = 0;
367 uint32_t wg_tile_k = args.matrix_k;
368 uint32_t boundary_n = (start_n + wg_tile_n) > args.matrix_n
369 ? args.matrix_n
370 : (start_n + wg_tile_n);
371 uint32_t boundary_m = (start_m + wg_tile_m) > args.matrix_m
372 ? args.matrix_m
373 : (start_m + wg_tile_m);
374 uint32_t boundary_k = wg_tile_k;
375 if constexpr (global_kslicing_ratio > 1) {
376 wg_tile_k = (wg_tile_k + global_kslicing_ratio - 1)
377 / global_kslicing_ratio;
378 start_k = start_k + ei.get_group(0) * wg_tile_k;
379 boundary_k = (start_k + wg_tile_k) > boundary_k
380 ? boundary_k
381 : (start_k + wg_tile_k);
382 }
383 if constexpr (local_kslicing_ratio > 1) {
384 wg_tile_k = (wg_tile_k + local_kslicing_ratio - 1)
385 / local_kslicing_ratio;
386 start_k = start_k + wg_id * wg_tile_k;
387 boundary_k = (start_k + wg_tile_k) > boundary_k
388 ? boundary_k
389 : (start_k + wg_tile_k);
390 }
391
392 int start_x_scale = start_n;
393 int start_y_scale = start_k / dequant_s;
394
395 int start_x_zero_pt = start_n / pack_ratio;
396 int start_y_zero_pt = start_k / dequant_s;
397
398 // set up arguments
399 uint32_t brgemm_slm_base = slm_base;
400 uint32_t brgemm_nbarr_base = nbarrier_base;
401 if constexpr (local_kslicing_ratio > 1) {
402 brgemm_slm_base = slm_base + wg_id * brgemm_slm_size;
403 brgemm_nbarr_base = nbarrier_base + wg_id * brgemm_nbarr_count;
404 }
405 uint32_t kslicing_slm_base
406 = slm_base + local_kslicing_ratio * brgemm_slm_size;
407 uint32_t kslicing_nbarr_base
408 = nbarrier_base + local_kslicing_ratio * brgemm_nbarr_count;
409 uint32_t epilogue_slm_base = kslicing_slm_base + kslicing_slm_size;
410 uint32_t epilogue_nbarr_base
411 = kslicing_nbarr_base + kslicing_nbarr_count;
412
413 mem_desc_a_t mem_desc_a;
414 mem_desc_b_t mem_desc_b;
415 mem_desc_c_t mem_desc_c;
416 //setup for matA
417
418 mem_desc_a.init(args.matA_base, {boundary_k, boundary_m, args.matA_ld},
419 {start_k, start_m});
420 mem_desc_b.init(args.matB_base,
421 {boundary_n / pack_ratio, boundary_k,
422 args.matB_ld / pack_ratio},
423 {int(start_n / pack_ratio), start_k});
424 mem_desc_scale_t mem_desc_scale(args.scale_base,
425 {args.matrix_n, args.matrix_k / dequant_s, args.scale_ld},
426 {start_x_scale, start_y_scale});
427 mem_desc_zero_pt_t mem_desc_zero_pt(args.zero_pt_base,
428 {args.matrix_n / pack_ratio, args.matrix_k / dequant_s,
429 args.zero_pt_ld / pack_ratio},
430 {start_x_zero_pt, start_y_zero_pt});
431
432 uint32_t inner_loop_count = (wg_tile_k + k_stride - 1) / k_stride;
433 brgemm_args_t brgemm_args(mem_desc_a, mem_desc_b, inner_loop_count,
434 mem_desc_scale, mem_desc_zero_pt);
435 matAcc_t matAcc;
436 matAcc.init(0);
437 brgemm_t brgemm;
438 brgemm(g, matAcc, brgemm_args, brgemm_slm_base, brgemm_nbarr_base);
439
440 kslicing_t kslicing(wg_id);
441 mat_slice_t mat_slice;
442 kslicing(g, mat_slice, matAcc, kslicing_slm_base, kslicing_nbarr_base);
443
444 //setup for matC
445 //set up cooperative offset for matC store
446 int32_t coop_offset_n = kslicing.coop_id_x * mat_slice_t::tile_size_x;
447 int32_t coop_offset_m = kslicing.coop_id_y * mat_slice_t::tile_size_y;
448 if constexpr (mem_desc_c_t::is_local) {
449 mem_desc_c.init(args.matC_base,
450 {real_wg_tile_n, real_wg_tile_m, real_wg_tile_n},
451 {coop_offset_n, coop_offset_m});
452 } else {
453 mem_desc_c.init(args.matC_base,
454 {boundary_n, boundary_m, args.matC_ld},
455 {start_n + coop_offset_n, start_m + coop_offset_m});
456 }
457 epilogue_t epilogue;
458 epilogue(g, mat_slice, mem_desc_c, args.epilogue_args,
459 epilogue_slm_base, epilogue_nbarr_base);
460 }
