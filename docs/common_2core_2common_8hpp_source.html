224template <lsc_action Action, cache_hint L1H, cache_hint L3H>
225constexpr void check_lsc_cache_hint() {
226 if constexpr (Action == lsc_action::prefetch) {
227 // https://gfxspecs.intel.com/Predator/Home/Index/53560
228 static_assert(
229 ((L3H == cache_hint::uncached || L3H == cache_hint::cached)
230 && (L1H == cache_hint::uncached
231 || L1H == cache_hint::cached
232 || L1H == cache_hint::streaming)),
233 "cache hint type not supported!");
234 } else if constexpr (Action == lsc_action::load) {
235 // https://gfxspecs.intel.com/Predator/Home/Index/53560
236 static_assert((L1H == cache_hint::none && L3H == cache_hint::none)
237 || ((L3H == cache_hint::uncached)
238 && (L1H == cache_hint::uncached
239 || L1H == cache_hint::cached
240 || L1H == cache_hint::streaming))
241 || ((L3H == cache_hint::cached)
242 && (L1H == cache_hint::uncached
243 || L1H == cache_hint::cached
244 || L1H == cache_hint::streaming
245 || L1H == cache_hint::read_invalidate)),
246 "unsupported cache hint!");
247 } else if constexpr (Action == lsc_action::store) {
248 // https://gfxspecs.intel.com/Predator/Home/Index/53561
249 static_assert((L1H == cache_hint::none && L3H == cache_hint::none)
250 || ((L3H == cache_hint::uncached)
251 && (L1H == cache_hint::uncached
252 || L1H == cache_hint::write_through
253 || L1H == cache_hint::streaming))
254 || ((L3H == cache_hint::write_back)
255 && (L1H == cache_hint::uncached
256 || L1H == cache_hint::write_through
257 || L1H == cache_hint::streaming
258 || L1H == cache_hint::write_back)),
259 "unsupported cache hint!");
260 } else if constexpr (Action == lsc_action::atomic) {
261 // https://gfxspecs.intel.com/Predator/Home/Index/53561
262 static_assert((L1H == cache_hint::none && L3H == cache_hint::none)
263 || (L1H == cache_hint::uncached
264 && (L3H == cache_hint::uncached
265 || L3H == cache_hint::write_back)),
266 "unsupported cache hint!");
267 }
268}
