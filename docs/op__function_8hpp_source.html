84template <typename T, post_kind post_op>
85__XETLA_API typename std::enable_if_t<post_op == post_kind::gelu> elemwise_op(
86 T &mat_Acc) {
87 using dtype = typename T::dtype;
88 constexpr dtype C0 = 0.044715f;
89 constexpr dtype sqrt_two_over_pi = 0.79788458347320556640625f;
90 // total flag register
91 constexpr int elems = 8 * 16;
92 constexpr int rounds = T::reg_tile_elems / elems;
93#pragma unroll
94 for (int i = 0; i < rounds; ++i) {
95 auto sub_vec = mat_Acc.reg.xetla_select<elems, 1>(elems * i);
96 xetla_vector<dtype, elems> sub_vec_x
97 = (sqrt_two_over_pi * sub_vec * (1.f + C0 * sub_vec * sub_vec));
98 xetla_vector<dtype, elems> tanh_value
99 = xetla_tanh<dtype, elems>(sub_vec_x);
100 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);
101 }
102 constexpr int remained_elems = T::reg_tile_elems % elems;
103 if constexpr (remained_elems != 0) {
104 auto sub_vec = mat_Acc.reg.xetla_select<remained_elems, 1>(
105 elems * (T::reg_tile_elems / elems));
106 xetla_vector<dtype, remained_elems> sub_vec_x
107 = (sqrt_two_over_pi * sub_vec * (1.f + C0 * sub_vec * sub_vec));
108 xetla_vector<dtype, remained_elems> tanh_value
109 = xetla_tanh<dtype, remained_elems>(sub_vec_x);
110 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);
111 }
112}

121__XETLA_API typename std::enable_if_t<post_op == post_kind::tanh> elemwise_op(
122 T &mat_Acc) {
123 constexpr int elems = T::tile_desc::block_elems;
124 constexpr int rounds = T::tile_desc::tile_elems / elems;
125 using dtype = typename T::dtype;
126#pragma unroll
127 for (int i = 0; i < rounds; ++i) {

129 sub_vec = xetla_tanh<dtype, elems>(sub_vec);

145template <typename T, post_kind post_op>
146__XETLA_API static typename std::enable_if_t<post_op == post_kind::sigmoid>
147elemwise_op(T &mat_Acc) {
148 constexpr int elems = T::tile_desc::block_elems;
149 constexpr int rounds = T::tile_desc::tile_elems / elems;
150 constexpr float one = 1.0f;
151#pragma unroll
152 for (int i = 0; i < rounds; ++i) {
153 auto sub_vec = mat_Acc.reg.xetla_select<elems, 1>(elems * i);
154 xetla_mask<elems> mask = sub_vec >= 10;
155 xetla_vector<typename T::dtype, elems> temp_vec
156 = xetla_exp<typename T::dtype, elems>(sub_vec);
157 xetla_vector<typename T::dtype, elems> sigmoid_value
158 = temp_vec / (temp_vec + one);
159 sigmoid_value.xetla_merge(1, mask);
160 sub_vec = sigmoid_value;
161 }
162 constexpr int remained_elems = T::tile_desc::tile_elems % elems;
163 if constexpr (remained_elems != 0) {
164 auto sub_vec = mat_Acc.reg.xetla_select<remained_elems, 1>(
165 elems * (T::reg_tile_elems / elems));
166 xetla_mask<remained_elems> mask = sub_vec >= 250;
167 xetla_vector<typename T::dtype, remained_elems> temp_vec
168 = xetla_exp<typename T::dtype, remained_elems>(sub_vec);
169 xetla_vector<typename T::dtype, remained_elems> sigmoid_value
170 = temp_vec / (temp_vec + one);
171 sigmoid_value.xetla_merge(1, mask);
172 sub_vec = sigmoid_value;
173 }

181template <typename T>
182__XETLA_API typename std::enable_if_t<T::tile_desc::register_layout
183 == reg_layout::vnni_tiled>
184vnni_reverse(T &mat_Acc) {
185 constexpr uint32_t reg_tile_size_y = T::tile_desc::tile_size_y;
186 constexpr uint32_t reg_tile_size_x = T::tile_desc::tile_size_x;
187 constexpr uint32_t reg_tile_elems = reg_tile_size_y * reg_tile_size_x;
188 constexpr uint32_t block_size_y = T::tile_desc::block_size_y;
189 constexpr uint32_t block_size_x = T::tile_desc::block_size_x;
190 constexpr uint32_t block_elems = block_size_y * block_size_x;
191 constexpr int32_t num_block_x = reg_tile_size_x / block_size_x;
192 using dtype = typename T::dtype;
193 constexpr int32_t move_stride = sizeof(uint32_t) / sizeof(dtype);
194 constexpr int32_t move_elems = block_size_x * move_stride;
195 xetla_vector<dtype, reg_tile_elems> rdst;
196#pragma unroll
197 for (int i = 0; i < reg_tile_size_y / block_size_y; i++) {
198#pragma unroll
199 for (int j = 0; j < num_block_x; j++) {
200 auto reg = (mat_Acc.reg)
201 .xetla_select<block_elems, 1>(
202 (i * num_block_x + j) * block_elems);
203 auto reg_dst = rdst.xetla_select<block_elems, 1>(
204 (i * num_block_x + j) * block_elems);
205 for (int row_i = 0; row_i < block_size_y; row_i += move_stride) {
206 auto reg_sub
207 = reg.xetla_select<move_elems, 1>(row_i * block_size_x);
208 auto reg_dst_sub = reg_dst.xetla_select<move_elems, 1>(
209 row_i * block_size_x);
210#pragma unroll
211 for (int move_i = 0; move_i < move_stride; move_i++) {
212 reg_dst_sub.xetla_select<block_size_x, 1>(
213 move_i * block_size_x)
214 = reg_sub.xetla_select<block_size_x, move_stride>(
215 move_i);
216 }
217 }
218 }
219 }
220 // process the tail
221 if constexpr ((reg_tile_size_y % block_size_y) != 0) {
222 constexpr int i = reg_tile_size_y / block_size_y;
223 constexpr uint32_t remain_elems_start
224 = i * block_size_y * reg_tile_size_x;
225 constexpr uint32_t remain_size_y = reg_tile_size_y % block_size_y;
226 constexpr uint32_t remain_block_elems = remain_size_y * block_size_x;
227#pragma unroll
228 for (int j = 0; j < num_block_x; j++) {
229 auto reg = (mat_Acc.reg)
230 .xetla_select<remain_block_elems, 1>(
231 remain_elems_start
232 + j * remain_block_elems);
233 auto reg_dst = rdst.xetla_select<remain_block_elems, 1>(
234 remain_elems_start + j * remain_block_elems);
235 // for mma, here we can guarantee that the remaining is a multiply of 8
236 for (int row_i = 0; row_i < remain_size_y; row_i += move_stride) {
237 auto reg_sub
238 = reg.xetla_select<move_elems, 1>(row_i * block_size_x);
239 auto reg_dst_sub = reg_dst.xetla_select<move_elems, 1>(
240 row_i * block_size_x);
241#pragma unroll
242 for (int move_i = 0; move_i < move_stride; move_i++) {
243 reg_dst_sub.xetla_select<block_size_x, 1>(
244 move_i * block_size_x)
245 = reg_sub.xetla_select<block_size_x, move_stride>(
246 move_i);
247 }
248 }
249 }
250 }
251 mat_Acc.reg = rdst;

261template <typename T_dst, typename T_src>
262__XETLA_API typename std::enable_if_t<(T_src::tile_desc::block_size_y
263 == T_dst::tile_desc::block_size_y)
264 && (T_src::tile_desc::block_size_x == T_dst::tile_desc::block_size_x)
265 && (T_src::tile_desc::tile_size_y == T_dst::tile_desc::tile_size_y)
266 && (T_src::tile_desc::tile_size_x == T_dst::tile_desc::tile_size_x)>
267vnni_transform(T_dst &dst, T_src &src) {
268 constexpr uint32_t reg_tile_size_y = T_dst::tile_desc::tile_size_y;
269 constexpr uint32_t reg_tile_size_x = T_dst::tile_desc::tile_size_x;
270 constexpr uint32_t reg_tile_elems = reg_tile_size_y * reg_tile_size_x;
271 constexpr uint32_t block_size_y = T_dst::tile_desc::block_size_y;
272 constexpr uint32_t block_size_x = T_dst::tile_desc::block_size_x;
273 constexpr uint32_t block_elems = block_size_y * block_size_x;
274 constexpr int32_t num_block_x = reg_tile_size_x / block_size_x;
275 using dtype_dst = typename T_dst::dtype;
276 using dtype_src = typename T_src::dtype;
277 constexpr uint32_t vnni_row_src = sizeof(uint32_t) / sizeof(dtype_src);
278 constexpr uint32_t vnni_row_dst = sizeof(uint32_t) / sizeof(dtype_dst);
279 constexpr int32_t vnni_row
280 = vnni_row_src > vnni_row_dst ? vnni_row_src : vnni_row_dst;
281 static_assert(block_size_y % vnni_row == 0);
282 static_assert(reg_tile_size_y % vnni_row == 0);
283 constexpr int32_t move_elems = vnni_row * block_size_x;
284 xetla_vector<dtype_dst, reg_tile_elems> reg_src
285 = xetla_cvt<dtype_dst, dtype_src, reg_tile_elems>(src.reg);
286 if constexpr (sizeof(dtype_src) == sizeof(dtype_dst)) {
287 dst.reg = reg_src;
288 return;
289 }
290 xetla_vector<dtype_dst, reg_tile_elems> reg_dst;
291 constexpr uint32_t scale_factor
292 = detail::gcd<vnni_row_src, vnni_row_dst>::value;
293 using move_dtype = get_uint_type_t<sizeof(dtype_dst) * scale_factor>;
294 constexpr uint32_t select_stride = vnni_row / scale_factor;
295#pragma unroll
296 for (int i = 0; i < reg_tile_size_y / block_size_y; i++) {
297#pragma unroll
298 for (int j = 0; j < num_block_x; j++) {
299 auto reg_src_blk = reg_src.xetla_select<block_elems, 1>(
300 (i * num_block_x + j) * block_elems);
301 auto reg_dst_blk = reg_dst.xetla_select<block_elems, 1>(
302 (i * num_block_x + j) * block_elems);
303 for (int row_i = 0; row_i < block_size_y; row_i += vnni_row) {
304 auto reg_src_move = reg_src_blk
305 .xetla_select<move_elems, 1>(
306 row_i * block_size_x)
307 .xetla_format<move_dtype>();
308 auto reg_dst_move = reg_dst_blk
309 .xetla_select<move_elems, 1>(
310 row_i * block_size_x)
311 .xetla_format<move_dtype>();
312#pragma unroll
313 for (int move_i = 0; move_i < select_stride; move_i++) {
314 if constexpr (sizeof(dtype_dst) > sizeof(dtype_src)) {
315 reg_dst_move.xetla_select<block_size_x, 1>(
316 move_i * block_size_x)
317 = reg_src_move.xetla_select<block_size_x,
318 select_stride>(move_i);
319 } else {
320 reg_dst_move.xetla_select<block_size_x, select_stride>(
321 move_i)
322 = reg_src_move.xetla_select<block_size_x, 1>(
323 move_i * block_size_x);
324 }
325 }
326 }
327 }
328 }
329 // process the tail
330 if constexpr ((reg_tile_size_y % block_size_y) != 0) {
331 constexpr int i = reg_tile_size_y / block_size_y;
332 constexpr uint32_t remain_elems_start
333 = i * block_size_y * reg_tile_size_x;
334 constexpr uint32_t remain_size_y = reg_tile_size_y % block_size_y;
335 constexpr uint32_t remain_block_elems = remain_size_y * block_size_x;
336#pragma unroll
337 for (int j = 0; j < num_block_x; j++) {
338 auto reg_src_blk = reg_src.xetla_select<remain_block_elems, 1>(
339 remain_elems_start + j * remain_block_elems);
340 auto reg_dst_blk = reg_dst.xetla_select<remain_block_elems, 1>(
341 remain_elems_start + j * remain_block_elems);
342 // for mma, here we can guarantee that the remaining is a multiple of
343 // vnni_row
344 for (int row_i = 0; row_i < remain_size_y; row_i += vnni_row) {
345 auto reg_src_move = reg_src_blk
346 .xetla_select<move_elems, 1>(
347 row_i * block_size_x)
348 .xetla_format<move_dtype>();
349 auto reg_dst_move = reg_dst_blk
350 .xetla_select<move_elems, 1>(
351 row_i * block_size_x)
352 .xetla_format<move_dtype>();
353#pragma unroll
354 for (int move_i = 0; move_i < select_stride; move_i++) {
355 if constexpr (sizeof(dtype_dst) > sizeof(dtype_src)) {
356 reg_dst_move.xetla_select<block_size_x, 1>(
357 move_i * block_size_x)
358 = reg_src_move.xetla_select<block_size_x,
359 select_stride>(move_i);
360 } else {
361 reg_dst_move.xetla_select<block_size_x, select_stride>(
362 move_i)
363 = reg_src_move.xetla_select<block_size_x, 1>(
364 move_i * block_size_x);
365 }
366 }
367 }
368 }
369 }
370 dst.reg = reg_dst;
371}

380template <typename T_dst, typename T_src>
381__XETLA_API typename std::enable_if_t<(T_dst::tile_desc::register_layout
382 == reg_layout::tiled)
383 && (T_src::tile_desc::register_layout == reg_layout::tiled)
384 && (T_src::tile_desc::tile_size_x == T_dst::tile_desc::tile_size_x)
385 && (T_src::tile_desc::tile_size_y == 1)>
386row_broadcast(T_dst &dst, T_src &src) {
387 static constexpr uint32_t dst_tile_size_y = T_dst::tile_desc::tile_size_y;
388 static constexpr uint32_t dst_tile_size_x = T_dst::tile_desc::tile_size_x;
389 static constexpr uint32_t dst_tile_elems = T_dst::tile_desc::tile_elems;
390 static constexpr uint32_t dst_block_size_y = T_dst::tile_desc::block_size_y;
391 static constexpr uint32_t dst_block_size_x = T_dst::tile_desc::block_size_x;
392 static constexpr uint32_t dst_block_elems = T_dst::tile_desc::block_elems;
393 static constexpr int32_t dst_num_block_y = T_dst::tile_desc::num_block_y;
394 static constexpr int32_t dst_num_block_x = T_dst::tile_desc::num_block_x;
395 using dst_dtype = typename T_dst::dtype;
396 using src_dtype = typename T_src::dtype;
397
398#pragma unroll
399 for (int i = 0; i < dst_tile_size_y / dst_block_size_y; i++) {
400#pragma unroll
401 for (int j = 0; j < dst_num_block_x; j++) {
402 auto dst_reg = (dst.reg)
403 .xetla_select<dst_block_elems, 1>(
404 (i * dst_num_block_x + j)
405 * dst_block_elems)
406 .xetla_format<dst_dtype, dst_block_size_y,
407 dst_block_size_x>();
408#pragma unroll
409 for (int row_i = 0; row_i < dst_block_size_y; row_i++) {
410 auto src_reg = src.reg.xetla_select<dst_block_size_x, 1>(
411 j * dst_block_size_x);
412 dst_reg.row(row_i)
413 = xetla_cvt<dst_dtype, src_dtype, dst_block_size_x>(
414 src_reg);
415 }
416 }
417 }
418
419 // process the tail
420 if constexpr ((dst_tile_size_y % dst_block_size_y) != 0) {
421 constexpr uint32_t tail_start_y
422 = dst_tile_size_y / dst_block_size_y * dst_block_size_y;
423 constexpr int32_t dst_tail_size_y = dst_tile_size_y % dst_block_size_y;
424 constexpr int32_t dst_tail_block_elems
425 = dst_tail_size_y * dst_block_size_x;
426#pragma unroll
427 for (int j = 0; j < dst_num_block_x; j++) {
428 auto dst_reg = (dst.reg)
429 .xetla_select<dst_tail_block_elems, 1>(
430 tail_start_y * dst_tile_size_x
431 + j * dst_tail_block_elems)
432 .xetla_format<dst_dtype, dst_tail_size_y,
433 dst_block_size_x>();
434#pragma unroll
435 for (int row_i = 0; row_i < dst_tail_size_y; row_i++) {
436 auto src_reg = src.reg.xetla_select<dst_block_size_x, 1>(
437 j * dst_block_size_x);
438 dst_reg.row(row_i)
439 = xetla_cvt<dst_dtype, src_dtype, dst_block_size_x>(
440 src_reg);
441 }
442 }
443 }
