84template <typename T, post_kind post_op>
85__XETLA_API typename std::enable_if_t<post_op == post_kind::gelu> elemwise_op(
86 T &mat_Acc) {
87 using dtype = typename T::dtype;
88 constexpr dtype C0 = 0.044715f;
89 constexpr dtype sqrt_two_over_pi = 0.79788458347320556640625f;
90 // total flag register
91 constexpr int elems = 8 * 16;
92 constexpr int rounds = T::tile_elems / elems;
93#pragma unroll
94 for (int i = 0; i < rounds; ++i) {
95 auto sub_vec = mat_Acc.reg.xetla_select<elems, 1>(elems * i);
96 xetla_vector<dtype, elems> sub_vec_x
97 = (sqrt_two_over_pi * sub_vec * (1.f + C0 * sub_vec * sub_vec));
98 xetla_vector<dtype, elems> tanh_value
99 = xetla_tanh<dtype, elems>(sub_vec_x);
100 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);
101 }
102 constexpr int remained_elems = T::tile_elems % elems;
103 if constexpr (remained_elems != 0) {
104 auto sub_vec = mat_Acc.reg.xetla_select<remained_elems, 1>(
105 elems * (T::tile_elems / elems));
106 xetla_vector<dtype, remained_elems> sub_vec_x
107 = (sqrt_two_over_pi * sub_vec * (1.f + C0 * sub_vec * sub_vec));
108 xetla_vector<dtype, remained_elems> tanh_value
109 = xetla_tanh<dtype, remained_elems>(sub_vec_x);
110 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);
111 }
112}

120template <typename T, post_kind post_op>
121__XETLA_API typename std::enable_if_t<post_op == post_kind::tanh> elemwise_op(
122 T &mat_Acc) {
123 constexpr int elems = T::block_elems;
124 constexpr int rounds = T::tile_elems / elems;
125 using dtype = typename T::dtype;
126#pragma unroll
127 for (int i = 0; i < rounds; ++i) {
128 auto sub_vec = mat_Acc.reg.xetla_select<elems, 1>(elems * i);
129 sub_vec = xetla_tanh<dtype, elems>(sub_vec);
130 }
131 constexpr int remained_elems = T::tile_elems % elems;
132 if constexpr (remained_elems != 0) {
133 auto sub_vec = mat_Acc.reg.xetla_select<remained_elems, 1>(
134 elems * (T::tile_elems / elems));
135 sub_vec = xetla_tanh<dtype, remained_elems>(sub_vec);
136 }

145template <typename T, post_kind post_op>
146__XETLA_API static typename std::enable_if_t<post_op == post_kind::sigmoid>
147elemwise_op(T &mat_Acc) {
148 constexpr int elems = T::block_elems;
149 constexpr int rounds = T::tile_elems / elems;
150 constexpr float one = 1.0f;
151#pragma unroll
152 for (int i = 0; i < rounds; ++i) {
153 auto sub_vec = mat_Acc.reg.xetla_select<elems, 1>(elems * i);
154 xetla_mask<elems> mask = sub_vec >= 10;
155 xetla_vector<typename T::dtype, elems> temp_vec
156 = xetla_exp<typename T::dtype, elems>(sub_vec);
157 xetla_vector<typename T::dtype, elems> sigmoid_value
158 = temp_vec / (temp_vec + one);
159 sigmoid_value.xetla_merge(1, mask);
160 sub_vec = sigmoid_value;
161 }
162 constexpr int remained_elems = T::tile_elems % elems;
163 if constexpr (remained_elems != 0) {
164 auto sub_vec = mat_Acc.reg.xetla_select<remained_elems, 1>(
165 elems * (T::tile_elems / elems));
166 xetla_mask<remained_elems> mask = sub_vec >= 250;
167 xetla_vector<typename T::dtype, remained_elems> temp_vec
168 = xetla_exp<typename T::dtype, remained_elems>(sub_vec);
169 xetla_vector<typename T::dtype, remained_elems> sigmoid_value
170 = temp_vec / (temp_vec + one);
171 sigmoid_value.xetla_merge(1, mask);
172 sub_vec = sigmoid_value;
173 }

181template <typename T>
182__XETLA_API typename std::enable_if_t<T::register_layout == reg_layout::tiled>
183vnni_reverse(T &mat_Acc) {
184 constexpr uint32_t tile_size_y = T::tile_size_y;
185 constexpr uint32_t tile_size_x = T::tile_size_x;
186 constexpr uint32_t tile_elems = tile_size_y * tile_size_x;
187 constexpr uint32_t block_size_y = T::block_size_y;
188 constexpr uint32_t block_size_x = T::block_size_x;
189 constexpr uint32_t block_elems = block_size_y * block_size_x;
190 constexpr int32_t num_block_x = tile_size_x / block_size_x;
191 using dtype = typename T::dtype;
192 constexpr int32_t vnni_stride = sizeof(uint32_t) / sizeof(dtype);
193 constexpr int32_t move_cols = block_size_x * vnni_stride;
194 constexpr int32_t move_rows = block_size_y / vnni_stride;
195 xetla_vector<dtype, tile_elems> rdst;
196 static_assert(block_size_y % vnni_stride == 0, "vnni alignement check");
197 if constexpr (tile_size_x == 1) { return; }
198#pragma unroll
199 for (int i = 0; i < tile_size_y / block_size_y; i++) {
200#pragma unroll
201 for (int j = 0; j < num_block_x; j++) {
202 auto reg = (mat_Acc.reg)
203 .xetla_select<block_elems, 1>(
204 (i * num_block_x + j) * block_elems);
205 auto reg_2d = reg.xetla_format<native_type_t<dtype>, move_rows,
206 move_cols>();
207 auto reg_dst = rdst.xetla_select<block_elems, 1>(
208 (i * num_block_x + j) * block_elems);
209 auto reg_dst_2d = reg_dst.xetla_format<native_type_t<dtype>,
210 block_size_y, block_size_x>();
211#pragma unroll
212 for (int vnni_i = 0; vnni_i < vnni_stride; vnni_i++) {
213 reg_dst_2d
214 .xetla_select<move_rows, vnni_stride, block_size_x, 1>(
215 vnni_i, 0)
216 = reg_2d.xetla_select<move_rows, 1, block_size_x,
217 vnni_stride>(0, vnni_i);
218 }
219 }
220 }
221 // process the tail
222 if constexpr ((tile_size_y % block_size_y) != 0) {
223 constexpr int i = tile_size_y / block_size_y;
224 constexpr uint32_t remain_elems_start = i * block_size_y * tile_size_x;
225 constexpr uint32_t remain_size_y = tile_size_y % block_size_y;
226 constexpr uint32_t remain_block_elems = remain_size_y * block_size_x;
227 static_assert(
228 remain_size_y % vnni_stride == 0, "vnni alignement check");
229 constexpr int32_t remain_move_cols = block_size_x * vnni_stride;
230 constexpr int32_t remain_move_rows = remain_size_y / vnni_stride;
231#pragma unroll
232 for (int j = 0; j < num_block_x; j++) {
233 auto reg = (mat_Acc.reg)
234 .xetla_select<remain_block_elems, 1>(
235 remain_elems_start
236 + j * remain_block_elems);
237 auto reg_2d = reg.xetla_format<native_type_t<dtype>,
238 remain_move_rows, remain_move_cols>();
239 auto reg_dst = rdst.xetla_select<remain_block_elems, 1>(
240 remain_elems_start + j * remain_block_elems);
241 auto reg_dst_2d = reg_dst.xetla_format<native_type_t<dtype>,
242 remain_size_y, block_size_x>();
243 for (int vnni_i = 0; vnni_i < vnni_stride; vnni_i++) {
244 reg_dst_2d.xetla_select<remain_move_rows, vnni_stride,
245 block_size_x, 1>(vnni_i, 0)
246 = reg_2d.xetla_select<remain_move_rows, 1, block_size_x,
247 vnni_stride>(0, vnni_i);
248 }
249 }
250 }
251 mat_Acc.reg = rdst;
252}

340template <typename T_dst, typename T_src>
341__XETLA_API
342 typename std::enable_if_t<(T_src::block_size_y == T_dst::block_size_y)
343 && (T_src::block_size_x == T_dst::block_size_x)
344 && (T_src::tile_size_y == T_dst::tile_size_y)
345 && (T_src::tile_size_x == T_dst::tile_size_x)>
346 vnni_transform(T_dst &dst, T_src &src) {
347 constexpr uint32_t tile_size_y = T_dst::tile_size_y;
348 constexpr uint32_t tile_size_x = T_dst::tile_size_x;
349 constexpr uint32_t tile_elems = tile_size_y * tile_size_x;
350 constexpr uint32_t block_size_y = T_dst::block_size_y;
351 constexpr uint32_t block_size_x = T_dst::block_size_x;
352 constexpr uint32_t block_elems = block_size_y * block_size_x;
353 constexpr int32_t num_block_x = tile_size_x / block_size_x;
354 using dtype_dst = typename T_dst::dtype;
355 using dtype_src = typename T_src::dtype;
356 constexpr uint32_t vnni_row_src = sizeof(uint32_t) / sizeof(dtype_src);
357 constexpr uint32_t vnni_row_dst = sizeof(uint32_t) / sizeof(dtype_dst);
358 constexpr int32_t vnni_row
359 = vnni_row_src > vnni_row_dst ? vnni_row_src : vnni_row_dst;
360 static_assert(block_size_y % vnni_row == 0);
361 static_assert(tile_size_y % vnni_row == 0);
362 constexpr int32_t move_elems = vnni_row * block_size_x;
363 xetla_vector<dtype_dst, tile_elems> reg_src
364 = xetla_cvt<dtype_dst, dtype_src, tile_elems>(src.reg);
365 if constexpr (sizeof(dtype_src) == sizeof(dtype_dst)) {
366 dst.reg = reg_src;
367 return;
368 }
369 xetla_vector<dtype_dst, tile_elems> reg_dst;
370 constexpr uint32_t scale_factor
371 = detail::gcd<vnni_row_src, vnni_row_dst>::value;
372 using move_dtype = get_uint_type_t<sizeof(dtype_dst) * scale_factor>;
373 constexpr uint32_t select_stride = vnni_row / scale_factor;
374#pragma unroll
375 for (int i = 0; i < tile_size_y / block_size_y; i++) {
376#pragma unroll
377 for (int j = 0; j < num_block_x; j++) {
378 auto reg_src_blk = reg_src.xetla_select<block_elems, 1>(
379 (i * num_block_x + j) * block_elems);
380 auto reg_dst_blk = reg_dst.xetla_select<block_elems, 1>(
381 (i * num_block_x + j) * block_elems);
382 for (int row_i = 0; row_i < block_size_y; row_i += vnni_row) {
383 auto reg_src_move = reg_src_blk
384 .xetla_select<move_elems, 1>(
385 row_i * block_size_x)
386 .xetla_format<move_dtype>();
387 auto reg_dst_move = reg_dst_blk
388 .xetla_select<move_elems, 1>(
389 row_i * block_size_x)
390 .xetla_format<move_dtype>();
391#pragma unroll
392 for (int move_i = 0; move_i < select_stride; move_i++) {
393 if constexpr (sizeof(dtype_dst) > sizeof(dtype_src)) {
394 reg_dst_move.xetla_select<block_size_x, 1>(
395 move_i * block_size_x)
396 = reg_src_move.xetla_select<block_size_x,
397 select_stride>(move_i);
398 } else {
399 reg_dst_move.xetla_select<block_size_x, select_stride>(
400 move_i)
401 = reg_src_move.xetla_select<block_size_x, 1>(
402 move_i * block_size_x);
403 }
404 }
405 }
406 }
407 }
408 // process the tail
409 if constexpr ((tile_size_y % block_size_y) != 0) {
410 constexpr int i = tile_size_y / block_size_y;
411 constexpr uint32_t remain_elems_start = i * block_size_y * tile_size_x;
412 constexpr uint32_t remain_size_y = tile_size_y % block_size_y;
413 constexpr uint32_t remain_block_elems = remain_size_y * block_size_x;
414#pragma unroll
415 for (int j = 0; j < num_block_x; j++) {
416 auto reg_src_blk = reg_src.xetla_select<remain_block_elems, 1>(
417 remain_elems_start + j * remain_block_elems);
418 auto reg_dst_blk = reg_dst.xetla_select<remain_block_elems, 1>(
419 remain_elems_start + j * remain_block_elems);
420 // for mma, here we can guarantee that the remaining is a multiple of
421 // vnni_row
422 for (int row_i = 0; row_i < remain_size_y; row_i += vnni_row) {
423 auto reg_src_move = reg_src_blk
424 .xetla_select<move_elems, 1>(
425 row_i * block_size_x)
426 .xetla_format<move_dtype>();
427 auto reg_dst_move = reg_dst_blk
428 .xetla_select<move_elems, 1>(
429 row_i * block_size_x)
430 .xetla_format<move_dtype>();
431#pragma unroll
432 for (int move_i = 0; move_i < select_stride; move_i++) {
433 if constexpr (sizeof(dtype_dst) > sizeof(dtype_src)) {
434 reg_dst_move.xetla_select<block_size_x, 1>(
435 move_i * block_size_x)
436 = reg_src_move.xetla_select<block_size_x,
437 select_stride>(move_i);
438 } else {
439 reg_dst_move.xetla_select<block_size_x, select_stride>(
440 move_i)
441 = reg_src_move.xetla_select<block_size_x, 1>(
442 move_i * block_size_x);
443 }
444 }
445 }
446 }
447 }
448 dst.reg = reg_dst;
449}

458template <typename T_dst, typename T_src>
459__XETLA_API
460 typename std::enable_if_t<(T_dst::register_layout == reg_layout::tiled)
461 && (T_src::register_layout == reg_layout::tiled)
462 && (T_src::tile_size_x == T_dst::tile_size_x)
463 && (T_src::tile_size_y == 1)>
464 row_broadcast(T_dst &dst, T_src &src) {
465 static constexpr uint32_t dst_tile_size_y = T_dst::tile_size_y;
466 static constexpr uint32_t dst_tile_size_x = T_dst::tile_size_x;
467 static constexpr uint32_t dst_tile_elems = T_dst::tile_elems;
468 static constexpr uint32_t dst_block_size_y = T_dst::block_size_y;
469 static constexpr uint32_t dst_block_size_x = T_dst::block_size_x;
470 static constexpr uint32_t dst_block_elems = T_dst::block_elems;
471 static constexpr int32_t dst_num_block_y = T_dst::num_block_y;
472 static constexpr int32_t dst_num_block_x = T_dst::num_block_x;
473 using dst_dtype = typename T_dst::dtype;
474 using src_dtype = typename T_src::dtype;
475
476#pragma unroll
477 for (int i = 0; i < dst_tile_size_y / dst_block_size_y; i++) {
478#pragma unroll
479 for (int j = 0; j < dst_num_block_x; j++) {
480 auto dst_reg = (dst.reg)
481 .xetla_select<dst_block_elems, 1>(
482 (i * dst_num_block_x + j)
483 * dst_block_elems)
484 .xetla_format<dst_dtype, dst_block_size_y,
485 dst_block_size_x>();
486#pragma unroll
487 for (int row_i = 0; row_i < dst_block_size_y; row_i++) {
488 auto src_reg = src.reg.xetla_select<dst_block_size_x, 1>(
489 j * dst_block_size_x);
490 dst_reg.row(row_i)
491 = xetla_cvt<dst_dtype, src_dtype, dst_block_size_x>(
492 src_reg);
493 }
494 }
495 }
496
497 // process the tail
498 if constexpr ((dst_tile_size_y % dst_block_size_y) != 0) {
499 constexpr uint32_t tail_start_y
500 = dst_tile_size_y / dst_block_size_y * dst_block_size_y;
501 constexpr int32_t dst_tail_size_y = dst_tile_size_y % dst_block_size_y;
502 constexpr int32_t dst_tail_block_elems
503 = dst_tail_size_y * dst_block_size_x;
504#pragma unroll
505 for (int j = 0; j < dst_num_block_x; j++) {
506 auto dst_reg = (dst.reg)
507 .xetla_select<dst_tail_block_elems, 1>(
508 tail_start_y * dst_tile_size_x
509 + j * dst_tail_block_elems)
510 .xetla_format<dst_dtype, dst_tail_size_y,
511 dst_block_size_x>();
512#pragma unroll
513 for (int row_i = 0; row_i < dst_tail_size_y; row_i++) {
514 auto src_reg = src.reg.xetla_select<dst_block_size_x, 1>(
515 j * dst_block_size_x);
516 dst_reg.row(row_i)
517 = xetla_cvt<dst_dtype, src_dtype, dst_block_size_x>(
518 src_reg);
519 }
520 }
521 }
522}
