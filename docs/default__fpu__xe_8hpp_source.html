298 XETLA_MARKER(
299 "This release function will wait until all the r/w and nbarrier "
300 "id used in this brgemm have been committed. By default, it will "
301 "use barrier_id 0 to do the entire workgroup sync if wg_size > 1. "
302 "If you call this function, please set a free barrier id or make "
303 "sure barrier_id 0 is not being occupied and you need to allocate "
304 "one more barrier count in addition to the brgemm barrier counts.")
305 __XETLA_API static void release(uint8_t nbarrier_id = 0) {
306 static constexpr bool need_local_fence
307 = (mem_space_a == mem_space::local)
308 || (mem_space_b == mem_space::local);
309 if constexpr (need_local_fence) {
310 xetla_fence<memory_kind::shared_local>();
311 }
312 xetla_fence<memory_kind::untyped_global>();
313 static constexpr uint32_t wg_size = wg_size_x * wg_size_y;
314 if constexpr (wg_size > 1) {
315 xetla_nbarrier_t<wg_size, wg_size> nbarrier;
316 nbarrier.init_nbarrier(
317 nbarrier_id, nbarrier_role::producer_consumer);
318 nbarrier.arrive_wait();
319 }
320 }

329 __XETLA_API KERNEL_FUNC void operator()(work_group_t &g, matAcc_t &matAcc,
330 arguments_t args, uint32_t slm_base = 0,
331 uint32_t nbarrier_base = 0) {
332 int32_t sg_idx = g.get_id() % wg_size_x;
333 int32_t sg_idy = g.get_id() / wg_size_x;
334 update_sg_tile_tdesc(args, sg_idx, sg_idy);
335 pre_processing_t pre_processing;
336 matA_t matA;
337 matB_t matB;
338 // >>>>>>>>>>>>>>>>>> pre_processing init
339 pre_processing.init(g, args.pre_processing_args);
340 matA_payload_t matA_payload(args.matA_base_desc);
341 matB_payload_t matB_payload(args.matB_base_desc);
342 matA_prefetch_payload_t matA_prefetch_payload(args.matA_base_desc, 0);
343 matB_prefetch_payload_t matB_prefetch_payload(args.matB_base_desc, 0);
344 xetla_nbarrier_t<wg_size_x, wg_size_x> nbarrier_a;
345 nbarrier_a.init_nbarrier(
346 sg_idy + nbarrier_base, nbarrier_role::producer_consumer);
347 xetla_nbarrier_t<wg_size_y, wg_size_y> nbarrier_b;
348 nbarrier_b.init_nbarrier(sg_idx + barrier_count_y + nbarrier_base,
349 nbarrier_role::producer_consumer);
350 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
351 matA, matA_payload);
352 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
353 matB, matB_payload);
354#pragma unroll
355 for (int i = 0; i < stages; i++) {
356 matA_prefetch_payload.template update_tdesc<update_dir_a>(
357 matA_t::tile_size_x);
358 matB_prefetch_payload.template update_tdesc<update_dir_b>(
359 matB_t::tile_size_y);
360 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
361 matA_prefetch_payload);
362 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
363 matB_prefetch_payload);
364 }
365
366 for (int i = 0; i < args.inner_loop_count; i++) {
367 if constexpr (enable_periodic_sync) {
368 if ((i % sync_freq) == 0) {
369 if constexpr (wg_size_x > 1) { nbarrier_a.arrive(); }
370 if constexpr (wg_size_y > 1) { nbarrier_b.arrive(); }
371 }
372 }
373 SW_BARRIER();
374 matA_payload.template update_tdesc<update_dir_a>(
375 matA_t::tile_size_x);
376 matB_payload.template update_tdesc<update_dir_b>(
377 matB_t::tile_size_y);
378 if constexpr (stages != 0) {
379 matA_prefetch_payload.template update_tdesc<update_dir_a>(
380 matA_t::tile_size_x);
381 matB_prefetch_payload.template update_tdesc<update_dir_b>(
382 matB_t::tile_size_y);
383 }
384 SW_BARRIER();
385 matA_acc_t matA_acc;
386 matB_acc_t matB_acc;
387 pre_processing(matA_acc, matB_acc, matA, matB);
388 SW_BARRIER();
389 tile_mma::mma(matA_acc, matB_acc, matAcc, matAcc);
390 SW_BARRIER();
391 if (i != args.inner_loop_count - 1) {
392 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
393 matA, matA_payload);
394 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
395 matB, matB_payload);
396 SW_BARRIER();
397 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
398 matA_prefetch_payload);
399 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
400 matB_prefetch_payload);
401 }
402 if constexpr (enable_periodic_sync) {
403 if ((i % sync_freq) == 0) {
404 if constexpr (wg_size_x > 1) { nbarrier_a.wait(); }
405 if constexpr (wg_size_y > 1) { nbarrier_b.wait(); }
406 }
407 }
408 }
409 SW_BARRIER();
410 }
