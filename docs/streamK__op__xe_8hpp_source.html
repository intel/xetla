91 template <typename matAcc_t>
92 __XETLA_API KERNEL_FUNC void operator()(work_group_t &g, matAcc_t &matAcc,
93 mem_desc_c_t mem_desc_c, mem_desc_d_t mem_desc_d,
94 mem_desc_atomic_sync_t mem_desc_atomic_sync, int group_idx,
95 int first_group_idx, bool tile_finished, bool tile_started,
96 epilogue_args_t epilogue_args, uint32_t slm_base = 0,
97 uint32_t nbarrier_base = 0) {
98
99 static constexpr uint32_t tile_size_x = matAcc_t::tile_size_x;
100 static constexpr uint32_t tile_size_y = matAcc_t::tile_size_y;
101 static constexpr uint32_t block_size_x = matAcc_t::block_size_x;
102 static constexpr uint32_t block_size_y = matAcc_t::block_size_y;
103 static constexpr uint32_t tile_elems = matAcc_t::tile_elems;
104 static constexpr uint32_t block_elems = matAcc_t::block_elems;
105
106 using matD_tile_desc_t = subgroup::tile_desc_t<tile_size_x, tile_size_y,
107 block_size_x, block_size_y, reg_layout::tiled>;
108
109 using matD_t = subgroup::tile_t<dtype_d, matD_tile_desc_t>;
110 using matD_atomic_payload_t
111 = subgroup::mem_payload_t<dtype_d, matD_tile_desc_t,
112 msg_type_d_atomic, mem_layout_d, mem_space_d, arch_tag>;
113
114 uint32_t nbarrier_id = nbarrier_base;
115 nbarrier.init_nbarrier(nbarrier_id, nbarrier_role::producer_consumer);
116
117 update_sg_tile_tdesc(g, mem_desc_d);
118
119 //Addressing for atomic signal
120 xetla_mask<16> pred(0);
121 pred[0] = 1;
122 xetla_vector<uint32_t, 16> flag_offsets
123 = xetla_vector_gen<uint32_t, 16>(0, 1);
124 flag_offsets
125 += first_group_idx; // first_group_idx indicates the first peer of the sliced tile
126 flag_offsets = flag_offsets * sizeof(dtype_flag);
127 int32_t sg_id = g.get_id();
128 dtype_flag *flag_pointer = mem_desc_atomic_sync.base.base;
129
130 //SK group , Sliced Tile - SK group handles starting slice or middle slice
131 if (!tile_finished) {
132
133 //Perform atomic writes and signal to atomic counter
134 matD_atomic_payload_t matD_atomic_payload(mem_desc_d);
135 //Atomic store with OOB check
136 subgroup::tile_store(matAcc, matD_atomic_payload);
137
138 //Fence to guarantee write completion
139 xetla_fence<memory_kind::untyped_global, fence_op::evict,
140 fence_scope::tile>();
141
142 //Group sync to make sure fence is sent
143 nbarrier.arrive();
144 nbarrier.wait();
145
146 //Signal to other peers
147 if (sg_id == 0) {
148 xetla_vector<dtype_flag, 16> signal_val(1);
149 xetla_tatomic_store_global<dtype_flag, 16, cache_hint::uncached,
150 cache_hint::write_back, atomic_op::iadd>(
151 flag_offsets + (uint64_t)flag_pointer, signal_val,
152 pred);
153 }
154
155 } else {
156
157 //last SK group of corresponding sliced tile
158 if (!tile_started) {
159
160 //Number of previous peers that have contributed to this sliced tile
161 uint32_t num_peers = group_idx - first_group_idx;
162
163 //Group sync
164 nbarrier.arrive();
165 nbarrier.wait();
166
167 if (sg_id == 0) {
168
169 xetla_vector<dtype_flag, 16> ret_val(0);
170 xetla_vector<dtype_flag, 16> old_val = num_peers;
171 xetla_vector<dtype_flag, 16> zero_val(0);
172
173 //Use atomic cmpxchg to test if previous peers have finished writing
174 //Exchange with value zero to clear the flag
175 while (ret_val[0] != num_peers) {
176
177 ret_val = xetla_atomic_global<atomic_op::cmpxchg,
178 dtype_flag, 16, data_size::default_size,
179 cache_hint::uncached, cache_hint::write_back>(
180 flag_pointer, flag_offsets, old_val, zero_val,
181 pred);
182 }
183 }
184 //Group sync
185 nbarrier.arrive();
186 nbarrier.wait();
187
188 //Invoke streamK residual op
189 residual_op_t residual_op;
190 residual_op_args_t residual_args(
191 mem_desc_d.base, mem_desc_d.shape);
192
193 residual_op(matAcc, mem_desc_d.coord, residual_args);
194 }
195
196 //Finishing SK groups and DP Groups perform normal epilogue operations - post_op fusion + output conversion and write to output buffer
197 epilogue_t epilogue;
198 epilogue(g, matAcc, mem_desc_c, epilogue_args, slm_base,
199 nbarrier_base);
200 }
201 }
