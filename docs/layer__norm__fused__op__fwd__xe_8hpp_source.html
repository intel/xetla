201 __XETLA_API xetla_vector<dtype_acc, sg_tile_n> pre_op(
202 xetla_vector<dtype_acc, sg_tile_n> input) {
203 subgroup::tile_load(res_in, res_in_payload);
204 res_in_payload.update_tdesc(wg_num_m * wg_tile_m * mat_ld);
205 xetla_vector<dtype_acc, sg_tile_n> bias_input
206 = xetla_cvt<dtype_acc, dtype_in>(bias_in.reg);
207 // bias_add
208 xetla_vector<dtype_acc, sg_tile_n> output
209 = reduce_helper<reduce_op::sum, dtype_acc, sg_tile_n>(
210 input, bias_input);
211 if (dropout_prob != 0) {
212 // dropout
213 subgroup::tile_load(mask_in, mask_in_payload);
214 SW_BARRIER();
215 mask_in_payload.update_tdesc(wg_num_m * wg_tile_m * mask_ld);
216 output = drop_out<dtype_acc, sg_tile_n>(
217 output, mask_in.reg, dropout_scale);
218 }
219 // res_add, generate mixed mode
220 xetla_vector<dtype_acc, sg_tile_n> res_input
221 = xetla_cvt<dtype_acc, dtype_in>(res_in.reg);
222 output = reduce_helper<reduce_op::sum, dtype_acc, sg_tile_n>(
223 output, res_input);
224 bias_dropout_res_out.reg = xetla_cvt<dtype_out, dtype_acc>(output);
225 subgroup::tile_store<cache_hint::uncached>(
226 bias_dropout_res_out, bias_dropout_res_out_payload);
227 bias_dropout_res_out_payload.update_tdesc(
228 wg_num_m * wg_tile_m * mat_ld);
229 return output;
230 }
