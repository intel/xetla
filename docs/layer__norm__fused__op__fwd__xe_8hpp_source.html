203 __XETLA_API xetla_vector<dtype_acc, chunk_size> pre_op(
204 xetla_vector<dtype_acc, chunk_size> input) {
205 subgroup::tile_load(res_in, res_in_payload);
206 if constexpr (n_chunks == 1) {
207 res_in_payload.update_tdesc(wg_num_m * wg_tile_m * mat_ld);
208 } else {
209 res_in_payload.update_tdesc(chunk_size);
210 }
211 if constexpr (n_chunks != 1) {
212 subgroup::tile_load(bias_in, bias_in_payload);
213 bias_in_payload.update_tdesc(chunk_size);
214 }
215 xetla_vector<dtype_acc, chunk_size> bias_input
216 = xetla_cvt<dtype_acc, dtype_in>(bias_in.reg);
217 // bias_add
218 xetla_vector<dtype_acc, chunk_size> output
219 = reduce_helper<reduce_op::sum, dtype_acc, chunk_size>(
220 input, bias_input);
221 if (dropout_prob != 0) {
222 // dropout
223 subgroup::tile_load(mask_in, mask_in_payload);
224 SW_BARRIER();
225 if constexpr (n_chunks == 1) {
226 mask_in_payload.update_tdesc(wg_num_m * wg_tile_m * mask_ld);
227 } else {
228 mask_in_payload.update_tdesc(chunk_size);
229 }
230 output = drop_out<dtype_acc, chunk_size>(
231 output, mask_in.reg, dropout_scale);
232 }
233 // res_add, generate mixed mode
234 xetla_vector<dtype_acc, chunk_size> res_input
235 = xetla_cvt<dtype_acc, dtype_in>(res_in.reg);
236 output = reduce_helper<reduce_op::sum, dtype_acc, chunk_size>(
237 output, res_input);
238 bias_dropout_res_out.reg = xetla_cvt<dtype_out, dtype_acc>(output);
239 subgroup::tile_store<cache_hint::uncached>(
240 bias_dropout_res_out, bias_dropout_res_out_payload);
241 if constexpr (n_chunks == 1) {
242 bias_dropout_res_out_payload.update_tdesc(
243 wg_num_m * wg_tile_m * mat_ld);
244 } else {
245 bias_dropout_res_out_payload.update_tdesc(chunk_size);
246 }
247 return output;
248 }
