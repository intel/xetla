129 template <tdesc_update_dir update_dir = tdesc_update_dir::x_dir>
130 __XETLA_API void update_tdesc(int offset) {
131 auto payloads_2d = payloads.xetla_format<uint32_t, num_block, 16>();
132 if constexpr (update_dir == tdesc_update_dir::x_dir) {
133#pragma unroll
134 for (int i = 0; i < num_block; i++) {
135 xetla_update_tdesc_offsetx(
136 payloads_2d.row(i), offset / scale_factor);
137 }
138 } else {
139#pragma unroll
140 for (int i = 0; i < num_block; i++) {
141 xetla_update_tdesc_offsety(payloads_2d.row(i), offset);
142 }
143 }
144 }
145
146private:
147 __XETLA_API void prepare_tdesc(xetla_tdescriptor base_tdesc) {
148 auto payloads_2d = payloads.xetla_format<uint32_t, num_block, 16>();
149 uint32_t base_offset_y = 0;
150#pragma unroll
151 for (int i = 0; i < num_block_y; i++) {
152 auto tdesc_row_2d = payloads_2d.xetla_select<num_block_x, 1, 16, 1>(
153 i * num_block_x, 0);
154 prepare_tile_desc_core<num_block_x, block_size_x, block_size_y, 1,
155 mem_transpose>(tdesc_row_2d, base_tdesc, base_offset_y);
156 base_offset_y += block_size_y;
157 }
158 // process the tail
159 if constexpr (remained_size_y > 0) {
160 auto tdesc_row_2d = payloads_2d.xetla_select<num_block_x, 1, 16, 1>(
161 num_block_y * num_block_x, 0);
162 prepare_tile_desc_core<num_block_x, block_size_x, remained_size_y,
163 1, mem_transpose>(tdesc_row_2d, base_tdesc, base_offset_y);
164 }
165 }

167 template <uint32_t num_tdesc, uint32_t size_x, uint32_t size_y,
168 uint8_t arr_len, bool trans>
169 __XETLA_API static void prepare_tile_desc_core(
170 xetla_matrix_ref<uint32_t, num_tdesc, 16> __REF__ payloads_row_2d,
171 xetla_tdescriptor base_tdesc, uint32_t base_offset_y) {
172 uint32_t base_offset_x = 0;
173#pragma unroll
174 for (int j = 0; j < num_tdesc; j++) {
175 payloads_row_2d.row(j) = base_tdesc;
176 // To mimic dw transpose for word/byte data type with transpose and pack
177 constexpr uint8_t block_width
178 = trans ? (size_y / scale_factor) : size_x;
179 constexpr uint8_t block_height = trans ? size_x : size_y;
180 constexpr uint32_t block_widthx_widthy_arrlen = (block_width - 1)
181 | ((block_height - 1) << 8) | ((arr_len - 1) << 16);
182 gpu::xetla::detail::xetla_set_block_widthx_widthy_arrlen(
183 payloads_row_2d.row(j), block_widthx_widthy_arrlen);
184
185 // To mimic dw transpose for word/byte data type with transpose and pack
186 uint32_t offset_width
187 = trans ? (base_offset_y / scale_factor) : base_offset_x;
188 uint32_t offset_height = trans ? base_offset_x : base_offset_y;
189
190 xetla_update_tdesc_offsetx(payloads_row_2d.row(j), offset_width);
191 xetla_update_tdesc_offsety(payloads_row_2d.row(j), offset_height);
192 base_offset_x += size_x * arr_len;
193 }
194 }

999 uint32_t base_offset_y = 0;
1000#pragma unroll
1001 for (int i = 0; i < tile_size_h / max_block_size_h; i++) {
1002 auto tdesc_row_2d = tdesc_2d.xetla_select<num_block_w, 1, 16, 1>(
1003 i * num_block_w, 0);
1004 prepare_tile_desc_core<num_block_w, max_block_size_w,
1005 max_block_size_h>(tdesc_row_2d, base_tdesc, base_offset_y);
1006 base_offset_y += max_block_size_h;
1007 }
1008 if constexpr ((tile_size_h % max_block_size_h) != 0) {
1009 constexpr int i = tile_size_h / max_block_size_h;
1010 auto tdesc_row_2d = tdesc_2d.xetla_select<num_block_w, 1, 16, 1>(
1011 i * num_block_w, 0);
1012 constexpr uint32_t remain_size_y = tile_size_h % max_block_size_h;

1014 remain_size_y>(tdesc_row_2d, base_tdesc, base_offset_y);

1030 tdesc_2d.row(j), block_widthx_widthy_arrlen);

1032 xetla_update_tdesc_offsetx(tdesc_2d.row(j), base_offset_x);
1033 xetla_update_tdesc_offsety(tdesc_2d.row(j), base_offset_y);
