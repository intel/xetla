132 template <tdesc_update_dir update_dir = tdesc_update_dir::x_dir>
133 __XETLA_API void update_tdesc(int offset) {
134 auto payloads_2d = payloads.xetla_format<uint32_t, num_block, 16>();
135 if constexpr (update_dir == tdesc_update_dir::x_dir) {
136#pragma unroll
137 for (int i = 0; i < num_block; i++) {
138 xetla_update_tdesc_offsetx(
139 payloads_2d.row(i), offset / scale_factor);
140 }
141 } else {
142#pragma unroll
143 for (int i = 0; i < num_block; i++) {
144 xetla_update_tdesc_offsety(payloads_2d.row(i), offset);
145 }
146 }
147 }
148
149private:
150 __XETLA_API void prepare_tdesc(xetla_tdescriptor base_tdesc) {
151 auto payloads_2d = payloads.xetla_format<uint32_t, num_block, 16>();
152 uint32_t base_offset_y = 0;
153#pragma unroll
154 for (int i = 0; i < num_block_y; i++) {
155 auto tdesc_row_2d = payloads_2d.xetla_select<num_block_x, 1, 16, 1>(
156 i * num_block_x, 0);
157 prepare_tile_desc_core<num_block_x, block_size_x, block_size_y, 1,
158 mem_transpose>(tdesc_row_2d, base_tdesc, base_offset_y);
159 base_offset_y += block_size_y;
160 }
161 // process the tail
162 if constexpr (remained_size_y > 0) {
163 auto tdesc_row_2d = payloads_2d.xetla_select<num_block_x, 1, 16, 1>(
164 num_block_y * num_block_x, 0);
165 prepare_tile_desc_core<num_block_x, block_size_x, remained_size_y,
166 1, mem_transpose>(tdesc_row_2d, base_tdesc, base_offset_y);
167 }
168 }

170 template <uint32_t num_tdesc, uint32_t size_x, uint32_t size_y,
171 uint8_t arr_len, bool trans>
172 __XETLA_API static void prepare_tile_desc_core(
173 xetla_matrix_ref<uint32_t, num_tdesc, 16> __REF__ payloads_row_2d,
174 xetla_tdescriptor base_tdesc, uint32_t base_offset_y) {
175 uint32_t base_offset_x = 0;
176#pragma unroll
177 for (int j = 0; j < num_tdesc; j++) {
178 payloads_row_2d.row(j) = base_tdesc;
179 // To mimic dw transpose for word/byte data type with transpose and pack
180 constexpr uint8_t block_width
181 = trans ? (size_y / scale_factor) : (size_x / scale_factor);
182 constexpr uint8_t block_height = trans ? size_x : size_y;
183 constexpr uint32_t block_widthx_widthy_arrlen = (block_width - 1)
184 | ((block_height - 1) << 8) | ((arr_len - 1) << 16);
185 gpu::xetla::detail::xetla_set_block_widthx_widthy_arrlen(
186 payloads_row_2d.row(j), block_widthx_widthy_arrlen);
187
188 // To mimic dw transpose for word/byte data type with transpose and pack
189 uint32_t offset_width = trans ? (base_offset_y / scale_factor)
190 : (base_offset_x / scale_factor);
191 uint32_t offset_height = trans ? base_offset_x : base_offset_y;
192
193 xetla_update_tdesc_offsetx(payloads_row_2d.row(j), offset_width);
194 xetla_update_tdesc_offsety(payloads_row_2d.row(j), offset_height);
195 base_offset_x += size_x * arr_len;
196 }
197 }

963#pragma unroll

980 base_offset_y += max_block_size_h;

983 constexpr int i = tile_size_h / max_block_size_h;

1007 xetla_update_tdesc_offsety(tdesc_2d.row(j), base_offset_y);
