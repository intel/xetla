134 template <tdesc_update_dir update_dir = tdesc_update_dir::x_dir>
135 __XETLA_API void update_tdesc(int offset) {
136 auto payloads_2d = payloads.xetla_format<uint32_t, num_block, 16>();
137 if constexpr (update_dir == tdesc_update_dir::x_dir) {
138#pragma unroll
139 for (int i = 0; i < num_block; i++) {
140 xetla_update_tdesc_offsetx(
141 payloads_2d.row(i), offset / scale_factor);
142 }
143 } else {
144#pragma unroll
145 for (int i = 0; i < num_block; i++) {
146 xetla_update_tdesc_offsety(payloads_2d.row(i), offset);
147 }
148 }
149 }
150
151private:
152 __XETLA_API void prepare_tdesc(xetla_tdescriptor base_tdesc) {
153 auto payloads_2d = payloads.xetla_format<uint32_t, num_block, 16>();
154 uint32_t base_offset_y = 0;
155#pragma unroll
156 for (int i = 0; i < num_block_y; i++) {
157 auto tdesc_row_2d = payloads_2d.xetla_select<num_block_x, 1, 16, 1>(
158 i * num_block_x, 0);
159 prepare_tile_desc_core<num_block_x, block_size_x, block_size_y, 1,
160 mem_transpose>(tdesc_row_2d, base_tdesc, base_offset_y);
161 base_offset_y += block_size_y;
162 }
163 // process the tail
164 if constexpr (remained_size_y > 0) {
165 auto tdesc_row_2d = payloads_2d.xetla_select<num_block_x, 1, 16, 1>(
166 num_block_y * num_block_x, 0);
167 prepare_tile_desc_core<num_block_x, block_size_x, remained_size_y,
168 1, mem_transpose>(tdesc_row_2d, base_tdesc, base_offset_y);
169 }
170 }

172 template <uint32_t num_tdesc, uint32_t size_x, uint32_t size_y,
173 uint8_t arr_len, bool trans>
174 __XETLA_API static void prepare_tile_desc_core(
175 xetla_matrix_ref<uint32_t, num_tdesc, 16> __REF__ payloads_row_2d,
176 xetla_tdescriptor base_tdesc, uint32_t base_offset_y) {
177 uint32_t base_offset_x = 0;
178#pragma unroll
179 for (int j = 0; j < num_tdesc; j++) {
180 payloads_row_2d.row(j) = base_tdesc;
181 // To mimic dw transpose for word/byte data type with transpose and pack
182 constexpr uint8_t block_width
183 = trans ? (size_y / scale_factor) : (size_x / scale_factor);
184 constexpr uint8_t block_height = trans ? size_x : size_y;
185 constexpr uint32_t block_widthx_widthy_arrlen = (block_width - 1)
186 | ((block_height - 1) << 8) | ((arr_len - 1) << 16);
187 gpu::xetla::detail::xetla_set_block_widthx_widthy_arrlen(
188 payloads_row_2d.row(j), block_widthx_widthy_arrlen);
189
190 // To mimic dw transpose for word/byte data type with transpose and pack
191 uint32_t offset_width = trans ? (base_offset_y / scale_factor)
192 : (base_offset_x / scale_factor);
193 uint32_t offset_height = trans ? base_offset_x : base_offset_y;
194
195 xetla_update_tdesc_offsetx(payloads_row_2d.row(j), offset_width);
196 xetla_update_tdesc_offsety(payloads_row_2d.row(j), offset_height);
197 base_offset_x += size_x * arr_len;
198 }
199 }
