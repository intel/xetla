299 XETLA_MARKER(
300 "This release function will wait until all the r/w and nbarrier "
301 "id used in this brgemm have been committed. By default, it will "
302 "use barrier_id 0 to do the entire workgroup sync if wg_size > 1. "
303 "If you call this function, please set a free barrier id or make "
304 "sure barrier_id 0 is not being occupied and you need to allocate "
305 "one more barrier count in addition to the brgemm barrier counts.")
306 __XETLA_API static void release(uint8_t nbarrier_id = 0) {
307 static constexpr bool need_local_fence
308 = (mem_space_a == mem_space::local)
309 || (mem_space_b == mem_space::local);
310 if constexpr (need_local_fence) {
311 xetla_fence<memory_kind::shared_local>();
312 }
313 xetla_fence<memory_kind::untyped_global>();
314 static constexpr uint32_t wg_size = wg_size_x * wg_size_y;
315 if constexpr (wg_size > 1) {
316 xetla_nbarrier_t<wg_size, wg_size> nbarrier;
317 nbarrier.init_nbarrier(
318 nbarrier_id, nbarrier_role::producer_consumer);
319 nbarrier.arrive_wait();
320 }
321 }

330 __XETLA_API KERNEL_FUNC void operator()(work_group_t &g, matAcc_t &matAcc,
331 arguments_t args, uint32_t slm_base = 0,
332 uint32_t nbarrier_base = 0) {
333 int32_t sg_idx = g.get_id() % wg_size_x;
334 int32_t sg_idy = g.get_id() / wg_size_x;
335 update_sg_tile_tdesc(args, sg_idx, sg_idy);
336
337 matA_t matA;
338 matB_t matB;
339 scale_t scale;
340 zero_pt_t zero_pt;
341
342 matA_payload_t matA_payload(args.matA_base_desc);
343 matB_payload_t matB_payload(args.matB_base_desc);
344 scale_payload_t scale_payload(args.scale_base_desc);
345 zero_pt_payload_t zero_pt_payload(args.zero_pt_base_desc);
346 matA_prefetch_payload_t matA_prefetch_payload(
347 args.matA_base_desc, sg_idx);
348 matB_prefetch_payload_t matB_prefetch_payload(
349 args.matB_base_desc, sg_idy);
350 scale_prefetch_payload_t scale_prefetch_payload(
351 args.scale_base_desc, 0);
352 zero_pt_prefetch_payload_t zero_pt_prefetch_payload(
353 args.zero_pt_base_desc, 0);
354
355 xetla_nbarrier_t<wg_size_x, wg_size_x> nbarrier_a;
356 nbarrier_a.init_nbarrier(
357 sg_idy + nbarrier_base, nbarrier_role::producer_consumer);
358 xetla_nbarrier_t<wg_size_y, wg_size_y> nbarrier_b;
359 nbarrier_b.init_nbarrier(sg_idx + barrier_count_y + nbarrier_base,
360 nbarrier_role::producer_consumer);
361
362 int scale_prefetch_addr_i = 0;
363 int scale_load_addr_i = 0;
364 SW_BARRIER();
365#pragma unroll
366 for (int i = 0; i < stages; i++) {
367 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
368 matA_prefetch_payload);
369 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
370 matB_prefetch_payload);
371 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
372 scale_prefetch_payload);
373 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
374 zero_pt_prefetch_payload);
375 scale_prefetch_addr_i++;
376 matA_prefetch_payload.template update_tdesc<update_dir_a>(
377 matA_t::tile_size_x);
378 matB_prefetch_payload.template update_tdesc<update_dir_b>(
379 matB_t::tile_size_y);
380 if ((scale_prefetch_addr_i % scale_addr_update_freq) == 0) {
381 scale_prefetch_payload
382 .template update_tdesc<tdesc_update_dir::y_dir>(
383 scale_t::tile_size_y);
384 zero_pt_prefetch_payload
385 .template update_tdesc<tdesc_update_dir::y_dir>(
386 zero_pt_t::tile_size_y);
387 }
388 }
389
390 for (int i = 0; i < args.inner_loop_count; i++) {
391 if constexpr (enable_periodic_sync) {
392 if ((i % sync_freq) == 0) {
393 if constexpr (wg_size_x > 1) { nbarrier_a.arrive(); }
394 if constexpr (wg_size_y > 1) { nbarrier_b.arrive(); }
395 }
396 }
397 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
398 matA, matA_payload);
399 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
400 matB, matB_payload);
401 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
402 scale, scale_payload);
403 subgroup::tile_load<cache_hint::cached, cache_hint::cached>(
404 zero_pt, zero_pt_payload);
405 scale_load_addr_i++;
406 SW_BARRIER();
407 if constexpr (stages != 0) {
408 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
409 matA_prefetch_payload);
410 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
411 matB_prefetch_payload);
412 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
413 scale_prefetch_payload);
414 subgroup::tile_prefetch<cache_hint::cached, cache_hint::cached>(
415 zero_pt_prefetch_payload);
416 scale_prefetch_addr_i++;
417 }
418 SW_BARRIER();
419 matA_payload.template update_tdesc<update_dir_a>(
420 matA_t::tile_size_x);
421 matB_payload.template update_tdesc<update_dir_b>(
422 matB_t::tile_size_y);
423 if ((scale_load_addr_i % scale_addr_update_freq) == 0) {
424 scale_payload.template update_tdesc<tdesc_update_dir::y_dir>(
425 scale_t::tile_size_y);
426 zero_pt_payload.template update_tdesc<tdesc_update_dir::y_dir>(
427 zero_pt_t::tile_size_y);
428 }
429 if constexpr (stages != 0) {
430 matA_prefetch_payload.template update_tdesc<update_dir_a>(
431 matA_t::tile_size_x);
432 matB_prefetch_payload.template update_tdesc<update_dir_b>(
433 matB_t::tile_size_y);
434 if ((scale_prefetch_addr_i % scale_addr_update_freq) == 0) {
435 scale_prefetch_payload
436 .template update_tdesc<tdesc_update_dir::y_dir>(
437 scale_t::tile_size_y);
438 zero_pt_prefetch_payload
439 .template update_tdesc<tdesc_update_dir::y_dir>(
440 zero_pt_t::tile_size_y);
441 }
442 }
443 SW_BARRIER();
444 matA_acc_t matA_acc;
445 matB_acc_t matB_acc;
446 if constexpr (is_vnni_tiled_a) { subgroup::vnni_reverse(matA); }
447 subgroup::elemwise_cvt(matA_acc, matA);
448 dequantize(matB_acc, matB, scale, zero_pt);
449 SW_BARRIER();
450 tile_mma::mma(matAcc, matAcc, matB_acc, matA_acc);
451 SW_BARRIER();
452 if constexpr (enable_periodic_sync) {
453 if ((i % sync_freq) == 0) {
454 if constexpr (wg_size_x > 1) { nbarrier_a.wait(); }
455 if constexpr (wg_size_y > 1) { nbarrier_b.wait(); }
456 }
457 }
458 }
459 SW_BARRIER();
460 }

463 inline void dequantize(matB_acc_t &matB_acc, matB_t &matB, scale_t &scale,
464 zero_pt_t &zero_pt) {
465 //no tail, because this is matB
466 constexpr uint32_t num_block_x = tile_size_x_b / block_size_x_b;
467 constexpr uint32_t num_block_y = tile_size_y_b / block_size_y_b;
468 constexpr uint32_t vnni_rows = sizeof(uint32_t) / sizeof(dtype_mma_b);
469 constexpr uint32_t block_b_y_per_scale = dequant_s / block_size_y_b;
470#pragma unroll
471 for (int i = 0; i < num_block_y; ++i) {
472#pragma unroll
473 for (int j = 0; j < num_block_x; ++j) {
474 int block_id = (i * num_block_x + j);
475 auto matB_blk = matB.reg.xetla_select<matB_t::block_elems, 1>(
476 block_id * matB_t::block_elems)
477 .xetla_format<uint8_t>();
478 int scale_block_id
479 = (i / block_b_y_per_scale * num_block_x + j);
480 auto scale_vec
481 = scale.reg.xetla_select<scale_t::block_size_x, 1>(
482 scale_block_id * scale_t::block_size_x);
483 auto zero_pt_vec
484 = zero_pt.reg
485 .xetla_select<zero_pt_t::block_size_x, 1>(
486 scale_block_id
487 * zero_pt_t::block_size_x)
488 .xetla_format<uint8_t>();
489
490 auto dst_blk
491 = matB_acc.reg.xetla_select<matB_acc_t::block_elems, 1>(
492 block_id * matB_acc_t::block_elems);
493
494 xetla_vector<uint8_t, block_size_x_b> zero_pt_sub;
495 //2: int8 includes 2 4bits data.
496 zero_pt_sub.xetla_select<block_size_x_b / 2, 2>(0)
497 = zero_pt_vec & 0x0f;
498 zero_pt_sub.xetla_select<block_size_x_b / 2, 2>(1)
499 = zero_pt_vec >> 4;
500
501 xetla_vector<uint8_t, block_size_x_b * block_size_y_b>
502 zero_pt_blk;
503#pragma unroll
504 for (int row = 0; row < block_size_y_b; row++) {
505 zero_pt_blk
506 .xetla_select<block_size_x_b, 1>(
507 row * block_size_x_b)
508 .xetla_format<int8_t>()
509 = zero_pt_sub.xetla_format<int8_t>() + int8_t(1);
510 }
511
512 xetla_vector<uint8_t, block_size_x_b * block_size_y_b> cvt_blk;
513 cvt_blk.xetla_select<matB_t::block_elems, 2>(0)
514 = matB_blk & 0x0f;
515 cvt_blk.xetla_select<matB_t::block_elems, 2>(1) = matB_blk >> 4;
516
517 xetla_vector<int32_t, block_size_x_b * block_size_y_b>
518 cvt_blk_i32;
519
520 cvt_blk_i32 = (cvt_blk.xetla_format<int8_t>()
521 - zero_pt_blk.xetla_format<int8_t>());
522
523 xetla_vector<dtype_mma_b, matB_acc_t::block_elems * vnni_rows>
524 temp_blk;
525 temp_blk.xetla_select<matB_acc_t::block_elems, vnni_rows>(0)
526 = cvt_blk_i32;
527
528#pragma unroll
529 for (int k = 0; k < block_size_y_b; k += vnni_rows) {
530#pragma unroll
531 for (int row = 0; row < vnni_rows; row++) {
532 temp_blk.xetla_select<block_size_x_b, vnni_rows>(
533 row + block_size_x_b * k * vnni_rows)
534 = temp_blk.xetla_select<block_size_x_b,
535 vnni_rows>(
536 (k + row) * block_size_x_b * vnni_rows);
537 }
538 }
539
540 xetla_vector<dtype_scale, block_size_x_b * vnni_rows> scale_blk;
541#pragma unroll
542 for (int row = 0; row < vnni_rows; row++) {
543 scale_blk.xetla_select<block_size_x_b, vnni_rows>(row)
544 = scale_vec;
545 }
546
547#pragma unroll
548 for (int k = 0; k < block_size_y_b; k += vnni_rows) {
549 dst_blk.xetla_select<block_size_x_b * vnni_rows, 1>(
550 k * block_size_x_b)
551 = temp_blk.xetla_select<block_size_x_b * vnni_rows,
552 1>(k * block_size_x_b * vnni_rows)
553 * scale_blk;
554 }
555 }
556 }
557 }

566 args.scale_base_desc.update_coord_x(tile_offset_n);
567 args.zero_pt_base_desc.update_coord_x(tile_offset_n / pack_ratio);
568 }
