27template <typename matA_t_, typename matB_t_, typename matAcc_src_t_,
28 typename matAcc_dst_t_>
29struct tile_mma_t<matA_t_, matB_t_, matAcc_src_t_, matAcc_dst_t_,
30 mma_engine::fpu, gpu_arch::Xe> {
31 using matA_t = matA_t_;
32 using matB_t = matB_t_;
33 using matSrc_t = matAcc_src_t_;
34 using matDst_t = matAcc_dst_t_;
35 using dtype_a = typename matA_t::dtype;
36 using dtype_b = typename matB_t::dtype;
37 using dtype_src = typename matSrc_t::dtype;
38 using dtype_dst = typename matDst_t::dtype;
39
40 using arch_attr = arch_attr_t<gpu_arch::Xe>;
41 using register_attr = typename arch_attr::register_attr;
42
43 static_assert(matA_t::tile_desc::reg_transpose,
44 "For FMAOp GEMM, the register layout of matA should be col-major");
45 static_assert(!matB_t::tile_desc::reg_transpose,
46 "For FMAOp GEMM, the register layout of matB should be row-major");
47
48 static constexpr uint32_t a_tile_w = matA_t::tile_desc::tile_size_y;
49 static constexpr uint32_t a_tile_h = matA_t::tile_desc::tile_size_x;
50 static constexpr uint32_t a_tile_elems = matA_t::tile_desc::tile_elems;
51 static constexpr uint32_t a_block_w = matA_t::tile_desc::block_size_y;
52 static constexpr uint32_t a_block_h = matA_t::tile_desc::block_size_x;
53 static constexpr uint32_t a_block_elems = matA_t::tile_desc::block_elems;
54
55 static constexpr uint32_t b_tile_w = matB_t::tile_desc::tile_size_x;
56 static constexpr uint32_t b_tile_h = matB_t::tile_desc::tile_size_y;
57 static constexpr uint32_t b_tile_elems = matB_t::tile_desc::tile_elems;
58 static constexpr uint32_t b_block_w = matB_t::tile_desc::block_size_x;
59 static constexpr uint32_t b_block_h = matB_t::tile_desc::block_size_y;
60 static constexpr uint32_t b_block_elems = matB_t::tile_desc::block_elems;
61
62 static constexpr uint32_t tile_size_m = matDst_t::tile_desc::tile_size_y;
63 static constexpr uint32_t tile_size_k = a_tile_h;
64 static constexpr uint32_t tile_size_n = matDst_t::tile_desc::tile_size_x;
65 static constexpr uint32_t tile_elems = tile_size_m * tile_size_n;
66 static constexpr uint32_t block_size_n = matDst_t::tile_desc::block_size_x;
67 static constexpr uint32_t block_size_m = matDst_t::tile_desc::block_size_y;
68 static constexpr uint32_t block_elems = block_size_m * block_size_n;
69 // in case of padding
70 static_assert(tile_size_m <= matA_t::tile_desc::tile_size_y,
71 "matAcc tile m should match with matA tile m");
72 static_assert(
73 a_tile_h == b_tile_h, "matA tile k should match with matB tile k");
74 static_assert(tile_size_n == matB_t::tile_desc::tile_size_x,
75 "matAcc tile n should match with matB tile n");
76 static_assert(block_size_m == a_block_w,
77 "matAcc block m should match with matA block m");
78 static_assert(block_size_n == b_block_w,
79 "matAcc block n should match with matB block n");
80 static_assert((a_tile_h == a_block_h) && (b_tile_h == a_block_h),
81 "For SGEMM, no split along k dim for every inner loop");
82
83 static constexpr int32_t num_block_n = matDst_t::tile_desc::num_block_x;
84 static constexpr int32_t num_block_m = matDst_t::tile_desc::num_block_y;
85 static constexpr int32_t num_block = num_block_m * num_block_n;
86
87 static constexpr int32_t num_acc = register_attr::acc_reg_in_bytes
88 / (block_size_n * sizeof(dtype_dst));
89 static_assert(block_size_m % num_acc == 0,
90 "block_size_m should be a multiple of num_acc");
91
92 __XETLA_API static void mma(
93 matA_t &a, matB_t &b, matSrc_t &src, matDst_t &dst) {
94
95#pragma unroll
96 for (int i = 0; i < tile_size_m / block_size_m; i++) {
97 xetla_vector<dtype_a, a_block_elems> a_subtile
98 = a.reg.xetla_select<a_block_elems, 1>(i * a_block_elems);
99#pragma unroll
100 for (int j = 0; j < num_block_n; j++) {
101 auto b_subtile = b.reg.xetla_select<b_block_elems, 1>(
102 j * b_block_elems);
103 auto src_subtile = src.reg.xetla_select<block_elems, 1>(
104 (i * num_block_n + j) * block_elems);
105 auto dst_subtile = dst.reg.xetla_select<block_elems, 1>(
106 (i * num_block_n + j) * block_elems);
107 auto dst_2d = dst_subtile.xetla_format<dtype_dst, block_size_m,
108 block_size_n>();
109 auto src_2d = src_subtile.xetla_format<dtype_src, block_size_m,
110 block_size_n>();
111 auto b_2d = b_subtile.xetla_format<dtype_b, b_block_h,
112 b_block_w>();
113#pragma unroll
114 for (int sub_m = 0; sub_m < block_size_m; sub_m += num_acc) {
115#pragma unroll
116 for (int i_acc = 0; i_acc < num_acc; i_acc++) {
117 xetla_vector<dtype_dst, block_size_n> acc_tmp;
118 acc_tmp = a_subtile[i_acc + sub_m] * b_2d.row(0)
119 + src_2d.row(i_acc + sub_m);
120#pragma unroll
121 for (int k = 1; k < tile_size_k - 1; k++) {
122 int a_offset = k * a_block_w + i_acc + sub_m;
123 acc_tmp += a_subtile[a_offset] * b_2d.row(k);
124 }
125 {
126 int a_offset = (tile_size_k - 1) * a_block_w + i_acc
127 + sub_m;
128 dst_2d.row(i_acc + sub_m) = acc_tmp
129 + a_subtile[a_offset]
130 * b_2d.row(tile_size_k - 1);
131 }
132 }
133
134 SW_BARRIER();
135 }
136 }
137 }
138 // process the tail
139 if constexpr ((tile_size_m % block_size_m) != 0) {
140 constexpr uint32_t tail_start_m
141 = block_size_m * (tile_size_m / block_size_m);
142 constexpr uint32_t a_subtile_w = a_tile_w - tail_start_m;
143 constexpr uint32_t a_subtile_elems = a_block_h * a_subtile_w;
144 constexpr uint32_t acc_subtile_m = tile_size_m - tail_start_m;
145 constexpr uint32_t acc_subtile_elems = acc_subtile_m * block_size_n;
146 xetla_vector<dtype_a, a_subtile_elems> a_subtile
147 = a.reg.xetla_select<a_subtile_elems, 1>(
148 a_block_h * tail_start_m);
149#pragma unroll
150 for (int j = 0; j < num_block_n; j++) {
151 auto b_subtile = b.reg.xetla_select<b_block_elems, 1>(
152 j * b_block_elems);
153 auto src_subtile = src.reg.xetla_select<acc_subtile_elems, 1>(
154 (tail_start_m * tile_size_n) + j * acc_subtile_elems);
155 auto dst_subtile = dst.reg.xetla_select<acc_subtile_elems, 1>(
156 (tail_start_m * tile_size_n) + j * acc_subtile_elems);
157 auto dst_2d = dst_subtile.xetla_format<dtype_dst, acc_subtile_m,
158 block_size_n>();
159 auto src_2d = src_subtile.xetla_format<dtype_src, acc_subtile_m,
160 block_size_n>();
161 auto b_2d = b_subtile.xetla_format<dtype_b, b_block_h,
162 b_block_w>();
163#pragma unroll
164 for (int sub_m = 0; sub_m <= (int)acc_subtile_m - num_acc;
165 sub_m += num_acc) {
166#pragma unroll
167 for (int i_acc = 0; i_acc < num_acc; i_acc++) {
168 xetla_vector<dtype_dst, block_size_n> acc_tmp;
169 acc_tmp = a_subtile[i_acc + sub_m] * b_2d.row(0)
170 + src_2d.row(i_acc + sub_m);
171#pragma unroll
172 for (int k = 1; k < tile_size_k - 1; k++) {
173 int a_offset = k * a_subtile_w + i_acc + sub_m;
174 acc_tmp += a_subtile[a_offset] * b_2d.row(k);
175 }
176 {
177 int a_offset = (tile_size_k - 1) * a_subtile_w
178 + i_acc + sub_m;
179 dst_2d.row(i_acc + sub_m) = acc_tmp
180 + a_subtile[a_offset]
181 * b_2d.row(tile_size_k - 1);
182 }
183 }
184 SW_BARRIER();
185 }
186 }
187 SW_BARRIER();
188 //process the tail, change the processing order
189 if constexpr ((tile_size_m % num_acc) != 0) {
190 constexpr uint32_t remain_num_acc = tile_size_m % num_acc;
191 constexpr uint32_t remain_start_m
192 = tile_size_m / num_acc * num_acc;
193 constexpr uint32_t tail_remain_offset
194 = remain_start_m - tail_start_m;
195#pragma unroll
196 for (int j = 0; j < num_block_n; j++) {
197
198 auto b_subtile = b.reg.xetla_select<b_block_elems, 1>(
199 j * b_block_elems);
200 auto src_subtile
201 = src.reg.xetla_select<acc_subtile_elems, 1>(
202 (tail_start_m * tile_size_n)
203 + j * acc_subtile_elems);
204 auto dst_subtile
205 = dst.reg.xetla_select<acc_subtile_elems, 1>(
206 (tail_start_m * tile_size_n)
207 + j * acc_subtile_elems);
208 auto dst_2d = dst_subtile.xetla_format<dtype_dst,
209 acc_subtile_m, block_size_n>();
210 auto src_2d = src_subtile.xetla_format<dtype_src,
211 acc_subtile_m, block_size_n>();
212 auto b_2d = b_subtile.xetla_format<dtype_b, b_block_h,
213 b_block_w>();
214#pragma unroll
215 for (int i_acc = 0; i_acc < remain_num_acc; i_acc++) {
216 uint32_t m_index = i_acc + tail_remain_offset;
217 xetla_vector<dtype_dst, block_size_n> acc_tmp;
218 acc_tmp = a_subtile[m_index] * b_2d.row(0)
219 + src_2d.row(m_index);
220#pragma unroll
221 for (int k = 1; k < tile_size_k - 1; k++) {
222 int a_offset = k * a_subtile_w + m_index;
223 acc_tmp += a_subtile[a_offset] * b_2d.row(k);
224 }
225 {
226 constexpr int kk = tile_size_k - 1;
227 int a_offset = kk * a_subtile_w + m_index;
228 dst_2d.row(m_index) = acc_tmp
229 + a_subtile[a_offset] * b_2d.row(kk);
230 }
231 }
232 }
233 }
234 SW_BARRIER();
235 }
236 }
237};
