91 template <int blk_m, int blk_n, int blk_k>
92 __XETLA_API static void mma_core(
93 xetla_vector_ref<dtype_dst, blk_m * blk_n> __REF__ dst,
94 xetla_vector_ref<dtype_src, blk_m * blk_n> __REF__ src,
95 xetla_vector_ref<dtype_b, blk_k * blk_n> __REF__ b_block,
96 xetla_vector_ref<dtype_a, blk_m * blk_k> __REF__ a_block) {
97 auto dst_blk_2d = dst.xetla_format<dtype_dst, blk_m, blk_n>();
98 auto b_blk_2d = b_block.xetla_format<dtype_dst, blk_k, blk_n>();
99 auto src_blk_2d = src.xetla_format<dtype_src, blk_m, blk_n>();
100#pragma unroll
101 for (int i = 0; i < blk_m / mma_m; i++) {
102 xetla_vector<dtype_dst, mma_m * blk_n> dst_tmp;
103 auto dst_tmp_2d = dst_tmp.xetla_format<dtype_dst, mma_m, blk_n>();
104#pragma unroll
105 for (int i_acc = 0; i_acc < mma_m; i_acc++) {
106 dst_tmp_2d.row(i_acc)
107 = a_block[i_acc + i * mma_m] * b_blk_2d.row(0)
108 + src_blk_2d.row(i_acc + i * mma_m);
109 }
110#pragma unroll
111 for (int k = 1; k < blk_k - 1; k++) {
112 for (int i_acc = 0; i_acc < mma_m; i_acc++) {
113 int a_offset = k * blk_m + i_acc + i * mma_m;
114 dst_tmp_2d.row(i_acc)
115 += a_block[a_offset] * b_blk_2d.row(k);
116 }
117 }
118 for (int i_acc = 0; i_acc < mma_m; i_acc++) {
119 int a_offset = (blk_k - 1) * blk_m + i_acc + i * mma_m;
120 dst_blk_2d.row(i_acc + i * mma_m)
121 = a_block[a_offset] * b_blk_2d.row(blk_k - 1)
122 + dst_tmp_2d.row(i_acc);
123 }
124 SW_BARRIER();
125 }
126
127 if constexpr ((blk_m % mma_m) != 0) {
128 constexpr uint32_t tail_start_m = blk_m / mma_m * mma_m;
129 constexpr uint32_t tail_m = blk_m % mma_m;
130 xetla_vector<dtype_dst, tail_m * blk_n> dst_tmp;
131 auto dst_tmp_2d = dst_tmp.xetla_format<dtype_dst, tail_m, blk_n>();
132#pragma unroll
133 for (int i_acc = 0; i_acc < tail_m; i_acc++) {
134 dst_tmp_2d.row(i_acc)
135 = a_block[i_acc + tail_start_m] * b_blk_2d.row(0)
136 + src_blk_2d.row(i_acc + tail_start_m);
137 }
138#pragma unroll
139 for (int k = 1; k < blk_k - 1; k++) {
140 for (int i_acc = 0; i_acc < tail_m; i_acc++) {
141 int a_offset = k * blk_m + i_acc + tail_start_m;
142 dst_tmp_2d.row(i_acc)
143 += a_block[a_offset] * b_blk_2d.row(k);
144 }
145 }
146 for (int i_acc = 0; i_acc < tail_m; i_acc++) {
147 int a_offset = (blk_k - 1) * blk_m + i_acc + tail_start_m;
148 dst_blk_2d.row(i_acc + tail_start_m)
149 = a_block[a_offset] * b_blk_2d.row(blk_k - 1)
150 + dst_tmp_2d.row(i_acc);
151 }
152 }
153 }

155 __XETLA_API static void mma(
156 matDst_t &dst, matSrc_t &src, matB_t &b, matA_t &a) {
157 { //k_blk=0
158 auto b_reg
159 = b.reg.xetla_select<b_block_size_y * b_tile_size_x, 1>(0);
160#pragma unroll
161 for (int i = 0; i < tile_size_m / block_size_m; i++) {
162 auto a_block = a.reg.xetla_select<a_block_elems, 1>(
163 i * num_block_k * a_block_elems);
164#pragma unroll
165 for (int j = 0; j < num_block_n; j++) {
166 auto b_block = b_reg.xetla_select<b_block_elems, 1>(
167 j * b_block_elems);
168 auto src_block = src.reg.xetla_select<block_elems, 1>(
169 (i * num_block_n + j) * block_elems);
170 auto dst_block = dst.reg.xetla_select<block_elems, 1>(
171 (i * num_block_n + j) * block_elems);
172 mma_core<block_size_m, block_size_n, block_size_k>(
173 dst_block, src_block, b_block, a_block);
174 }
175 }
176
177 // process the tail
178 if constexpr ((tile_size_m % block_size_m) != 0) {
179 constexpr uint32_t tail_start_m
180 = tile_size_m / block_size_m * block_size_m;
181 constexpr uint32_t a_tail_blk_w = a_tile_size_y - tail_start_m;
182 constexpr uint32_t a_tail_blk_elems
183 = a_block_size_h * a_tail_blk_w;
184 constexpr uint32_t tail_size_m = tile_size_m - tail_start_m;
185 constexpr uint32_t acc_tail_blk_elems
186 = tail_size_m * block_size_n;
187 auto a_block = a.reg.xetla_select<a_tail_blk_elems, 1>(
188 a_tile_size_x * tail_start_m);
189#pragma unroll
190 for (int j = 0; j < num_block_n; j++) {
191 auto b_block = b_reg.xetla_select<b_block_elems, 1>(
192 j * b_block_elems);
193 auto src_block
194 = src.reg.xetla_select<acc_tail_blk_elems, 1>(
195 (tail_start_m * tile_size_n)
196 + j * acc_tail_blk_elems);
197 auto dst_block
198 = dst.reg.xetla_select<acc_tail_blk_elems, 1>(
199 (tail_start_m * tile_size_n)
200 + j * acc_tail_blk_elems);
201 mma_core<tail_size_m, block_size_n, block_size_k>(
202 dst_block, src_block, b_block, a_block);
203 }
204 }
205 }
206 // different K block
207#pragma unroll
208 for (int k_i = 1; k_i < num_block_k; k_i++) {
209
210 auto b_reg = b.reg.xetla_select<b_block_size_y * b_tile_size_x, 1>(
211 k_i * b_block_size_y * b_tile_size_x);
212#pragma unroll
213 for (int i = 0; i < tile_size_m / block_size_m; i++) {
214 auto a_block = a.reg.xetla_select<a_block_elems, 1>(
215 (i * num_block_k + k_i) * a_block_elems);
216#pragma unroll
217 for (int j = 0; j < num_block_n; j++) {
218 auto b_block = b_reg.xetla_select<b_block_elems, 1>(
219 j * b_block_elems);
220 auto dst_block = dst.reg.xetla_select<block_elems, 1>(
221 (i * num_block_n + j) * block_elems);
222 mma_core<block_size_m, block_size_n, block_size_k>(
223 dst_block, dst_block, b_block, a_block);
224 }
225 }
226 // process the tail
227 if constexpr ((tile_size_m % block_size_m) != 0) {
228 constexpr uint32_t tail_start_m
229 = tile_size_m / block_size_m * block_size_m;
230 constexpr uint32_t a_tail_blk_w = a_tile_size_y - tail_start_m;
231 constexpr uint32_t a_tail_blk_elems
232 = a_block_size_h * a_tail_blk_w;
233 constexpr uint32_t tail_size_m = tile_size_m - tail_start_m;
234 constexpr uint32_t acc_tail_blk_elems
235 = tail_size_m * block_size_n;
236 auto a_block = a.reg.xetla_select<a_tail_blk_elems, 1>(
237 a_tile_size_x * tail_start_m + k_i * a_tail_blk_elems);
238#pragma unroll
239 for (int j = 0; j < num_block_n; j++) {
240 auto b_block = b_reg.xetla_select<b_block_elems, 1>(
241 j * b_block_elems);
242 auto dst_block
243 = dst.reg.xetla_select<acc_tail_blk_elems, 1>(
244 (tail_start_m * tile_size_n)
245 + j * acc_tail_blk_elems);
246 mma_core<tail_size_m, block_size_n, block_size_k>(
247 dst_block, dst_block, b_block, a_block);
248 }
249 }
250 }
251 }
