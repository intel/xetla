316 __XETLA_API static void call(xetla_exec_item<3> &ei, arguments_t *args) {
317
318 int tru_seqlen = 0;
319 int tru_seqlen_ex = 0;
320 int seqlen_entry = 0;
321
322 int groupid = ei.get_group(0);
323 int hiddensize = 1024;
324 int numhead = 16;
325 int hdsz = 64;
326 int wg_tile_QKT_k = hdsz; //args->matrix_k;
327 int wg_tile_out_k;
328 int batchid = groupid / numhead;
329 int headid = groupid % numhead;
330
331 work_group_t g_thd32_tid;
332 int tid_linear = ei.get_local_linear_id();
333 g_thd32_tid.init(tid_linear);
334
335 uint32_t batch_offset = sizeof(uint32_t) * list_width * batchid;
336 xetla_vector<uint32_t, list_width> list_offsets
337 = xetla_vector_gen<uint32_t, list_width>(0, 1);
338 list_offsets *= sizeof(uint32_t);
339 list_offsets += batch_offset;
340
341 xetla_vector<uint32_t, list_width> list_vec
342 = xetla_load_global<uint32_t, 1, data_size::default_size,
343 cache_hint::read_invalidate, cache_hint::cached,
344 list_width>(args->mList_ptr, list_offsets);
345 tru_seqlen = list_vec[0];
346 seqlen_entry = list_vec[1];
347 wg_tile_out_k = tru_seqlen;
348 tru_seqlen_ex = tru_seqlen; //DW align
349 if (sfx_type_size == 2)
350 tru_seqlen_ex = (((tru_seqlen + 1) >> 1) << 1);
351 else if (sfx_type_size == 1)
352 tru_seqlen_ex = (((tru_seqlen + 3) >> 2) << 2);
353 //float totalscaling = args->Pinv * args->Scaling;
354
355 xetla_rand_t<Rand_SIMD> Rand_Gen;
356 uint32_t rand_threshold = rand_threshold_const;
357 if constexpr (Dopt_RandGenflag == true) {
358 uint64_t rand_seed = 67280421310721;
359 uint64_t rand_subseq
360 = (groupid * ThreadNum + tid_linear) * Rand_SIMD;
361 uint64_t rand_offset = list_vec.xetla_format<uint64_t>()[1];
362 if (list_vec[4] != 0) rand_threshold = list_vec[4];
363 if (rand_offset == 0) {
364 xetla_vector<uint32_t, 4> time_stamp = get_time_stamp();
365 rand_offset = time_stamp.xetla_format<uint64_t>()[0];
366 }
367 Rand_Gen.init(rand_seed, rand_subseq, rand_offset);
368 }
369
370 //std_leqlen = 256
371 int all_vert_loop_num = 2;
372 int all_vert_stride = 128;
373 int all_vert128_shift = 0;
374 int block_16x16_num = 4;
375 int tid_x_shift = 0;
376
377 int std_seqlen;
378 if (tru_seqlen <= 128) {
379 std_seqlen = 128;
380 tid_x_shift = 2; // 16x32 128/32 = 4
381 all_vert_loop_num = 1;
382 block_16x16_num = 2;
383 } else if (tru_seqlen <= 256) {
384 std_seqlen = 256;
385 tid_x_shift = 2; // 16x64 256/64 = 4
386 } else if (tru_seqlen <= 384) {
387 std_seqlen = 384;
388 all_vert_stride = 64;
389 all_vert128_shift = 1;
390 block_16x16_num = 3;
391 tid_x_shift = 3; // 16x48 384/48 = 8
392 all_vert_loop_num = (tru_seqlen + all_vert_stride - 1) >> 6;
393 } else if (tru_seqlen <= 512) {
394 std_seqlen = 512;
395 all_vert_stride = 64;
396 all_vert128_shift = 1;
397 tid_x_shift = 3; // 16x64 512/64 = 8
398 all_vert_loop_num = (tru_seqlen + all_vert_stride - 1) >> 6;
399 } else if (tru_seqlen <= 1024) {
400 std_seqlen = 1024;
401 all_vert_stride = 32;
402 all_vert128_shift = 2;
403 tid_x_shift = 4; // 16x64 1024/64 = 16
404 all_vert_loop_num = (tru_seqlen + all_vert_stride - 1) >> 5;
405 } else if (tru_seqlen <= 2048) {
406 std_seqlen = 2048;
407 all_vert_stride = 16;
408 all_vert128_shift = 3;
409 tid_x_shift = 5; // 16x64 2048/64 = 32
410 all_vert_loop_num = (tru_seqlen + all_vert_stride - 1) >> 4;
411 }
412 all_vert_loop_num = ((all_vert_loop_num + (1 << all_vert128_shift) - 1)
413 >> all_vert128_shift)
414 << all_vert128_shift;
415 int tid_x = tid_linear & ((1 << tid_x_shift) - 1);
416 int tid_y = tid_linear >> tid_x_shift;
417
418 xetla_nbarrier_t<32, 32> first_nbarr;
419 xetla_nbarrier_t<32, 32> second_nbarr;
420 xetla_nbarrier_t<32, 32> third_nbarr;
421 first_nbarr.init_nbarrier(31, nbarrier_role::producer_consumer);
422 second_nbarr.init_nbarrier(30, nbarrier_role::producer_consumer);
423 third_nbarr.init_nbarrier(29, nbarrier_role::producer_consumer);
424
425 xetla_vector<int8_t, 4 * 16> attn_mk_4x16;
426 int valid_block_16x16_x = (tid_x + 1) * 16 * block_16x16_num;
427 {
428 int bndy_block_num = 0;
429 if (valid_block_16x16_x <= tru_seqlen)
430 valid_block_16x16_x = block_16x16_num;
431 else {
432 bndy_block_num = valid_block_16x16_x;
433 valid_block_16x16_x = (tru_seqlen + 15 + 16 * block_16x16_num
434 - valid_block_16x16_x)
435 >> 4;
436 bndy_block_num = bndy_block_num
437 + (valid_block_16x16_x - block_16x16_num) * 16
438 - tru_seqlen;
439 }
440
441 xetla_vector<uint32_t, 16> address_attn_mk
442 = xetla_vector_gen<uint32_t, 16>(0, 1);
443 int attn_mk_address_offset
444 = (batchid * Max_SeqLen) + (tid_x * 16 * block_16x16_num);
445 address_attn_mk *= sizeof(uint32_t);
