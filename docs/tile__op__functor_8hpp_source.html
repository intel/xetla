84struct sigmoid_op_t {
85 struct arguments_t {};
86 template <typename matAcc_t, typename coord_t>
87 __XETLA_API KERNEL_FUNC void operator()(matAcc_t &matAcc,
88 const coord_t &coord, uint32_t slm_base = 0,
89 uint32_t nbarrier_base = 0) {
90 constexpr int elems = matAcc_t::tile_desc::block_elems;
91 constexpr int rounds = matAcc_t::tile_desc::tile_elems / elems;
92 constexpr float one = 1.0f;
93#pragma unroll
94 for (int i = 0; i < rounds; ++i) {
95 auto sub_vec = matAcc.reg.xetla_select<elems, 1>(elems * i);
96 xetla_mask<elems> mask = sub_vec >= 10;
97 xetla_vector<typename matAcc_t::dtype, elems> temp_vec
98 = xetla_exp<typename matAcc_t::dtype, elems>(sub_vec);
99 xetla_vector<typename matAcc_t::dtype, elems> sigmoid_value
100 = temp_vec / (temp_vec + one);
101 sigmoid_value.xetla_merge(1, mask);
102 sub_vec = sigmoid_value;
103 }
104 constexpr int remained_elems = matAcc_t::tile_desc::tile_elems % elems;
105 if constexpr (remained_elems != 0) {
106 auto sub_vec = matAcc.reg.xetla_select<remained_elems, 1>(
107 elems * (matAcc_t::tile_elems / elems));
108 xetla_mask<remained_elems> mask = sub_vec >= 250;
109 xetla_vector<typename matAcc_t::dtype, remained_elems> temp_vec
110 = xetla_exp<typename matAcc_t::dtype, remained_elems>(
111 sub_vec);
112 xetla_vector<typename matAcc_t::dtype, remained_elems> sigmoid_value
113 = temp_vec / (temp_vec + one);
114 sigmoid_value.xetla_merge(1, mask);
115 sub_vec = sigmoid_value;
116 }
117 }

123struct gelu_fwd_op_t {
124 struct arguments_t {};
125 template <typename matAcc_t, typename coord_t>
126 __XETLA_API KERNEL_FUNC void operator()(matAcc_t &matAcc,
127 const coord_t &coord, const arguments_t &args,
128 uint32_t slm_base = 0, uint32_t nbarrier_base = 0) {
129 using dtype = typename matAcc_t::dtype;
130 constexpr dtype C0 = 0.044715f;
131 constexpr dtype sqrt_two_over_pi = 0.79788458347320556640625f;
132 // total flag register
133 constexpr int elems = 8 * 16;
134 constexpr int rounds = matAcc_t::tile_elems / elems;
135#pragma unroll
136 for (int i = 0; i < rounds; ++i) {
137 auto sub_vec = matAcc.reg.xetla_select<elems, 1>(elems * i);
138 xetla_vector<dtype, elems> sub_vec_x = (sqrt_two_over_pi * sub_vec
139 * (1.f + C0 * sub_vec * sub_vec));
140 xetla_vector<dtype, elems> tanh_value
141 = xetla_tanh<dtype, elems>(sub_vec_x);
142 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);
143 }
144 constexpr int remained_elems = matAcc_t::tile_elems % elems;
145 if constexpr (remained_elems != 0) {
146 auto sub_vec = matAcc.reg.xetla_select<remained_elems, 1>(
147 elems * (matAcc_t::tile_elems / elems));
148 xetla_vector<dtype, remained_elems> sub_vec_x = (sqrt_two_over_pi
149 * sub_vec * (1.f + C0 * sub_vec * sub_vec));
150 xetla_vector<dtype, remained_elems> tanh_value
151 = xetla_tanh<dtype, remained_elems>(sub_vec_x);
152 sub_vec = 0.5f * sub_vec * (1.f + tanh_value);
153 }
154 }

186 uint32_t slm_base = 0, uint32_t nbarrier_base = 0) {
187 using dtype_acc = typename matAcc_t::dtype;
188 static constexpr uint32_t tile_size_x = matAcc_t::tile_size_x;
189 static constexpr uint32_t tile_size_y = matAcc_t::tile_size_y;
190 static constexpr uint32_t block_size_x = matAcc_t::block_size_x;
191 static constexpr uint32_t block_size_y = matAcc_t::block_size_y;

193 mem_desc_w_t mem_desc_w(args.base, args.shape, coord);
194 using bwd_w_tile_desc_t = tile_desc_t<block_size_x, block_size_y,
195 block_size_x, block_size_y, reg_layout::tiled>;
196 using bwd_w_tile_t = tile_t<dtype_out, bwd_w_tile_desc_t>;
197 using bwd_w_payload_t = mem_payload_t<mem_desc_w_t, bwd_w_tile_desc_t,
198 msg_type::block_2d, arch_tag>;
199 bwd_w_tile_t bwd_w;
200 bwd_w_payload_t bwd_w_payload(mem_desc_w);
201 // start compute
202 constexpr dtype_acc c0 = 0.044715f;
203 constexpr dtype_acc d0 = 0.134145f;
204 constexpr dtype_acc sqrt_two_over_pi = 0.79788458347320556640625f;
205 constexpr uint32_t block_elems = matAcc_t::block_elems;
206 constexpr uint32_t num_block_x = matAcc_t::num_block_x;
207#pragma unroll
208 for (int i = 0; i < tile_size_y / block_size_y; ++i) {
209#pragma unroll
210 for (int j = 0; j < num_block_x; ++j) {
211 auto x = matAcc.reg.xetla_select<block_elems, 1>(
212 block_elems * (i * num_block_x + j));
213 xetla_vector<dtype_acc, block_elems> z
214 = xetla_tanh<dtype_acc, block_elems>(
215 sqrt_two_over_pi * (x + c0 * x * x * x));
216 xetla_vector<dtype_acc, block_elems> w = (0.5f * (1.f + z)
217 + 0.5f * x * (1.f - z * z)
218 * (sqrt_two_over_pi * (1.f + d0 * x * x)));
219 x = 0.5f * x * (1.f + z);
220 bwd_w.reg = xetla_cvt<dtype_out, dtype_acc, block_elems>(w);
221 tile_store<cache_hint::uncached>(bwd_w, bwd_w_payload);
222 bwd_w_payload.template update_tdesc<tdesc_update_dir::x_dir>(
223 block_size_x);
224 }
225 bwd_w_payload.template update_tdesc<tdesc_update_dir::x_dir>(
226 -1 * tile_size_x);
227 bwd_w_payload.template update_tdesc<tdesc_update_dir::y_dir>(
228 block_size_y);
229 }
230 if constexpr (tile_size_y % block_size_y != 0) {
231 constexpr uint32_t remain_size_y = tile_size_y % block_size_y;
232 constexpr uint32_t remain_y_start
233 = tile_size_y / block_size_y * block_size_y;
234 constexpr uint32_t remain_elems_start
235 = remain_y_start * tile_size_x;
236 constexpr uint32_t remain_block_elems
237 = remain_size_y * block_size_x;
238
239 using remain_bwd_w_tile_desc_t
240 = tile_desc_t<block_size_x, remain_size_y, block_size_x,
241 remain_size_y, reg_layout::tiled>;
242 using remain_bwd_w_tile_t
243 = tile_t<dtype_out, remain_bwd_w_tile_desc_t>;
244 using remain_bwd_w_payload_t = mem_payload_t<mem_desc_w_t,
245 remain_bwd_w_tile_desc_t, msg_type::block_2d, arch_tag>;
246
247 mem_desc_w.update_coord_y(remain_y_start);
248 remain_bwd_w_payload_t remain_bwd_w_payload(mem_desc_w);
249 remain_bwd_w_tile_t remain_bwd_w;
250#pragma unroll
251 for (int j = 0; j < num_block_x; ++j) {
252 auto x = matAcc.reg.xetla_select<remain_block_elems, 1>(
253 remain_elems_start + remain_block_elems * j);
254 xetla_vector<dtype_acc, remain_block_elems> z
255 = xetla_tanh<dtype_acc, remain_block_elems>(
256 sqrt_two_over_pi * (x + c0 * x * x * x));
257 xetla_vector<dtype_acc, remain_block_elems> w = (0.5f
258 * (1.f + z)
259 + 0.5f * x * (1.f - z * z)
260 * (sqrt_two_over_pi * (1.f + d0 * x * x)));
261 x = 0.5f * x * (1.f + z);
262 remain_bwd_w.reg
263 = xetla_cvt<dtype_out, dtype_acc, remain_block_elems>(
264 w);
265 tile_store<cache_hint::uncached>(
266 remain_bwd_w, remain_bwd_w_payload);
267 remain_bwd_w_payload
268 .template update_tdesc<tdesc_update_dir::x_dir>(
269 block_size_x);
270 }
271 }
