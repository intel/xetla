55template <cache_hint L1 = cache_hint::cached,
56 cache_hint L3 = cache_hint::cached, typename payload_t>
57__XETLA_API typename std::enable_if_t<
58 detail::check_prefetch_type<payload_t>::is_global_2d_xe>
59tile_prefetch(payload_t &payload) {
60 using dtype = typename payload_t::dtype;
61 static constexpr uint32_t num_tdesc = payload_t::num_tdesc;
62 auto tdesc_2d
63 = payload.tdesc_prefetch.xetla_format<uint32_t, num_tdesc, 16>();
64
65#pragma unroll
66 for (int i = 0; i < num_tdesc; i++) {
67 xetla_tprefetch_global<dtype, L1, L3>(tdesc_2d.row(i));
68 }
69}

79template <cache_hint L1 = cache_hint::cached,
80 cache_hint L3 = cache_hint::cached, typename payload_t>
81__XETLA_API typename std::enable_if_t<
82 detail::check_prefetch_type<payload_t>::is_global_block_1d_xe>
83tile_prefetch(payload_t &payload) {
84 using dtype = typename payload_t::dtype;
85 using tile_desc = typename payload_t::tile_desc;
86 using prefetch_dtype = typename payload_t::prefetch_dtype;
87 constexpr uint32_t prefetch_len
88 = tile_desc::tile_size_x / payload_t::scale_factor;
89 if constexpr (prefetch_len >= 64) {
90#pragma unroll
91 for (int j = 0; j < prefetch_len / 64; j++) {
92 uint32_t offset_x = j * 64 * payload_t::scale_factor;
93 uint32_t address_offset = offset_x * sizeof(dtype);
94 xetla_prefetch_global<prefetch_dtype, 64, data_size::default_size,
95 L1, L3>(
96 payload.base_ptr, payload.base_offset + address_offset);
97 }
98 }
99 constexpr uint32_t tail_len = prefetch_len % 64;
100 uint32_t tail_offset = prefetch_len / 64 * 64 * payload_t::scale_factor;
101 detail::process_1d_tail<tail_len, 32, L1, L3, payload_t>(
102 payload, tail_offset);
103}
