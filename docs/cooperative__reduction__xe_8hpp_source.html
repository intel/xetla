38template <reduce_op reduce_kind, typename tile_shape_, typename matAcc_t,
39 uint32_t num_cooperative_wg>
40class cooperative_reduce_t<reduce_kind, tile_shape_, matAcc_t,
41 num_cooperative_wg, gpu_arch::Xe> {
42public:
43 static constexpr gpu_arch arch_tag = gpu_arch::Xe;
44 using tile_shape = tile_shape_;
45 using dtype = typename matAcc_t::dtype;
46
47private:
48 static constexpr uint32_t sg_tile_m = tile_shape::sg_tile_size_y;
49 static constexpr uint32_t sg_tile_n = tile_shape::sg_tile_size_x;
50 static constexpr uint32_t wg_size_x = tile_shape::wg_size_x;
51 static constexpr uint32_t wg_size_y = tile_shape::wg_size_y;
52 static constexpr uint32_t real_wg_tile_m = sg_tile_m * wg_size_y;
53 static constexpr uint32_t real_wg_tile_n = sg_tile_n * wg_size_x;
54
55 static constexpr uint32_t wg_tile_size
56 = real_wg_tile_m * real_wg_tile_n * sizeof(dtype);
57 using work_group_t = typename tile_shape::work_group_t;
58 static constexpr uint32_t work_group_size = work_group_t::size;
59 // cooperative split, y dir first
60 static_assert((num_cooperative_wg & (num_cooperative_wg - 1)) == 0,
61 "num_cooperative_wg should be power of 2");
62
63public:
64 static constexpr uint32_t coop_num_y
65 = gpu::xetla::subgroup::detail::gcd<num_cooperative_wg,
66 sg_tile_m>::value;
67 static constexpr uint32_t coop_remain_num_x
68 = num_cooperative_wg / coop_num_y;
69 static constexpr bool has_redundant_wg
70 = (coop_remain_num_x * 16) > sg_tile_n;
71 static constexpr uint32_t tile_size_y = sg_tile_m / coop_num_y;
72 static constexpr uint32_t tile_size_x
73 = has_redundant_wg ? 16 : sg_tile_n / coop_remain_num_x;
74 static constexpr uint32_t coop_num_x = sg_tile_n / tile_size_x;
75 static constexpr uint32_t num_reduce_wg = coop_num_x * coop_num_y;
76
77private:
78 static constexpr uint32_t src_block_size_x = matAcc_t::block_size_x;
79 static constexpr uint32_t src_block_size_y = matAcc_t::block_size_y;
80
81 static constexpr uint32_t block_size_x
82 = gpu::xetla::subgroup::detail::gcd<tile_size_x,
83 src_block_size_x>::value;
84 static constexpr uint32_t block_size_y
85 = (tile_size_y > src_block_size_y) ? src_block_size_y : tile_size_y;
86
87 using local_st_tile_desc_t = subgroup::tile_desc_t<sg_tile_n, sg_tile_m,
88 src_block_size_x, src_block_size_y, reg_layout::tiled>;
89 using local_st_tile_t = subgroup::tile_t<dtype, local_st_tile_desc_t>;
90 using local_st_payload_t = subgroup::mem_payload_t<dtype,
91 local_st_tile_desc_t,
92 subgroup::msg_type_v<local_st_tile_desc_t, mem_space::local>,
93 mem_layout::row_major, mem_space::local, arch_tag>;
94 using local_ld_tile_desc_t = subgroup::tile_desc_t<tile_size_x, tile_size_y,
95 block_size_x, block_size_y, reg_layout::tiled>;
96 using local_ld_tile_t = subgroup::tile_t<dtype, local_ld_tile_desc_t>;
97 using local_ld_payload_t = subgroup::mem_payload_t<dtype,
98 local_ld_tile_desc_t,
99 subgroup::msg_type_v<local_ld_tile_desc_t, mem_space::local>,
100 mem_layout::row_major, mem_space::local, arch_tag>;
101
102public:
103 using mat_slice_t = subgroup::tile_t<dtype, local_ld_tile_desc_t>;
104
105 static constexpr uint32_t barrier_count = work_group_size;
106 static constexpr uint32_t slm_size = wg_tile_size * num_cooperative_wg;
107
108 uint32_t coop_id;
109 uint32_t coop_id_x;
110 uint32_t coop_id_y;
111 inline cooperative_reduce_t(uint32_t coop_id_) : coop_id(coop_id_) {
112 coop_id_x = coop_id % coop_remain_num_x;
113 coop_id_y = coop_id / coop_remain_num_x;
114 }

126 inline KERNEL_FUNC void operator()(work_group_t &g, mat_slice_t &mat_slice,
127 matAcc_t &matAcc, uint32_t slm_base = 0,
128 uint32_t nbarrier_base = 0) {
129 uint32_t sg_idx = g.get_id() % wg_size_x;
130 uint32_t sg_idy = g.get_id() / wg_size_x;
131
132 int32_t slm_store_offset_x = sg_idx * sg_tile_n;
133 int32_t slm_store_offset_y
134 = coop_id * real_wg_tile_m + sg_idy * sg_tile_m;
135 local_st_tile_t local_st;
136 local_st_payload_t local_st_payload(slm_base, real_wg_tile_n,
137 real_wg_tile_m * num_cooperative_wg, real_wg_tile_n,
138 slm_store_offset_x, slm_store_offset_y);
139 local_st.reg = matAcc.reg;
140 tile_store(local_st, local_st_payload);
141
142 xetla_nbarrier_t<num_cooperative_wg, num_cooperative_wg> nbarrier;
143 uint32_t nbar_id = nbarrier_base + g.get_id();
144 nbarrier.init_nbarrier(nbar_id, nbarrier_role::producer_consumer);
145 xetla_fence<memory_kind::shared_local>();
146 nbarrier.arrive();
147 nbarrier.wait();
148
149 if (coop_id_x < coop_num_x) {
150 // nbarrier.init_nbarrier(nbar_id, nbarrier_role::consumer);
151 // nbarrier.arrive();
152 int32_t slm_load_offset_x
153 = sg_idx * sg_tile_n + coop_id_x * tile_size_x;
154 int32_t slm_load_offset_y
155 = sg_idy * sg_tile_m + coop_id_y * tile_size_y;
156
157 local_ld_tile_t local_ld;
158 local_ld_payload_t local_ld_payload(slm_base, real_wg_tile_n,
159 real_wg_tile_m * num_cooperative_wg, real_wg_tile_n,
160 slm_load_offset_x, slm_load_offset_y);
161
162 tile_load(local_ld, local_ld_payload);
163 mat_slice.reg = local_ld.reg;
164#pragma unroll
165 for (int i = 1; i < num_cooperative_wg; i++) {
166 local_ld_payload.template update_tdesc<tdesc_update_dir::y_dir>(
167 real_wg_tile_m);
168 tile_load(local_ld, local_ld_payload);
169 mat_slice.reg = reduce_helper<reduce_kind, dtype>(
170 mat_slice.reg, local_ld.reg);
171 }
172 }
173 }
174};
