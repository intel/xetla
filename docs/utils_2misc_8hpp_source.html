60template <typename dtype_acc, uint32_t N, uint32_t num_flag = 4,
61 typename dtype_mask = uint8_t>
62__XETLA_API xetla_vector<dtype_acc, N> drop_out(xetla_vector<dtype_acc, N> in,
63 xetla_vector<dtype_mask, N> mask, dtype_acc scale) {
64 xetla_vector<dtype_acc, N> out = in * scale;
65 constexpr uint32_t unroll_size = num_flag * 16;
66 SW_BARRIER();
67#pragma unroll
68 for (int i = 0; i < N / unroll_size; i++) {
69 xetla_mask<unroll_size> mask_flag
70 = mask.xetla_select<unroll_size, 1>(i * unroll_size) > 0;
71 out.xetla_select<unroll_size, 1>(i * unroll_size)
72 .xetla_merge(0, mask_flag);
73 }
74 if constexpr (N % unroll_size != 0) {
75 constexpr uint32_t remain_len = N % unroll_size;
76 constexpr uint32_t remain_start = N / unroll_size * unroll_size;
77 xetla_mask<remain_len> mask_flag
78 = mask.xetla_select<remain_len, 1>(remain_start) > 0;
79 out.xetla_select<remain_len, 1>(remain_start).xetla_merge(0, mask_flag);
80 }
81 return out;
82}

123template <reduce_op reduce_kind, typename dtype, int N_x, int N_y>
124__XETLA_API typename std::enable_if_t<(N_y > 1), xetla_vector<dtype, N_x>>
125recur_row_reduce(xetla_vector<dtype, N_x * N_y> in) {
126 static_assert(((N_y) & (N_y - 1)) == 0, "N_y should be power of 2");
127 xetla_vector<dtype, N_x * N_y / 2> temp;
128 temp = reduce_helper<reduce_kind, dtype, N_x * N_y / 2>(
129 in.xetla_select<N_x * N_y / 2, 1>(0),
130 in.xetla_select<N_x * N_y / 2, 1>(N_x * N_y / 2));
131
132 return recur_row_reduce<reduce_kind, dtype, N_x, N_y / 2>(temp);
133}

141__XETLA_API typename std::enable_if_t<(N_x > 1), xetla_vector<dtype, N_y>>
142recur_col_reduce(xetla_vector<dtype, N_x * N_y> in) {
143 static_assert(((N_x) & (N_x - 1)) == 0, "N_x should be power of 2");
144 xetla_vector<dtype, N_x * N_y / 2> temp;

149
150 return recur_col_reduce<reduce_kind, dtype, N_x / 2, N_y>(temp);
