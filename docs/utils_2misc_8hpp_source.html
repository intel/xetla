72template <typename dtype_acc, uint32_t N, uint32_t num_flag = 4,
73 typename dtype_mask = uint8_t>
74__XETLA_API xetla_vector<dtype_acc, N> drop_out(xetla_vector<dtype_acc, N> in,
75 xetla_vector<dtype_mask, N> mask, dtype_acc scale) {
76 xetla_vector<dtype_acc, N> out = in * scale;
77 constexpr uint32_t unroll_size = num_flag * 16;
78 SW_BARRIER();
79#pragma unroll
80 for (int i = 0; i < N / unroll_size; i++) {
81 xetla_mask<unroll_size> mask_flag
82 = mask.xetla_select<unroll_size, 1>(i * unroll_size) > 0;
83 out.xetla_select<unroll_size, 1>(i * unroll_size)
84 .xetla_merge(0, mask_flag);
85 }
86 if constexpr (N % unroll_size != 0) {
87 constexpr uint32_t remain_len = N % unroll_size;
88 constexpr uint32_t remain_start = N / unroll_size * unroll_size;
89 xetla_mask<remain_len> mask_flag
90 = mask.xetla_select<remain_len, 1>(remain_start) > 0;
91 out.xetla_select<remain_len, 1>(remain_start).xetla_merge(0, mask_flag);
92 }
93 return out;
94}

135template <reduce_op reduce_kind, typename dtype, int N_x, int N_y>
136__XETLA_API typename std::enable_if_t<(N_y > 1), xetla_vector<dtype, N_x>>
137recur_row_reduce(xetla_vector<dtype, N_x * N_y> in) {
138 static_assert(((N_y) & (N_y - 1)) == 0, "N_y should be power of 2");
139 xetla_vector<dtype, N_x * N_y / 2> temp;
140 temp = reduce_helper<reduce_kind, dtype, N_x * N_y / 2>(
141 in.xetla_select<N_x * N_y / 2, 1>(0),
142 in.xetla_select<N_x * N_y / 2, 1>(N_x * N_y / 2));
143
144 return recur_row_reduce<reduce_kind, dtype, N_x, N_y / 2>(temp);
145}

153__XETLA_API typename std::enable_if_t<(N_x > 1), xetla_vector<dtype, N_y>>
154recur_col_reduce(xetla_vector<dtype, N_x * N_y> in) {
155 static_assert(((N_x) & (N_x - 1)) == 0, "N_x should be power of 2");
156 xetla_vector<dtype, N_x * N_y / 2> temp;
157 auto in_2d = in.xetla_format<dtype, N_y, N_x>();
158 temp = reduce_helper<reduce_kind, dtype, N_y * N_x / 2>(
159 in_2d.xetla_select<N_y, 1, N_x / 2, 1>(0, 0),
160 in_2d.xetla_select<N_y, 1, N_x / 2, 1>(0, N_x / 2));
161
162 return recur_col_reduce<reduce_kind, dtype, N_x / 2, N_y>(temp);
163}
