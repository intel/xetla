29template <typename tile_t, typename payload_t>
30struct check_load_type {
31 static constexpr bool is_global_2d_xe
32 = (payload_t::memory_space == mem_space::global
33 && (payload_t::message_type == msg_type::block_2d)
34 && (payload_t::arch_tag == gpu_arch::Xe));
35
36 static constexpr bool is_global_block_1d_xe
37 = ((payload_t::memory_space == mem_space::global)
38 && (tile_t::tile_size_y == 1) && (tile_t::block_size_y == 1)
39 && (payload_t::message_type == msg_type::block_1d)
40 && (payload_t::arch_tag == gpu_arch::Xe));
41
42 static constexpr bool is_local_scatter_xe
43 = ((payload_t::memory_space == mem_space::local)
44 && (payload_t::message_type == msg_type::scatter)
45 && (payload_t::arch_tag == gpu_arch::Xe));
46
47 static constexpr bool is_local_block_1d_xe
48 = ((payload_t::memory_space == mem_space::local)
49 && (payload_t::message_type == msg_type::block_1d)
50 && (payload_t::arch_tag == gpu_arch::Xe));
51};
52
53template <typename tile_t, typename payload_t>
54inline constexpr void check_load_condition() {
55 using dtype = typename payload_t::dtype;
56 if constexpr (check_load_type<tile_t, payload_t>::is_global_2d_xe) {
57 using tile_desc = typename payload_t::tile_desc;
58 constexpr bool reg_transpose = tile_desc::reg_transpose;
59 constexpr reg_layout reg_layout_ = tile_desc::register_layout;
60 constexpr bool mem_transpose = payload_t::mem_transpose;
61
62 using load_store_attr =
63 typename arch_attr_t<payload_t::arch_tag>::load_store_attr;
64 constexpr int32_t max_vnni_block_width
65 = load_store_attr::max_vnni_load_width_in_elems;
66 static_assert(!payload_t::mem_transform
67 || tile_desc::block_size_x <= max_vnni_block_width,
68 "For VNNI transform, the maximum block width is 16 width.");
69 constexpr int32_t max_block_width
70 = load_store_attr::max_load_width_in_bytes / sizeof(dtype);
71 static_assert((max_block_width % tile_desc::block_size_x) == 0,
72 "max_block_width should be a multiply of block size x.");
73
74 } else if constexpr (check_load_type<tile_t,
75 payload_t>::is_global_block_1d_xe) {
76 static_assert(sizeof(typename payload_t::mem_dtype) >= 4,
77 "store size should at least DW aligned");
78
79 } else if constexpr (check_load_type<tile_t,
80 payload_t>::is_local_scatter_xe) {
81 static_assert(payload_t::memory_layout == mem_layout::row_major,
82 "only support row major in local load, you can use local store "
83 "to "
84 "do the transpose");
85 static_assert(sizeof(typename payload_t::mem_dtype) >= 4,
86 "load size should at least DW aligned");
87 static_assert((payload_t::tile_desc::block_size_x * sizeof(dtype)
88 % sizeof(typename payload_t::mem_dtype))
89 == 0,
90 "bytes per row should be a multiply of sizeof load_dtype");
91 static_assert(((payload_t::tile_bytes % payload_t::min_bytes) == 0
92 && (payload_t::block_bytes % payload_t::min_bytes)
93 == 0),
94 "currently, we are not able to handle the corner case");
95 static_assert((payload_t::num_channel_x > 0)
96 && (payload_t::num_channel_x <= payload_t::num_channel),
97 "The number of simd channel x should be greater than 0 and "
98 "less "
99 "than num_channel");
100
101 } else if constexpr (check_load_type<tile_t,
102 payload_t>::is_local_block_1d_xe) {
103 static_assert(sizeof(typename payload_t::mem_dtype) >= 4,
104 "store size should at least DW aligned");
105 }
106}

122template <cache_hint L1 = cache_hint::cached,
123 cache_hint L3 = cache_hint::cached, typename tile_t, typename payload_t>
124__XETLA_API typename std::enable_if_t<
125 detail::check_load_type<tile_t, payload_t>::is_global_2d_xe>
126tile_load(tile_t &tile, payload_t &payload) {
127 detail::check_load_condition<tile_t, payload_t>();
128 using dtype = typename tile_t::dtype;
129 using load_dtype = typename payload_t::mem_dtype;
130 using tile_desc = typename tile_t::tile_desc;
131 static constexpr uint32_t tile_size_x = tile_desc::tile_size_x;
132 static constexpr uint32_t tile_size_y = tile_desc::tile_size_y;
133 static constexpr uint32_t block_size_x = tile_desc::block_size_x;
134 static constexpr uint32_t block_size_y = tile_desc::block_size_y;
135 static constexpr uint32_t remained_size_y = tile_desc::remained_size_y;
136
137 static constexpr uint32_t block_elems = tile_desc::block_elems;
138
139 static constexpr uint32_t num_block_x = tile_desc::num_block_x;
140 static constexpr uint32_t num_block_y = tile_desc::num_block_y;
141 static constexpr uint32_t num_block = tile_desc::num_block;
142
143 static constexpr gpu_arch arch_tag = payload_t::arch_tag;
144
145 static constexpr reg_layout reg_layout_ = tile_desc::register_layout;
146 static constexpr bool is_vnni_reverse = payload_t::mem_dword_transpose
147 && ((reg_layout_ == reg_layout::tiled)
148 || (reg_layout_ == reg_layout::transpose_tiled));
149 static constexpr bool reg_transpose = tile_desc::reg_transpose;
150
151 static constexpr mem_layout mem_layout_ = payload_t::memory_layout;
152 static constexpr bool mem_transpose = payload_t::mem_transpose;
153 static constexpr bool trans = reg_transpose ^ mem_transpose;
154 static constexpr uint32_t scale_factor = payload_t::scale_factor;
155
156 static constexpr bool mem_transform = payload_t::mem_transform;
157
158 using load_store_attr = typename arch_attr_t<arch_tag>::load_store_attr;
159 static constexpr uint32_t elems_per_CL
160 = load_store_attr::cache_line_size_in_bytes / sizeof(dtype);
161 static constexpr uint32_t elems_per_reg
162 = arch_attr_t<arch_tag>::register_attr::reg_in_bytes
163 / sizeof(dtype);
164 static constexpr int32_t max_load_block_height
165 = load_store_attr::max_load_height_in_elem;
166 static constexpr int32_t max_block_width
167 = load_store_attr::max_load_width_in_bytes / sizeof(dtype);
168 static constexpr int32_t max_trans_block_width
169 = load_store_attr::max_trans_load_width_in_bytes / sizeof(dtype);
170 static constexpr int32_t max_vnni_block_width
171 = load_store_attr::max_vnni_load_width_in_elems;
172
173 static constexpr uint32_t ld_blk_size_y_limit
174 = mem_transpose ? max_trans_block_width : max_load_block_height;
175 static constexpr uint32_t ld_blk_size_y = reg_transpose
176 ? block_size_y
177 : (block_size_y > ld_blk_size_y_limit ? ld_blk_size_y_limit
178 : block_size_y);
179
180 // array len is used to make sure memory load is cache line aligned
181 // disabled while register or memory transpose
182 static constexpr uint8_t arr_len
183 = (reg_transpose
184 || mem_transpose
185 // block elements should be integer
186 // times of register bytes
187 || ((block_size_y * block_size_x) % elems_per_reg != 0)
188 // tail blocks also need to meet above condition
189 || (((tile_size_y % block_size_y) * block_size_x)
190 % elems_per_reg
191 != 0))
192 || (block_size_y > ld_blk_size_y_limit)
193 ? 1
194 : ((((tile_size_x % elems_per_CL) == 0)
195 && ((elems_per_CL % block_size_x) == 0))
196 ? elems_per_CL / block_size_x
197 : 1);
198
199 static_assert(reg_transpose || mem_transpose
200 || (!mem_transpose
201 && (block_size_x * arr_len) <= max_block_width),
202 "When reg_transpose was disabled, check 2d block width "
203 "restriction");
204 static_assert(!reg_transpose
205 || (!mem_transpose
206 && (block_size_x * arr_len)
207 <= max_trans_block_width)
208 || (mem_transpose
209 && (block_size_y * arr_len) <= max_block_width),
210 "When reg_transpose was enabled, check 2d block width "
211 "restriction");
212 static_assert(!reg_transpose
213 || (!mem_transpose
214 && (block_size_y <= max_load_block_height))
215 || (mem_transpose
216 && (block_size_x) <= max_load_block_height),
217 "When reg_transpose was enabled, check 2d block height "
218 "restriction");
219 static_assert(tile_size_x % (block_size_x * arr_len) == 0,
220 "tile_size_x should be a multiple of (block_size_x * arr_len)");
221 static_assert(
222 (reg_transpose
223 && ((block_size_x * sizeof(dtype)) % sizeof(load_dtype)
224 == 0))
225 || ((block_size_y * sizeof(dtype)) % sizeof(load_dtype)
226 == 0),
227 "check vnni limitation for DW transpose");
228
229 auto payload_2d = payload.payloads.xetla_format<uint32_t, num_block, 16>();
230 uint32_t base_offset_y = 0;
231#pragma unroll
232 for (int i = 0; i < num_block_y; ++i) {
233 constexpr uint32_t load_block_elems = block_elems * arr_len;
234 auto payload_row = payload_2d.xetla_select<num_block_x, 1, 16, 1>(
235 i * num_block_x, 0);
236 detail::reset_tile_desc_core<num_block_x, block_size_x, ld_blk_size_y,
237 scale_factor, arr_len, mem_transpose>(payload_row);
238#pragma unroll
239 for (int j = 0; j < num_block_x; j += arr_len) {
240 xetla_tdescriptor tdesc = payload_row.row(j);
241 auto reg_blk = tile.reg.xetla_select<load_block_elems, 1>(
242 (i * num_block_x + j) * block_elems);
243 constexpr uint32_t ld_blk_height = (reg_transpose && trans)
244 ? detail::getNextPowerOf2<ld_blk_size_y>()
245 : ld_blk_size_y;
246 constexpr uint32_t tmp_size
247 = ld_blk_height * block_size_x * arr_len;
248 xetla_vector<dtype, tmp_size> reg_tmp;
249#pragma unroll
250 for (int ii = 0; ii < block_size_y / ld_blk_size_y; ++ii) {
251 constexpr uint32_t load_elems
252 = ld_blk_size_y * block_size_x * arr_len;
253
254 reg_tmp.xetla_format<native_type_t<load_dtype>>()
255 = xetla_tload_global<load_dtype,
256 ld_blk_height * block_size_x * arr_len
257 / scale_factor,
258 L1, L3, trans, mem_transform>(tdesc);
259
260 if constexpr (reg_transpose && trans) {
261 reg_blk.xetla_select<load_elems, 1>(ii * load_elems)
262 .xetla_format<native_type_t<load_dtype>>()
263 = reg_tmp.xetla_format<load_dtype,
264 block_size_x / scale_factor,
265 ld_blk_height>()
266 .xetla_select<block_size_x / scale_factor,
267 1, ld_blk_size_y, 1>(0, 0);
268 } else {
269 reg_blk.xetla_select<tmp_size, 1>(ii * tmp_size) = reg_tmp;
270 }
271
272 if constexpr (mem_transpose) {
273 xetla_update_tdesc_offsetx(tdesc.xetla_format<uint32_t>(),
274 ld_blk_size_y / scale_factor);
275 } else {
276 xetla_update_tdesc_offsety(
277 tdesc.xetla_format<uint32_t>(), ld_blk_size_y);
278 }
279 }
280 // exceed HW limitation
281 if constexpr (block_size_y % ld_blk_size_y != 0) {
282 constexpr uint32_t remained_start_y
283 = block_size_y / ld_blk_size_y * ld_blk_size_y;
284 constexpr uint32_t remained_start
285 = remained_start_y * block_size_x * arr_len;
286 constexpr uint32_t remained_blk_size_y
287 = block_size_y % ld_blk_size_y;
288 constexpr uint32_t load_elems
289 = remained_blk_size_y * block_size_x * arr_len;
290
291 constexpr uint8_t block_width = mem_transpose
292 ? (remained_blk_size_y / scale_factor)
293 : block_size_x;
294 constexpr uint8_t block_height
295 = trans ? block_size_x : remained_blk_size_y;
296 constexpr uint32_t block_widthx_widthy_arrlen
297 = (block_width - 1) | ((block_height - 1) << 8);
298 gpu::xetla::detail::xetla_set_block_widthx_widthy_arrlen(
299 tdesc.xetla_format<uint32_t>(),
300 block_widthx_widthy_arrlen);
301
302 reg_blk.xetla_select<load_elems, 1>(remained_start)
303 .xetla_format<native_type_t<load_dtype>>()
304 = xetla_tload_global<load_dtype,
305 (load_elems / scale_factor), L1, L3, trans,
306 mem_transform>(tdesc);
307 }
308 }
309 }
310 // process tail
311 if constexpr (remained_size_y > 0) {
312 constexpr uint32_t remained_block_elems
313 = block_size_x * remained_size_y;
314 constexpr uint32_t processed_elems
315 = num_block_y * num_block_x * block_elems;
316 constexpr uint32_t remained_ld_blk_size_y
317 = (!reg_transpose && (remained_size_y > ld_blk_size_y_limit))
318 ? ld_blk_size_y_limit
319 : remained_size_y;
320 auto payload_row = payload_2d.xetla_select<num_block_x, 1, 16, 1>(
321 num_block_y * num_block_x, 0);
322 detail::reset_tile_desc_core<num_block_x, block_size_x,
323 remained_ld_blk_size_y, scale_factor, arr_len, mem_transpose>(
324 payload_row);
325#pragma unroll
326 for (int j = 0; j < num_block_x; j += arr_len) {
327 xetla_tdescriptor tdesc = payload_row.row(j);
328 auto reg_blk
329 = tile.reg.xetla_select<remained_block_elems * arr_len, 1>(
330 processed_elems + j * remained_block_elems);
331 constexpr uint32_t ld_blk_height = (reg_transpose && trans)
332 ? detail::getNextPowerOf2<remained_ld_blk_size_y>()
333 : remained_ld_blk_size_y;
334 constexpr uint32_t tmp_size
335 = ld_blk_height * block_size_x * arr_len;
336 xetla_vector<dtype, tmp_size> reg_tmp;
337#pragma unroll
338 for (int ii = 0; ii < remained_size_y / remained_ld_blk_size_y;
339 ++ii) {
340 constexpr uint32_t load_elems
341 = remained_ld_blk_size_y * block_size_x * arr_len;
342
343 reg_tmp.xetla_format<native_type_t<load_dtype>>()
344 = xetla_tload_global<load_dtype,
345 (ld_blk_height * block_size_x * arr_len
346 / scale_factor),
347 L1, L3, trans, mem_transform>(tdesc);
348
349 if constexpr (reg_transpose && trans) {
350 reg_blk.xetla_select<load_elems, 1>(ii * load_elems)
351 .xetla_format<native_type_t<load_dtype>>()
352 = reg_tmp.xetla_format<load_dtype,
353 block_size_x / scale_factor,
354 ld_blk_height>()
355 .xetla_select<block_size_x / scale_factor,
356 1, remained_ld_blk_size_y, 1>(
357 0, 0);
358 } else {
359 reg_blk.xetla_select<tmp_size, 1>(ii * tmp_size) = reg_tmp;
360 }
361 if constexpr (mem_transpose) {
362 xetla_update_tdesc_offsetx(tdesc.xetla_format<uint32_t>(),
363 remained_ld_blk_size_y / scale_factor);
364 } else {
365 xetla_update_tdesc_offsety(tdesc.xetla_format<uint32_t>(),
366 remained_ld_blk_size_y);
367 }
368 }
369 constexpr uint32_t final_ld_blk_size_y
370 = remained_size_y % remained_ld_blk_size_y;
371 if constexpr (final_ld_blk_size_y != 0) {
372 constexpr uint32_t final_start = remained_size_y
373 / remained_ld_blk_size_y * remained_ld_blk_size_y
374 * block_size_x * arr_len;
375 constexpr uint32_t final_load_elems
376 = final_ld_blk_size_y * block_size_x * arr_len;
377 constexpr uint8_t block_width = mem_transpose
378 ? (final_ld_blk_size_y / scale_factor)
379 : block_size_x;
380 constexpr uint8_t block_height
381 = trans ? block_size_x : final_ld_blk_size_y;
382 constexpr uint32_t block_widthx_widthy_arrlen
383 = (block_width - 1) | ((block_height - 1) << 8);
384 gpu::xetla::detail::xetla_set_block_widthx_widthy_arrlen(
385 tdesc.xetla_format<uint32_t>(),
386 block_widthx_widthy_arrlen);
387 reg_blk.xetla_select<final_load_elems, 1>(final_start)
388 .xetla_format<native_type_t<load_dtype>>()
389 = xetla_tload_global<load_dtype,
390 final_load_elems / scale_factor, L1, L3, trans,
391 mem_transform>(tdesc);
392 }
393 }
394 }
395
396 if constexpr (is_vnni_reverse) {
397 SW_BARRIER();
398 vnni_reverse(tile);
399 }

413template <cache_hint L1 = cache_hint::cached,
414 cache_hint L3 = cache_hint::cached, typename tile_t, typename payload_t>
415__XETLA_API typename std::enable_if_t<
416 detail::check_load_type<tile_t, payload_t>::is_global_block_1d_xe>
417tile_load(tile_t &tile, payload_t &payload) {
418 detail::check_load_condition<tile_t, payload_t>();
419 using dtype = typename tile_t::dtype;
420 using load_dtype = typename payload_t::mem_dtype;
421 static constexpr uint32_t tile_size_x = tile_t::tile_size_x;
422 static constexpr uint32_t scale_factor = payload_t::scale_factor;
423
424 constexpr uint32_t load_len = tile_size_x / scale_factor;
425 if constexpr (load_len >= 64) {
426#pragma unroll
427 for (int i = 0; i < load_len / 64; i++) {
428 uint32_t offset_x = i * 64 * scale_factor;
429 auto reg_sub
430 = tile.reg.xetla_select<64 * scale_factor, 1>(offset_x);
431 uint32_t address_offset = offset_x * sizeof(dtype);
432 reg_sub.xetla_format<load_dtype>() = xetla_load_global<load_dtype,
433 64, data_size::default_size, L1, L3>(
434 payload.base_ptr, payload.base_offset + address_offset);
435 }
436 }
437 constexpr uint32_t tail_len = load_len % 64;
438 uint32_t tail_offset = load_len / 64 * 64 * scale_factor;
439 detail::process_1d_tail<tail_len, 32, detail::process_flag::load, L1, L3>(
440 tile, payload, tail_offset);
441}

455template <cache_hint L1 = cache_hint::cached,
456 cache_hint L3 = cache_hint::cached, typename tile_t, typename payload_t>
457__XETLA_API typename std::enable_if_t<
458 detail::check_load_type<tile_t, payload_t>::is_local_scatter_xe>
459tile_load(tile_t &tile, payload_t &payload) {
460 detail::check_load_condition<tile_t, payload_t>();
461 using dtype = typename payload_t::dtype;
462 using tile_desc = typename payload_t::tile_desc;
463 using load_dtype = typename payload_t::mem_dtype;
464 constexpr uint32_t num_channel_y = payload_t::num_channel_y;
465 constexpr uint32_t load_elems = num_channel_y * tile_desc::block_size_x;
466#pragma unroll
467 for (int i = 0; i < tile_desc::tile_size_y / tile_desc::block_size_y; i++) {
468 uint32_t offset_y = i * tile_desc::block_size_y;
469#pragma unroll
470 for (int j = 0; j < tile_desc::num_block_x; j++) {
471 uint32_t offset_x = j * tile_desc::block_size_x;
472 auto reg_sub = tile.reg.xetla_select<tile_desc::block_elems, 1>(
473 (i * tile_desc::num_block_x + j) * tile_desc::block_elems);
474#pragma unroll
475 for (int sub_block_y = 0; sub_block_y < tile_desc::block_size_y;
476 sub_block_y += num_channel_y) {
477 uint32_t address_offset = offset_x * sizeof(dtype)
478 + (sub_block_y + offset_y) * payload.pitch_in_bytes;
479 reg_sub.xetla_select<load_elems, 1>(
480 sub_block_y * tile_desc::block_size_x)
481 .xetla_format<load_dtype>()
482 = xetla_load_local<load_dtype>(
483 payload.address + address_offset);
484 }
485 }
486 }
487 //process the tail
488 if constexpr ((tile_desc::tile_size_y % tile_desc::block_size_y) != 0) {
489 constexpr uint32_t remained_size_y = tile_desc::remained_size_y;
490 constexpr uint32_t offset_y = tile_desc::tile_size_y - remained_size_y;
491 constexpr uint32_t processed_elems = offset_y * tile_desc::tile_size_x;
492 constexpr uint32_t remain_block_elems
493 = remained_size_y * tile_desc::block_size_x;
494#pragma unroll
495 for (int j = 0; j < tile_desc::num_block_x; j++) {
496 uint32_t offset_x = j * tile_desc::block_size_x;
497 auto reg_sub = tile.reg.xetla_select<remain_block_elems, 1>(
498 processed_elems + j * remain_block_elems);
499#pragma unroll
500 for (int sub_block_y = 0; sub_block_y < remained_size_y;
501 sub_block_y += num_channel_y) {
502 uint32_t address_offset = offset_x * sizeof(dtype)
503 + (sub_block_y + offset_y) * payload.pitch_in_bytes;
504 reg_sub.xetla_select<load_elems, 1>(
505 sub_block_y * tile_desc::block_size_x)
506 .xetla_format<load_dtype>()
507 = xetla_load_local<load_dtype>(
508 payload.address + address_offset);
509 }
510 }
