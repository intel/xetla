42template <typename op, typename matAcc_t>
43inline void tile_broadcast_op(matAcc_t &matAcc,
44 xetla_vector<typename matAcc_t::dtype, matAcc_t::tile_size_y> data) {
45 static constexpr uint32_t tile_size_y = matAcc_t::tile_size_y;
46 static constexpr uint32_t tile_size_x = matAcc_t::tile_size_x;
47 static constexpr uint32_t tile_elems = matAcc_t::tile_elems;
48 static constexpr uint32_t block_size_y = matAcc_t::block_size_y;
49 static constexpr uint32_t block_size_x = matAcc_t::block_size_x;
50 static constexpr uint32_t block_elems = matAcc_t::block_elems;
51 static constexpr int32_t num_block_y = matAcc_t::num_block_y;
52 static constexpr int32_t num_block_x = matAcc_t::num_block_x;
53 using dtype = typename matAcc_t::dtype;
54#pragma unroll
55 for (int i = 0; i < tile_size_y / block_size_y; i++) {
56#pragma unroll
57 for (int j = 0; j < num_block_x; j++) {
58 auto acc_reg = (matAcc.reg)
59 .xetla_select<block_elems, 1>(
60 (i * num_block_x + j) * block_elems);
61 auto acc_reg_2d
62 = acc_reg.xetla_format<dtype, block_size_y, block_size_x>();
63#pragma unroll
64 for (int row_i = 0; row_i < block_size_y; row_i++) {
65 acc_reg_2d.row(row_i) = op::template func<dtype, block_size_x>(
66 acc_reg_2d.row(row_i), data[block_size_y * i + row_i]);
67 }
68 }
69 }
70 // process the tail
71 if constexpr ((tile_size_y % block_size_y) != 0) {
72 constexpr uint32_t tail_start_y
73 = tile_size_y / block_size_y * block_size_y;
74 constexpr uint32_t tail_size_y = tile_size_y % block_size_y;
75 constexpr uint32_t tail_block_elems = tail_size_y * block_size_x;
76#pragma unroll
77 for (int j = 0; j < num_block_x; j++) {
78 auto acc_reg = (matAcc.reg)
79 .xetla_select<tail_block_elems, 1>(
80 tail_start_y * tile_size_x
81 + j * tail_block_elems);
82 auto acc_reg_2d
83 = acc_reg.xetla_format<dtype, tail_size_y, block_size_x>();
84#pragma unroll
85 for (int row_i = 0; row_i < tail_size_y; row_i++) {
86 acc_reg_2d.row(row_i) = op::template func<dtype, block_size_x>(
87 acc_reg_2d.row(row_i), data[tail_start_y + row_i]);
88 }
89 }
90 }
91}
