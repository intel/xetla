322 __XETLA_API KERNEL_FUNC void operator()(xetla_exec_item<3> &ei,
323 const arguments_t &args, uint32_t slm_base = 0,
324 uint32_t nbarrier_base = 0) {
325 // set up workgroup level coordinates and boundaries
326 work_group_t g(ei.get_local_linear_id() % work_group_size);
327 uint32_t wg_id = ei.get_local_linear_id() / work_group_size;
328 int start_n = ei.get_group(2) * wg_tile_n;
329 int start_m = ei.get_group(1) * wg_tile_m;
330 int start_k = 0;
331 uint32_t wg_tile_k = args.matrix_k;
332 uint32_t boundary_n = (start_n + wg_tile_n) > args.matrix_n
333 ? args.matrix_n
334 : (start_n + wg_tile_n);
335 uint32_t boundary_m = (start_m + wg_tile_m) > args.matrix_m
336 ? args.matrix_m
337 : (start_m + wg_tile_m);
338 uint32_t boundary_k = wg_tile_k;
339 if constexpr (global_kslicing_ratio > 1) {
340 wg_tile_k = (wg_tile_k + global_kslicing_ratio - 1)
341 / global_kslicing_ratio;
342 start_k = start_k + ei.get_group(0) * wg_tile_k;
343 boundary_k = (start_k + wg_tile_k) > boundary_k
344 ? boundary_k
345 : (start_k + wg_tile_k);
346 }
347 if constexpr (local_kslicing_ratio > 1) {
348 wg_tile_k = (wg_tile_k + local_kslicing_ratio - 1)
349 / local_kslicing_ratio;
350 start_k = start_k + wg_id * wg_tile_k;
351 boundary_k = (start_k + wg_tile_k) > boundary_k
352 ? boundary_k
353 : (start_k + wg_tile_k);
354 }
355
356 // set up arguments
357 uint32_t brgemm_slm_base = slm_base;
358 uint32_t brgemm_nbarr_base = nbarrier_base;
359 if constexpr (local_kslicing_ratio > 1) {
360 brgemm_slm_base = slm_base + wg_id * brgemm_slm_size;
361 brgemm_nbarr_base = nbarrier_base + wg_id * brgemm_nbarr_count;
362 }
363 uint32_t kslicing_slm_base
364 = slm_base + local_kslicing_ratio * brgemm_slm_size;
365 uint32_t kslicing_nbarr_base
366 = nbarrier_base + local_kslicing_ratio * brgemm_nbarr_count;
367 uint32_t epilogue_slm_base = kslicing_slm_base + kslicing_slm_size;
368 uint32_t epilogue_nbarr_base
369 = kslicing_nbarr_base + kslicing_nbarr_count;
370
371 mem_desc_a_t mem_desc_a;
372 mem_desc_b_t mem_desc_b;
373 mem_desc_c_t mem_desc_c;
374 //setup for matA
375 if constexpr (mem_desc_a_t::is_local) {
376 mem_desc_a.init(args.matA_base,
377 {wg_tile_k, real_wg_tile_m, wg_tile_k}, {0, 0});
378 } else {
379 mem_desc_a.init(args.matA_base,
380 {boundary_k, boundary_m, args.matA_ld}, {start_k, start_m});
381 }
382 //setup for matB
383 if constexpr (mem_desc_b_t::is_local) {
384 mem_desc_b.init(args.matB_base,
385 {real_wg_tile_n, wg_tile_k, real_wg_tile_n}, {0, 0});
386 } else {
387 mem_desc_b.init(args.matB_base,
388 {boundary_n, boundary_k, args.matB_ld}, {start_n, start_k});
389 }
390
391 uint32_t inner_loop_count = (wg_tile_k + k_stride - 1) / k_stride;
392 brgemm_args_t brgemm_args(mem_desc_a, mem_desc_b, inner_loop_count);
393 matAcc_t matAcc;
394 matAcc.init(0);
395 brgemm_t brgemm;
396 brgemm(g, matAcc, brgemm_args, brgemm_slm_base, brgemm_nbarr_base);
397
398 kslicing_t kslicing(wg_id);
399 mat_slice_t mat_slice;
400 kslicing(g, mat_slice, matAcc, kslicing_slm_base, kslicing_nbarr_base);
401
402 if (kslicing.coop_id_x < kslicing_t::coop_num_x) {
403 //setup for matC
404 //set up cooperative offset for matC store
405 int32_t coop_offset_n
406 = kslicing.coop_id_x * mat_slice_t::tile_size_x;
407 int32_t coop_offset_m
408 = kslicing.coop_id_y * mat_slice_t::tile_size_y;
409 if constexpr (mem_desc_c_t::is_local) {
410 mem_desc_c.init(args.matC_base,
411 {real_wg_tile_n, real_wg_tile_m, real_wg_tile_n},
412 {coop_offset_n, coop_offset_m});
413 } else {
414 mem_desc_c.init(args.matC_base,
415 {boundary_n, boundary_m, args.matC_ld},
416 {start_n + coop_offset_n, start_m + coop_offset_m});
417 }
418 epilogue_t epilogue;
419 epilogue(g, mat_slice, mem_desc_c, args.epilogue_args,
420 epilogue_slm_base, epilogue_nbarr_base);
421 }
422 }
