82 void get_sk_workgroups(int &sk_groups,
83 int &savings_iters,
84 int sk_tiles, int iters_per_tile, int avail_xecores,
85 bool allow_partial_wave) const {
86
87 savings_iters = INT_MIN;
88 sk_groups = 0;
89
90 if (sk_tiles == 0) { return; }
91
92 int sk_iters = sk_tiles * iters_per_tile;
93
94 int dp_equiv_waves = (sk_tiles + avail_xecores - 1) / avail_xecores;
95 int dp_equiv_iters = iters_per_tile * dp_equiv_waves;
96
97 int min_sk_groups = (allow_partial_wave)
98 ? std::min(avail_xecores, sk_tiles + 1)
99 : avail_xecores;
100 int max_sk_groups
101 = std::min(avail_xecores, sk_iters / kMinItersPerSkGroup);
102
103 for (int trial_sk_groups = min_sk_groups;
104 trial_sk_groups <= max_sk_groups; trial_sk_groups++) {
105
106 int sk_waves
107 = (trial_sk_groups + avail_xecores - 1) / avail_xecores;
108 int max_sk_iters_per_group
109 = (sk_iters + trial_sk_groups - 1) / trial_sk_groups;
110 int sk_iter_equiv = max_sk_iters_per_group * sk_waves;
111
112 int num_peers = ((trial_sk_groups + sk_tiles - 1) / sk_tiles) + 1;
113 float iter_cost = 0.02f * float(num_peers) * float(sk_iter_equiv);
114
115 if (trial_sk_groups % sk_tiles == 0) {
116
117 //aligned
118 num_peers = (trial_sk_groups / sk_tiles);
119 iter_cost = 0.0f;
120 }
121
122 float peer_cost = 2.0f * float(num_peers);
123 float base_cost = 2.0f * float(sk_waves);
124
125 int fixup_iter_equiv = int(base_cost + iter_cost + peer_cost);
126
127 int trial_savings_iter
128 = dp_equiv_iters - sk_iter_equiv - fixup_iter_equiv;
129
130 if (trial_savings_iter >= savings_iters) {
131
132 savings_iters = trial_savings_iter;
133 sk_groups = trial_sk_groups;
134 }
135 }
136 }

139 void get_groups(int &dp_tiles, int &sk_groups, int output_tiles,
140 int iters_per_tile, int avail_xecores) {
141
142 int full_waves = output_tiles / avail_xecores;
143 int full_wave_tiles = full_waves * avail_xecores;
144 int partial_wave_tiles = output_tiles - full_wave_tiles;
145
146 if (partial_wave_tiles == 0) {
147 //No tails
148 return;
149 }
150 int score = -1;
151 dp_tiles = output_tiles;
152 sk_groups = 0;
153
154 if (full_waves < 1) {
155
156 dp_tiles = full_wave_tiles;
157
158 get_sk_workgroups(sk_groups, score, partial_wave_tiles,
159 iters_per_tile, avail_xecores, true);
160
161 if (score < 0) {
162 //Not profitable
163 dp_tiles = output_tiles;
164 sk_groups = 0;
165 }
166
167 return;
168 }
169
170 //Form the SK wave by combining the last full wave and the partial wave
171 dp_tiles = full_wave_tiles - avail_xecores;
172
173 get_sk_workgroups(sk_groups, score, partial_wave_tiles + avail_xecores,
174 iters_per_tile, avail_xecores,
175 false); // cannot run with less than a full wave of SK-groups
176
177 std::cout << "SK Score: " << score << "\n\n";
178
179 if (score < 0) { //Not profitable for streamK split
180
181 sk_groups = 0;
182 dp_tiles = output_tiles;
183 }
184 }

192 inline WorkgroupSplitStreamK_t(uint32_t matrix_m_, uint32_t matrix_k_,
193 uint32_t matrix_n_, uint32_t wg_tile_m_, uint32_t wg_tile_k_,
194 uint32_t wg_tile_n_, uint32_t sg_tile_m_, uint32_t sg_tile_n_,
195 uint32_t avail_xecores_)
196 : matrix_m(matrix_m_)
197 , matrix_k(matrix_k_)
198 , matrix_n(matrix_n_)
199 , wg_tile_m(wg_tile_m_)
200 , wg_tile_k(wg_tile_k_)
201 , wg_tile_n(wg_tile_n_)
202 , sg_tile_m(sg_tile_m_)
203 , sg_tile_n(sg_tile_n_)
204 , avail_xecores(avail_xecores_) {
205
206 int iters_per_tile = (matrix_k + wg_tile_k - 1) / wg_tile_k;
207
208 //Default values for sk parameters
209 int sk_iters_per_normal_group = 0;
210 int sk_iters_per_big_group = 0;
211 sk_regions
212 = 1; // Default : a single region of iteration space across all SK tiles
213 sk_groups_per_region = 1;
214 sk_waves = 0;
215
216 int num_tiles_m = (matrix_m + wg_tile_m - 1) / wg_tile_m;
217 int num_tiles_n = (matrix_n + wg_tile_n - 1) / wg_tile_n;
218
219 int output_tiles = num_tiles_m * num_tiles_n;
220 int waves = (output_tiles + avail_xecores - 1) / avail_xecores;
221 float dp_efficiency
222 = float(output_tiles) / float(waves * avail_xecores);
223
224 int dp_tiles = output_tiles;
225 int sk_groups = 0;
226
227 //Use heuristics to get streamK split
228 get_groups(dp_tiles, sk_groups, output_tiles, iters_per_tile,
229 avail_xecores);
230
231 sk_tiles = output_tiles - dp_tiles;
232
233 // Compute SK group iteration details
234 if (sk_groups > 0) {
235
236 sk_waves = (sk_groups + avail_xecores - 1) / avail_xecores;
237 //Compute global iteration space - tiles_m*tiles_n*k_iters
238 int sk_iters = sk_tiles * iters_per_tile;
239 sk_groups = std::min(sk_groups, sk_iters);
240
241 //sk_iters may not divide sk_groups evenly; some groups perform one additional iteration
242 sk_iters_per_normal_group = sk_iters / sk_groups;
243 int extra_sk_iters
244 = sk_iters - (sk_iters_per_normal_group * sk_groups);
245 int sk_big_groups = extra_sk_iters;
246 sk_iters_per_big_group = sk_iters_per_normal_group + 1;
247
248 //KSlicing to fill up multiple regions within groups
249 if ((sk_groups > sk_tiles) && (sk_groups % sk_tiles == 0)) {
250
251 sk_regions = sk_tiles;
252 }
253
254 sk_groups_per_region = sk_groups / sk_regions;
255 sk_big_groups_per_region = sk_big_groups / sk_regions;
256 sk_iters_per_region = sk_iters / sk_regions;
257
258 //Initialize fast divmod counters related to SK
259 div_mod_sk_regions = FastDivMod(sk_regions);
260 div_mod_sk_groups_per_region = FastDivMod(sk_groups_per_region);
261 div_mod_sk_iters_per_normal_group
262 = FastDivMod(sk_iters_per_normal_group);
263 div_mod_sk_iters_per_big_group = FastDivMod(sk_iters_per_big_group);
264 }
265
266 div_mod_tiles_m = FastDivMod(num_tiles_m);
267 div_mod_tiles_n = FastDivMod(num_tiles_n);
268 div_mod_iters_per_tile = FastDivMod(iters_per_tile);
269
270 dp_groups = dp_tiles;
271 num_workgroups = get_num_active_groups();
272
273 //Print the stats
274 uint32_t total_tiles = num_tiles_m * num_tiles_n;
275 std::cout << " problem size: (" << matrix_m << "," << matrix_n << ")"
276 << ", tiled_shape: (" << num_tiles_m << "," << num_tiles_n
277 << ")"
278 << ", tiles: " << total_tiles
279 << ", dp_tiles: " << total_tiles - sk_tiles
280 << ", sk_tiles: " << sk_tiles
281 << ", iters_per_tile: " << iters_per_tile
282 << ", num_workgroups: " << num_workgroups
283 << ", dp_workgroups: " << dp_groups
284 << ", dp_waves: " << dp_groups / avail_xecores
285 << ", sk_groups_per_region: " << sk_groups_per_region
286 << ", sk_regions: " << sk_regions
287 << ", sk_waves: " << sk_waves
288 << ", sk_iters_per_normal_group: "
289 << sk_iters_per_normal_group
290 << ", sk_big_groups_per_region: " << sk_big_groups_per_region
291 << ", avail_xecores: " << avail_xecores << "\n\n";
292 }

344 __XETLA_API KERNEL_FUNC void get_iter_extents(int sk_group_idx,
345 int &group_iter_begin, int &group_iter_end) const {
346 int region_idx;
347 int group_idx_in_region;
348 div_mod_sk_groups_per_region.fast_divmod(
349 region_idx, group_idx_in_region, sk_group_idx);
350
351 group_iter_begin = (region_idx * sk_iters_per_region)
352 + (group_idx_in_region * get_sk_iters_per_normal_group());
353
354 //Adjust extents for the first num_big_group groups that get one extra iteration
355 int group_iters = get_sk_iters_per_normal_group();
356 if (group_idx_in_region < sk_big_groups_per_region) {
357
358 group_iter_begin += group_idx_in_region;
359 group_iters += 1;
360 } else {
361
362 //This is a regular group
363 group_iter_begin += sk_big_groups_per_region;
364 }
365
366 group_iter_end = group_iter_begin + group_iters;
367 }

370 __XETLA_API KERNEL_FUNC int get_first_group_idx(
371 int tile_idx, int group_idx) const {
372
373 if (tile_idx >= sk_tiles) {
374 //DP group
375 return group_idx;
376 }
377
378 int iter = tile_idx * get_iters_per_tile();
379
380 int region_idx, iter_in_region;
381
382 div_mod_sk_iters_per_region.fast_divmod(
383 region_idx, iter_in_region, iter);
384
385 //Number of iterations in the big group region
386 int big_group_iters
387 = sk_big_groups_per_region * get_sk_iters_per_normal_group()
388 + sk_big_groups_per_region;
389
390 //Number of iterations in the normal group region
391 int normal_group_iters = iter_in_region - big_group_iters;
392
393 int big_group_idx_in_region
394 = div_mod_sk_iters_per_big_group.div(iter_in_region);
395
396 int normal_group_idx_in_region = sk_big_groups_per_region
397 + div_mod_sk_iters_per_normal_group.div(normal_group_iters);
398
399 int group_idx_in_region
400 = (big_group_idx_in_region < sk_big_groups_per_region)
401 ? big_group_idx_in_region
402 : normal_group_idx_in_region;
403
404 int owning_group_idx = (get_sk_groups_per_region() * region_idx)
405 + group_idx_in_region;
406
407 return owning_group_idx;
408 }
