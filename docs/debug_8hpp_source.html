30using element_type = uint16_t;
31static constexpr size_t element_num = 8;
32static constexpr size_t max_dims = 3;
33static constexpr size_t dims_pos = 0;
34static constexpr size_t dims_global_start = 1;
35static constexpr size_t dims_local_start = 1 + max_dims;
36
37static constexpr size_t nd_item_offset
38 = reg_start - element_num * sizeof(element_type);
39static inline ESIMD_PRIVATE ESIMD_REGISTER(nd_item_offset)
40 __ESIMD_NS::simd<element_type, element_num> saved_nd_item;
41
42template <size_t dims>
43static inline void set(sycl::nd_item<dims> item) {
44 static_assert(dims <= max_dims);
45
46 saved_nd_item[dims_pos] = dims;
47
48#pragma unroll
49 for (auto i = 0; i < dims; i++) {
50 saved_nd_item[dims_global_start + i] = item.get_group(i);
51 }
52
53#pragma unroll
54 for (auto i = 0; i < dims; i++) {
55 saved_nd_item[dims_local_start + i] = item.get_local_id(i);
56 }
57}
58
59static inline uint16_t get_dims() {
60 return saved_nd_item[dims_pos];
61}
62
63static inline int16_t get_group_id(size_t dim) {
64 return saved_nd_item[dims_global_start + dim];
65}
66
67static inline int16_t get_local_id(size_t dim) {
68 return saved_nd_item[dims_local_start + dim];
69}
