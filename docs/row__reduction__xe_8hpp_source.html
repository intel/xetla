52 static constexpr uint32_t wg_tile_n = reduction_attr::wg_tile_n;
53 static constexpr uint32_t sg_tile_m = reduction_attr::sg_tile_m;
54 static constexpr uint32_t sg_tile_n = reduction_attr::sg_tile_n;

64 static constexpr uint32_t max_load_height_in_elem

67 = load_store_attr::max_load_width_in_bytes;
68 static constexpr uint32_t max_store_width_in_bytes
69 = load_store_attr::max_store_width_in_bytes;
70 static constexpr uint32_t max_load_width_in_elem
71 = max_load_width_in_bytes / sizeof(dtype_in);
72 static constexpr uint32_t max_store_width_in_elem
73 = max_store_width_in_bytes / sizeof(dtype_out);
74
75 static constexpr uint32_t tile_size_x = sg_tile_n;
76 static constexpr uint32_t tile_size_y = sg_tile_m;

78 static constexpr uint32_t max_simd_len = max_store_width_in_elem;

81 static constexpr uint32_t block_size_x
82 = max_load_width_in_elem > tile_size_x
83 ? tile_size_x
84 : gpu::xetla::subgroup::detail::gcd<tile_size_x,
85 max_load_width_in_elem>::value;
86 static_assert(block_size_x >= 8,
87 "if block_size_x less than 8, the efficiency will be low. Please "
88 "choose another tile_size_x");
89 static constexpr uint32_t block_size_y

95 using global_ld_tile_desc_t = subgroup::tile_desc_t<tile_size_x,

146 __XETLA_API static void call(xetla_exec_item<3> &ei, arguments_t *args,
147 fused_op_arguments_t *fused_op_args = nullptr,
148 uint32_t slm_base = 0, uint32_t nbarrier_base = 0) {
149 work_group_t g;
150 g.init(ei.get_local_linear_id());
151 int sg_idx = g.get_id() % wg_size_x;
152 int sg_idy = g.get_id() / wg_size_x;
153
154 int global_start_x_in
155 = ei.get_group(2) * wg_tile_n + sg_idx * sg_tile_n;
156 int global_start_y_in = sg_idy * sg_tile_m;
157 xetla_nbarrier_t<wg_size_y, wg_size_y> nbarrier;
158 nbarrier.init_nbarrier(
159 nbarrier_base + sg_idx, nbarrier_role::producer_consumer);
160 if constexpr (use_dynamic_job) {
162 xetla_vector<uint32_t, SIMD> offsets(
163 slm_base + row_buffer_size + sg_idx * SIMD * sizeof(int));
164 xetla_mask<SIMD> pred(0);
165 pred[0] = 1;
166 if (sg_idy == 0) {
167 xetla_vector<int, SIMD> init(wg_size_y);
168 xetla_store_local<int, 1, data_size::default_size, SIMD>(
169 offsets, init, pred);
170 xetla_fence<memory_kind::shared_local>();
171 }
172 nbarrier.arrive();
173 }
174
175 global_ld_t mat_global_ld;
176 fused_op_t fused_op(
177 fused_op_args, global_start_x_in, global_start_y_in);
178 global_ld_payload_t mat_global_ld_payload(args->mat_in_ptr,
179 args->matrix_n, args->matrix_m, args->mat_in_ld,
180 global_start_x_in, global_start_y_in);
181 mat_buffer_t mat_buffer(0);
182 if constexpr (use_dynamic_job) {
183 nbarrier.wait();
184 int job_id = sg_idy;
185 xetla_vector<uint32_t, SIMD> offsets(
186 slm_base + row_buffer_size + sg_idx * SIMD * sizeof(int));
187 xetla_mask<SIMD> pred(0);
188 pred[0] = 1;
189 while (job_id * tile_size_y < args->matrix_m) {
190 xetla_vector<int, SIMD> next_job
191 = xetla_atomic_local<atomic_op::iinc, int, SIMD>(
192 offsets, pred);
193 subgroup::tile_load(mat_global_ld, mat_global_ld_payload);
194 matAcc_t matAcc;
195 subgroup::elemwise_cvt<matAcc_t, global_ld_t>(
196 matAcc, mat_global_ld);
197 fused_op(matAcc);
198 subgroup::tile_row_reduce(mat_buffer, matAcc);
199 mat_global_ld_payload
200 .template update_tdesc<tdesc_update_dir::y_dir>(
201 (next_job[0] - job_id) * tile_size_y);
202 fused_op.update_tdesc(0, (next_job[0] - job_id) * tile_size_y);
203 job_id = next_job[0];
204 }
205 } else {
206 for (int job_id = sg_idy; job_id * tile_size_y < args->matrix_m;
207 job_id += wg_size_y) {
208 subgroup::tile_load(mat_global_ld, mat_global_ld_payload);
209 matAcc_t matAcc;
210 subgroup::elemwise_cvt<matAcc_t, global_ld_t>(
211 matAcc, mat_global_ld);
212 fused_op(matAcc);
213 subgroup::tile_row_reduce(mat_buffer, matAcc);
214 fused_op.update_tdesc(0, wg_size_y * tile_size_y);
215 mat_global_ld_payload
216 .template update_tdesc<tdesc_update_dir::y_dir>(
217 wg_size_y * tile_size_y);
218 }
219 }
220
221 row_reduce_store_t row_reduce_store;
222 uint32_t slm_row_reduce_base = slm_base;
223 uint32_t nbarrier_row_reduce_base = nbarrier_base;
224 row_reduce_store.init(
225 sg_idx, sg_idy, slm_row_reduce_base, nbarrier_row_reduce_base);
226 row_reduce_store(args->mat_out_ptr, args->matrix_n, 1, args->matrix_n,
227 global_start_x_in, 0, mat_buffer.reg);
228 }
